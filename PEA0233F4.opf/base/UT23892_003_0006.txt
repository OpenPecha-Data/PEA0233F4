༄༅། །སྒྲུབ་བརྒྱུད་རིན་པོ་ཆེའི་ཁྱད་ཆོས་ལྟ་སྒོམ་སྤྱོད་པའི་གནད་ཀྱི་དྲིས་ལན་ཞེས་བྱ་བ་བཞུགས་སོ།། 

།བླ་མ་རྣམས་དང་ཆོས་བདག་མཁའ་འགྲོ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །སྒྲུབ་བརྒྱུད་ཅེས་པ་འདི་ལ། དོན་མང་དུ་འདུག་སྟེ། དེ་ཡང་སྤྱིར་རྒྱུ་མི་དགེ་སྒྲུབ་པའང་སྒྲུབ་བརྒྱུད་ཡིན་ལ། རྒྱུ་དགེ་བ་ཟག་པ་དང་བཅས་པ་སྒྲུབ་པ་དང༌། བདག་དང་བདག་མེད་པ་སྒྲུབ་པ་སོགས་ཀྱང་སྒྲུབ་བརྒྱུད་ཡིན། དེ་རྣམས་ནི་སྒྲུབ་བརྒྱུད་ཐུན་མོང་པ་ཡིན་ལ། ཐུན་མོང་མ་ཡིན་པའི་སྒྲུབ་བརྒྱུད་ནི། རྗེ་བཙུན་བྱམས་པས། སྟོན་པའི་དམ་ཆོས་ཞེས་སོགས་གསུངས། ཕ་རོལ་དུ་ཕྱིན་པ་བས། བསྐལ་པ་གྲངས་མེད་གསུམ་དུ་ཚོགས་གཉིས་བསགས་ནས་སངས་རྒྱས་སྒྲུབ་པ་ཡང་སྒྲུབ་བརྒྱུད་ཡིན་ལ། ད་ལན་འདིར་ངེད་རང་སྒྲུབ་བརྒྱུད་རིན་པོ་ཆེ་ནི་ཡུལ་དེ་རྣམས་ལ་མ་ལྟོས་པར་ཚེ་གཅིག་གམ་ལུས་གཅིག་གམ་སྐད་ཅིག་ལ་སེམས་ཅན་དོན་དུ་རྡོ་རྗེ་འཆང་སྒྲུབ་པའི་སྒྲུབ་བརྒྱུད་ཡིན་ཏེ། མཚན་བརྗོད་ལས། སྐད་ཅིག་གཅིག་གིས་རྫོགས་སངས་རྒྱས། །ཞེས་པས། དེ་ཡང་ཤིན་ཏུ་འགྲུབ་པ་ཞིག་ཡིན་ཏེ། ཆོས་ཐམས་ཅད་བསམ་པ་སྐད་ཅིག་མའི་རང་གཟུགས་སུ་འགྲུབ་པ་ཡིན་པའི་ཕྱིར་རོ། །དེ་ཡང་གཞན་དོན་དུ་སྒྲུབ་པ་མ་གཏོགས། རང་དོན་དུ་ཆོས་སྐུ་ཡང་སྒྲུབ་པ་མ་ཡིན་ནོ། །དེ་ལྟར་བསྒྲུབ་པར་བྱ་བའི་རྡོ་རྗེ་འཆང་ནི་ཕྱི་ནང་གསང་གསུམ་དེ་ཁོ་ན་ཉིད་ཀྱི་རྡོ་རྗེ་འཆང་ལས། དེ་ཁོ་ན་ཉིད་ཀྱི་རྡོ་རྗེ་འཆང་ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས། ལྟ་བ་རྟོགས་པ་དང་ལྡན་པའམ། དཀར་པོ་ཆིག་ཐུབ་ཀྱི་དོན་ཡིན། ལྟ་བ་མངོན་དུ་བྱས་པ་ལ་རྡོ་རྗེ་འཆང་ཐོབ་པ་ཞེས་བརྗོད་དེ། དེ་མངོན་དུ་བྱེད་པའི་ཐབས་ལ་མ་ངེས་པ་སྣ་ཚོགས་པས་རྟོགས་པར་འདོད་དེ། ཆེན་པོ་རྣམ་གསུམ་དང༌། མཚན་ཉིད་དང་མི་ལྡན་པའི་བླ་མས་ཡོན་ཏན་བསྐྱེད་ནུས་ཞེས་པ་དང༌། ཐོས་བསམ་སྒོམ་གསུམ་གྱིས་རྟོགས་པར་འདོད་དེ༑ བརྒྱུད་པ་རིན་པོ་ཆེའི་བཞེད་པས། དེ་རྣམས་ཀྱིས་རྟོགས་པ་མ་ཡིན་ཏེ། ཐེག་ཆེན་རྒྱུད་བླ་མ་ལས། ཕྲ་ཕྱིར་ཐོས་པའི་ཡུལ་མིན་ཕྱིར། །དོན་དམ་ཡིན་ཕྱིར་བསམ་པས་མིན། །ཆོས་ཉིད་ཟབ་ཕྱིར་འཇིག་རྟེན་པའི། །སྒོམ་པ་ལ་སོགས་ཡུལ་མ་ཡིན། །ཞེས་དང༌།

དོ་ཧ་ལས། དགེ་ཚུལ་དགེ་སློང་གནས་བརྟན་ཞེས་བྱ་བ། །བནྡེ་རྣམས་ནི་དེ་ལྟར་རབ་བྱུང་ནས། ཁ་ཅིག་མདོ་སྡེ་འཆད་པར་བྱེད་ཅིང་འདུག །ལ་ལ་དཀྱིལ་འཁོར་འཁོར་ལོ་མ་ལུས་བསྒོམ། །གཞན་དག་བཞི་པའི་དོན་འཆད་པ་ལ་ཞུགས༑ ཁ་ཅིག་རོ་གཅིག་སེམས་ཀྱིས་ཚུལ་འཛིན་མཐོང༌། །ལ་ལ་ནམ་མཁའི་ཁམས་ལ་རྟོག་པར་སྣང༌། །ཕལ་ཆེར་མི་མཐུན་ཕྱོགས་ལ་ཞུགས་པ་ཡིན། །ཞེས་དང༌༑ བསྐྱེད་རྫོགས་ལ་བྱང་ཆུབ་མི་རེ་བ། །མར་པ་ལོ་ཙཱའི་ཁྱད་ཆོས་ཡིན། །ཡིད་བྱེད་ལ་སངས་རྒྱས་མི་རེ་བ། །དྭགས་པོ་ལྷ་རྗེའི་ཁྱད་ཆོས་ཡིན། །བག་ཆགས་ལ་ཀུན་མཁྱེན་མི་རེ་བ། །དུས་གསུམ་མཁྱེན་པའི་ཁྱད་ཆོས་ཡིན། །ཞེས་གསུངས་སོ། །འདུལ་བ་མདོ་སྡེ་མངོན་པའི་སྡེ་སྣོད་སོགས་ལུང་རིག་མཐའ་ཡས་སོ། །ལྟ་བ་རྟོགས་པ་དང་ལྡན་པ་དེ་ལ་གོམས་འདྲིས་བྱེད་པ་ནི་སྒོམ་པ་ཡིན་ལ། སྤྱོད་པ་ནི་བླང་དོར་དང་བྲལ་བ་ཚུལ་ཁྲིམས་རིན་པོ་ཆེ་ཡིན་ཏེ། དེ་ཡང་བླང་བ་སྟེ་ནག་པོ་མི་དགེ་བའི་ཕྱོགས་བླང་དུ་མེད་པ་དང༌། དོར་བ་སྟེ་དགེ་བ་དཀར་པོའི་ཕྱོགས་དོར་དུ་མེད་པ་ཡིན་ལ། དེའི་ཚུལ་ཡང༌། དངོས་པོ་ཕྲ་བ་གཅིག་ལའང༌། གཤིས་འོད་གསལ་དུ་གནས་པ་ལ། གནས་སྐབས་སུ་ཡག་སྙམ་པས་ཆགས་པ། ལོད་སྙམ་པས་སྡང་བ་སྟེ། དེ་ལ་སྤང་བླང་བྱེད་པ་ནི། ཚུལ་ཁྲིམས་ལས་འདས་པའོ། །དེས་ན༑ སྤང་བླང་མེད་པའི་ཉམས་སུ་ལེན་པ་ནི། སྤྱོད་པ་བླང་དོར་དང་བྲལ་བ་ཚུལ་ཁྲིམས་རིན་པོ་ཆེ་ཡིན་ནོ། །ལྟ་སྒོམ་སྤྱོད་གསུམ་མི་གཅིག་པར་འདོད་པ་ཡོད་ཀྱང༌༑ འདིར་ལྟ་སྒོམ་སྤྱོད་གསུམ་གཅིག་པ་ཡིན་ཏེ། དེའི་ཚུལ་ཡང་ལྟ་བ་ནི་གཉུག་མའི་གནས་ལུགས་རྟོགས་པ་དང་ལྡན་པ་དེ་ཡིན་ལ། དེ་ལ་མཉམ་པར་འཇོག་པའམ་གོམས་པར་བྱེད་པ་སྒོམ་པ། དེ་ཉིད་བྲི་གང་མེད་པར་ཉམས་སུ་ལེན་པ་སྤྱོད་པ་སྟེ། དེས་ན་ལྟ་སྒོམ་སྤྱོད་གསུམ་གཅིག་པ་སྟེ། དེ་སྐད་དུ། སྙིང་པོ་དོན་གསུམ་ལས༑ གང་ཤར་ངོས་འཛིན་ལྟ་བའི་གནད། །ཅེས་གསུངས་སོ། །ཕྱག་རྒྱ་ཆེན་པོ་དང་ཚུལ་ཁྲིམས་གཅིག་ཚུལ་ནི།

ཚུལ་ཞེས་པ་ཆོས་ཐམས་ཅད་ཀྱི་གནས་ཚུལ་ཡིན་ལ༑ དེ་ལས་མི་འདའ་བ་ནི་ཁྲིམས་ཞེས་བྱའོ། །ཕྱག་ཞེས་པས་འཁོར་འདས་ཀྱི་གནས་ལུགས་དེ་ཡིན་ལ། རྒྱ་ཞེས་པ་ནི་དེ་ལས་མི་འདའ་བའི་དོན་ཡིན་པས། དེས་ན་ཕྱག་རྒྱ་ཆེན་པོ་དང་ཚུལ་ཁྲིམས་རིན་པོ་ཆེ་གཅིག་པའི་གནད་རིག་པར་བྱའོ། ། ཞེས་པ་འདི་ཡང་ཀུན་དགའི་མིང་ཅན་བདག་གིས་བགྱིས་པའོ། །མངྒལཾ།། །།
