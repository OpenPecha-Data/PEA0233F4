པ། སློབ་དཔོན་སྟོན་པ། བསོད་དབང༌། ལོ་མཆོག དབང་ལོ། བསོད་གྲགས། གྲགས་རྒྱལ། རིན་རྒྱལ་བ། རྒྱལ་བ་ཕྱག་ན། ཆོས་ཀྱི་རྗེ་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། རྟ་མགྲིན་གསང་སྒྲུབ་སྐྱེར་སྒང་ལུགས་ཀྱི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། དཔལ་རྟ་མགྲིན། དབོན་སྟོན་སྐྱེར་པ་སྒང་པ། སངས་རྒྱས་སྟོན་པ། འཇག་ཆེན་ཤེས་རབ་ཚུལ་ཁྲིམས། འཇགས་ཆེན་རྒྱལ་མཚན་འབུམ། ཁྱུང་པོ་ཚུལ་ཁྲིམས་མགོན། རིན་པོ་ཆེ་གྲགས་ཚུལ་བ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན། མཁན་ཆེན་ཟླ་བ་བློ་གྲོས། མཁན་ཆེན་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། རྣམ་འཇོམས་བ་རི་ལུགས་ཀྱི་རྗེས་གནང་གི་བརྒྱུད་པ་ནི། སྟོན་པ་སངས་རྒྱས། གསང་བའི་བདག་པོ། ཛེ་ཏཱ་རི་པ། རྡོ་རྗེ་གདན་པ་ཆེ་ཆུང༌། བ་རི་ལོ་ཙཱ་བ།་དབུས་པ་ཟི་སྟོན། གནམ་ཁོའུ་པ། གཏུམ་སྟོན་བློ་གྲོས་གྲགས། ཤེས་རབ་དཔལ། གློ་བོ་ཤེས་རབ་རིན་ཆེན། ཆོས་རྒྱལ་བ། རོང་པ་ཤེར་སེང༌། ཀུན་དགའ་ཚུལ་ཁྲིམས། བག་སྟོན། ཚུལ་རྒྱལ་བ། བསོད་རིན་པ། ཆོས་གྲགས་པ། རིན་ཆེན་རྒྱལ་མཚན། རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་རིན་པོ་ཆེ་ཐམས་ཅད་མཁྱེན་པ་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། སྒྲོལ་མ་ལྗང་ཁུའི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས། ཛེ་ཏཱ་རི། རྡོ་རྗེ་གདན་པ་གཉིས། བ་རི་ལོ་ཙཱ་བ། ས་ཆེན་ཀུན་དགའ་སྙིང་པོ། སློབ་དཔོན་བསོད་རྩེ། རྗེ་བཙུན་གྲགས་པ་རྒྱལ་མཚན། ཆོས་རྗེ་ས་པཎ། ཆོས་རྒྱལ་འཕགས་པ། སྣུར་ཡེ་རྒྱལ། སྣུར་ཆོས་འཕགས། གར་ཐིག་པ་སངས་རྒྱལ། མཁན་ཆེན་ཤེས་རབ་རྡོ་རྗེ། ཕག་གྲུ་མཐིལ་གྱི་ཆོས་སྒོ་བ། ཕག་གྲུ་མཐིལ་གྱི་མཁན་པོ།

གྲགས་རྡོར་བ། བླ་མ་བྱང་ཆུབ་རྡོ་རྗེ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན། འཇམ་དབྱངས་ཉི་མ་རྒྱལ་མཚན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་རིན་ཆེན་དཔལ་བཟང་པོ། །དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། རྗེ་བཙུན་མི་གཡོ་བ་དཀར་པོའི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། མི་གཡོ་བ། གྲུབ་ཐོབ་འོད་བྱེད་འབྱུང་གནས། པཎ་ཆེན་ཤཱཀྱ་ཤྲཱི། པཎྜི་ཏ་དཱ་ན་ཤཱི་ལ། མཁས་པ་ཡོན་ཏན་ཁྲི། འདུལ་འཛིན་བྱང་ཆུབ་ཚུལ་ཁྲིམས། རྒྱལ་ཚ་ལུང་མང་པོ། རྒྱལ་མཚན་གཞོན་ནུ་གྲགས། མཁན་ཆེན་ཕྱོགས་ལས་རྣམ་རྒྱལ། མཁས་གྲུབ་སངས་རྒྱས་དཔལ། རོང་སྟོན་ཤེས་བྱ་ཀུན་གཟིགས། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། ཐུགས་རྗེ་ཆེན་པོ་སེང་གེ་སྒྲའི་རྗེས་གནང་གི་བརྒྱུད་པ་ནི། སྤྱན་རས་གཟིགས། སེང་ལྡིང་ནགས་ཀྱི་སྒྲོལ་མ། ཙནྡྲ་གོ་མི། བི་ཀ་ལམ་ཀ རྡོ་རྗེ་གདན་པ། ཅལ་ཆོས་བཟང་པ། དངོས་གྲུབ་དཔལ། བླ་མ་ཆོས་གྲགས་པ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན། དེ་གཉིས་ཀ་ལ་སེམས་དཔའ་ཆེན་པོ་ཟླ་བ་བློ་གྲོས་པས་གསན། དེ་ལ་མཁན་ཆེན་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས་པས་གསན། དེས་བདག་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། གང་བློ་མའི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། འཇམ་དབྱངས་དྷི་ར་པ་ལ། རྔོག་ལོ་ཁྱུང་རིན་ཆེན་གྲགས། སྟོད་ལུང་རྒྱ་དམར། ཆ་བ། གཙང་ནག་པ། དན་འབག་བ། སྨྱལ་ཞིག བཟང་རིངས། གྲ་ཐང་པ། བསོད་རྒྱལ་བ། ཡང་རྩེ་བ། བུ་སྟོན། མཁན་ཆེན་ཡོན་རྒྱལ་བ། བླ་མ་ཨ་མོ་གྷ། སློབ་དཔོན་ཀུན་དགའ་དཔལ། དེ་གཉིས་ལ་འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་བས་གསན། ཆོས་རྗེ་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ།

མི་གཡོ་བ་སྔོན་པོ་པུས་བཙུགས་ཅན་གྱི་རྗེས་གནང་གི་བརྒྱུད་པ་ནི། མི་གཡོ་བ། བྱང་སེམས་ཀུན་དགའ། སྟོན་ཡེ། བྱམས་ཆེན་པ་བསོད་འཕེལ་བ། བག་སྟོན་གཞོན་ཚུལ་བ། བླ་མ་ཚུལ་རྒྱལ་བ། མཁན་ཆེན་ཤེས་རྡོར་བ། ཕག་གྲུའི་མཁན་པོ་གྲགས་རྡོར་བ། བྱང་ཆུབ་རྡོ་རྗེ། རིན་ཆེན་རྒྱལ་མཚན། རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས། དེས་བདག་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། ཛཾ་བྷ་ལ་ནག་པོ་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་གདན་པ། བ་རི་ལོ་ཙཱ་བ། རྔོག་སྟོན། སྨྱལ་པ། གཙང་དཀར། མར་སྟོན། གཏུམ་སྟོན། ཙང་ཇོ་མོ། ཤཱཀྱ་འབུམ། ཤཱཀྱ་གཞོན་ནུ། བློ་སྟོན་ཇོ་མགོན། དེ་གསུམ་ཀ་ལ་བློ་ལྡན་པས་གསན། དེ་ལ་བླ་མ་དབང་རྡོར་བས། དེ་ལ་རིན་ཆེན་རྒྱལ་མཚན་བས། དེ་ལ་རྒྱལ་བ་ཕྱག་ནས། དེ་ལ་མཁན་པོ་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས་པས་གསན་པ་ཡིན། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། ཛཾ་བྷ་ལ་སེར་པོའི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་གདན་པ། བ་རི་ལོ་ཙཱ་བ། སློབ་དཔོན་རིན་པོ་ཆེ། བཙུན་པ་དཀོན་མཆོག་སྐྱབས། ལྕེ་མདོ་སྡེ་སེང་གེ། དེ་ཕྱིན་ནི་གོང་གི་ཛཾ་བྷ་ལ་ནག་པོའི་རྒྱུད་པ་དང་འདྲའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། སངས་རྒྱས་སྨན་བླའི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། སྟོན་པ་སངས་རྒྱས། འཇམ་དཔལ། ཤནྟ་རཀྵི་ཏ། ཁྲི་སྲོང་ལྡེའུ་བཙན། སད་ན་ལེགས། རལ་པ་ཅན། ལྷ་ལུང་དཔལ་གྱི་རྡོ་རྗེ། འོད་སྲུང༌། དཔལ་འཁོར་བཙན། ཉི་མ་མགོན། ཡེ་ཤེས་འོད། བཀྲ་ཤིས་སྡེ། འོད་སྡེ། རྩེ་སྡེ། མངའ་རིས་པ། ཁུ་ལྡན་པ། ར་ས་པ། དཔལ་ལྡན་གྲོ། སངས་རྒྱས་སྒོམ་པ། སྒྲགས་སྙིང་པོ་རྒྱལ་མཚན། ཀུན་སྨོན། གངས་པ། བསོད་གྲགས་པ། བསོད་བཟུངས་པ། གྲགས་རྒྱལ་བ། རྣམ་རྒྱལ་བ། རྒྱལ་པོ་ཕྱག་ན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།།

༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། ཕྱག་ན་རྡོ་རྗེ་འགྲོ་བཟང་གི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང༌། ཕྱག་ན་རྡོ་རྗེ། ཆོས་ཀྱི་རྡོ་རྗེ། ཀུན་ཏུ་བཟང་པོ། འགྲོ་བཟང་སྙིང་པོ། ཁ་ཆེ་རིན་ཆེན་རྡོ་རྗེ། ཟངས་དཀར་ཞུས་སྐྱས། གྲགས་སེང༌། རྒྱ་སྟོན། རྒྱ་སྟོན་གྲུབ་མཆོག རྡོ་རྗེ་དཔལ་གཞོན་ནུ། སེང་གེ བསོད་རྒྱལ་བ། བུ་སྟོན། ལོ་ཙཱ་བ་རིན་ཆེན་རྣམ་རྒྱལ། བླ་མ་ཆོས་གྲགས་པ། བསོད་རིན་པ། ཆོས་རྗེ་རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོའི་ཞབས། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། བ་རི་བརྒྱ་རྩའི་རྗེས་གནང་ཐོབ་པའི་རྣམ་གྲངས་ནི། དཔལ་རྡོ་རྗེ་གདན་སྐུ་མདོག་སེར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། གཙོ་འཁོར་གསུམ་པ་བ་རི་ལོ་ཙཱས་བསྒྱུར་བ། འཇམ་དཔལ་བརྟན་པའི་འཁོར་ལོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། སློབ་དཔོན་སྦ་ཀནྡྷས་མཛད་པ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ལོ་ཙཱ་བ་བ་རིས་བསྒྱུར་བ། ཡང་འཇམ་དཔལ་བརྟན་པའི་འཁོར་ལོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཚིག་ལྷུག་པ། རྒྱ་གར་གྱི་མཁན་པོ་བིདྱི་བྷ་དྲ་དང་ལོ་ཙཱ་བ་རིན་ཆེན་བཟང་པོས་བསྒྱུར་བ། ཨ་ར་པ་ཙ་ན་སྐུ་མདོག་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་སློབ་དཔོན་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པ། ཡང་ཨ་ར་པ་ཙ་ན་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཙོ་འཁོར་ལྔ་པ་བ་རི་ལོ་ཙཱས་བསྒྱུར་བ། ཡང་འཇམ་དཔལ་ཞལ་གསུམ་ཕྱག་དྲུག་པ་སྐུ་མདོག་དམར་མོ། འཇམ་དཔལ་རྡོ་རྗེ་གཙོ་འཁོར་ལྔ་པ་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། ཡང་འཕགས་པ་འཇམ་དཔལ་ཡི་གེ་དྲུག་པ་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཙོ་འཁོར་ལྔ་པ། འཇམ་དཔལ་ངག་གི་རྒྱལ་པོ་སེར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་འཁོར་གཅིག་པ་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། རྒྱལ་པོ་འཇམ་དཔལ་རོལ་པ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་དྷརྨ་ཀཱིརྟིས་བསྒྱུར་བ། རིག་པ་འཛིན་པའི་སྡེ་སྣོད་ཀྱི་བསྡུས་པའི་འཇམ་དཔལ་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཉུང་ངུ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ལོ་ཙཱ་བ་བ་རིས་བསྒྱུར་བ།

ཤེས་རབ་སྐྱེ་བར་བྱེད་པའི་འཇམ་དཔལ་གྱི་སྦྱོར་བ་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། ཡན་ལག་མེད་པའི་རྡོ་རྗེ་འཇམ་དཔལ་དབྱངས་སྐུ་མདོག་སེར་པོ། ཞལ་གཅིག་ཕྱག་དྲུག་པ་འམ། ཡང་ན་ཞལ་གཅིག་ཕྱག་བཞི་པ། ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཡང་འཇམ་དཔལ་ཆོས་ཀྱི་དབྱིངས་ངག་གི་དབང་ཕྱུག་དཀར་པོ་ཞལ་བཞི་ཕྱག་བརྒྱད་པ། བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། མཚན་ཡང་དག་པར་བརྗོད་པ་འཇམ་པའི་དབྱངས་དམར་ཞིང་དཀར་བ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཟླ་བའི་མདོག་ཅན། ཡང་འཇམ་རྡོར་དཔའ་བོ་གཅིག་པ་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། འཇམ་དཔལ་ཤེས་རབ་འཁོར་ལོ་སྐུ་མདོག་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། འཕྲལ་དུ་ཉམས་སུ་མྱོང་བའི་འཇམ་དཔལ་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། གཤིན་རྗེ་མཐར་བྱེད་ནག་པོ་ཞལ་དྲུག་ཞབས་དྲུག་པའམ། ཁ་ཅིག་ཞལ་དྲུག་ཕྱག་དྲུག་པ། བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། མཐར་བྱེད་སྔོན་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། རྡོ་རྗེ་ཆོས་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་པཱུ་ཛ་ན་བྷ་དྲས་མཛད་པ། རྗེ་བཙུན་སེང་གེ་སྒྲ་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། སྔོན་པོ་ཙནྡྲ་གོ་མིས་མཛད་པ། ཡང་མཐར་བྱེད་ནག་པོ་ཞལ་གསུམ་ཕྱག་དྲུག་པ། བཅོམ་ལྡན་འདས་ཧ་ལ་ཧ་ལ་སྔོན་པོ་ཞལ་གསུམ་ཕྱག་དྲུག་པ་ཚིགས་སུ་བཅད་པ་ལྷན་ཅིག་སྐྱེས་པ་རོལ་པས་མཛད་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། ཁ་སརྤ་ཎི་སྐུ་མདོག་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཙོ་འཁོར་ལྔ་པའི་སྒྲུབ་ཐབས་ཉུང་ངུ་སློབ་དཔོན་རྙོག་པ་མེད་པས་མཛད་པ། པཎྜི་ཏ་ཨ་མོ་གྷ་བཛྲ་དང་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ཡི་གེ་དྲུག་པ། སྐུ་མདོག་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཙོ་འཁོར་གསུམ་དུ་སྒྲུབ་པའི་ཐབས་སློབ་དཔོན་ལྷན་ཅིག་སྐྱེས་པའི་རོལ་པས་མཛད་པ། པཎྜི་ཏ་ཨ་མོ་གྷ་བཛྲ་དང་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཧ་རི་ཧ་རི་ལ་ཞོན་པར་གྱུར་པ་སེར་པོ་ཞལ་དྲུག་པ་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་གསུངས་པ།

པཎྜི་ཏ་ཨ་མོ་ག་བཛྲ་དང༌། བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། བཅོམ་ལྡན་འདས་ཧ་ལ་ཧ་ལ་དཀར་པོ་ཞལ་གསུམ་ཕྱག་དྲུག་པ་སློབ་དཔོན་ལྷན་ཅིག་སྐྱེས་པའི་རོལ་པས་མཛད་པ་པཎྜི་ཏ་ཨ་མོ་གྷ་བཛྲ་དང་དགེ་སློང་བ་རིས་བསྒྱུར་བ། འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ཡན་ལག་ཐམས་ཅད་དམར་བ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཙོ་འཁོར་བཅུ་པ་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་གསུངས་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་དགེ་སློང་བ་རིས་བསྒྱུར་བ། སྒྱུ་འཕྲུལ་དྲྭ་བའི་རིམ་པའི་སྤྱན་རས་གཟིགས་སྐུ་མདོག་ནག་པོ་ཞལ་ལྔ་ཕྱག་བཅུ་གཉིས་པ་སློབ་དཔོན་དགྲ་ལས་རྣམ་པར་རྒྱལ་བས་མཛད་པ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། དགེ་སློང་བ་རིས་བསྒྱུར་བ། སེང་ལྡེང་ནགས་ཀྱི་སྒྲོལ་མ་ལྗང་ཁུ་བླ་མ་རྡོ་རྗེས་མྱུ་གུས་མཛད་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེས་དང་ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། སྒྲོལ་མ་དཀར་མོ་ཕྱག་བཞི་མ་སློབ་དཔོན་ཡིད་བཞིན་ནོར་བུ་རྒྱལ་པོས་མཛད་པ། ཡང་སེང་ལྡེང་ནགས་ཀྱི་སྒྲོལ་མ་སྐུ་མདོག་ལྗང་ཁུ་ཞལ་གཅིག་ཕྱག་གཉིས་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་གསུངས་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཡང་སྒྲོལ་མ་དཀར་མོ་ཞལ་གཅྀག་ཕྱག་གཉིས་མ་ངག་གི་དབང་ཕྱུག་གྲགས་པའི་སློབ་དཔོན་མཐུ་བོས་མཛད་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཨུ་རྒྱན་གྱི་འོད་ཟེར་ཅན་སྐུ་མདོག་དམར་པོ་ཞལ་དྲུག་ཕྱག་བཅུ་གཉིས་པ་ཨ་མོ་གྷ་བཛྲ་དང་ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་དྷརྨ་ཀཱིརྟིས་བསྒྱུར་བ། རྟོགས་པ་ལས་བྱུང་བའི་འོད་ཟེར་ཅན་སེར་མོ་ཞལ་གསུམ་ཕྱག་བརྒྱད་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ལོ་ཙཱ་བ་དགེ་སློང་དྷརྨ་ཀཱིརྟིས་བསྒྱུར་བ། རྟོགས་པ་ལས་གསུངས་པའི་འོད་ཟེར་ཅན་དཀར་མོ་ཞལ་ལྔ་ཕྱག་བཅུ་ཞབས་བཞི་པ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། མྱ་ངན་མེད་པའི་ཤིང་གི་འོད་ཟེར་ཅན་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌།

ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཨུ་རྒྱན་གྱི་འོད་ཟེར་དམར་མོ་ཞལ་དྲུག་ཕྱག་བཅུ་གཉིས་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། དགེ་སློབ་བ་རིས་བསྒྱུར་བ། ཨུ་རྒྱན་གྱི་རིམ་པའི་འོད་ཟེར་ཅན་དམར་མོ་ཞལ་དྲུག་ཕྱག་བཅུ་གཉིས་མ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། དགེ་སློང་བ་རིས་བསྒྱུར་བ། ལྷ་མོ་ཙུན་དྷ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ལྷ་མོ་ནོར་རྒྱུན་མ་སྐུ་མདོག་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཨུ་ཙུས་ས་མ་ཏ་ཛཾ་བྷ་ལ་སྐུ་མདོག་ནག་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་དགེ་སློང་བ་རིས་བསྒྱུར་བ། འཕགས་མ་ཁྲོ་གཉེར་ཅན་སེར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་དགེ་སློང་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། རྒྱལ་བ་བྱམས་པ་ཞལ་གསུམ་ཕྱག་བཞི་པ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ཁམས་པ་ལོ་ཙཱ་བས་བསྒྱུར་བ། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་བ་རི་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཡང་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་བ་རི་ལོ་ཙཱ་བས་བསྒྱུར་བ། ཡང་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་སྐུ་མདོག་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ནོར་སྦྱིན་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཤེས་ཕྱིན་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་མདོར་བསྡུས་པ། མྀ་གཡོ་བ་སྔོན་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཛེ་ཏཱ་རིས་མཛད་པ། མི་གཡོ་བའི་ལས་མཐའ་སློབ་དཔོན་པྲ་བྷ་ཀཱིརྟིས་མཛད་པ། གཏུམ་པོ་སྐུ་མདོག་སྔོན་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། བ་རི་ལོ་ཙཱའི་འགྱུར། ལོ་མ་གྱོན་མ་སེར་མོ་ཞལ་གསུམ་ཕྱག་དྲུག་མ། འོད་ཟེར་ཅན་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་རྡོ་རྗེ་གདན་པས་མཛད། པ་པཎྜི་ཏ་དེ་ཉིད་དང་ཁམས་པ་ལོ་ཙཱ་བས་བསྒྱུར་བ།

འཕགས་མ་འོད་ཟེར་ཅན་གྱི་གཟུངས་རྡོ་རྗེ་གདན་པ་དང་བ་རིའི་འགྱུར། འཕགས་མ་དབྱངས་ཅན་མ་སྐུ་མདོག་དམར་མོ་ཞལ་གསུམ་ཕྱག་དྲུག་མ་སློབ་དཔོན་པད་འཛིན་གྱིས་མཛད་པ། དཔལ་བགེགས་ཀྱི་དགྲ་སྔོན་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། རལ་པ་གཅིག་མ་ནག་པོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། འཕགས་མ་དུག་སེལ་མ་ཞལ་དྲུག་མ་དེའི་གཟུངས་རིག་བྱེད་མ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཞི་མ། རྟོག་པ་ལས་གསུངས་པའི་རིགས་བྱེད་མ། དམར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ། ཡང་རྟོགས་པ་ལས་གསུངས་པའི་རིག་བྱེད་མ་དམར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། ཨུ་རྒྱན་ལས་བྱུང་བའི་རིག་བྱེད་མ་དམར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། ཁམས་གསུམ་དབང་དུ་བྱེད་པའི་སྤྱན་རས་གཟིགས་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ས་ར་ཧས་མཛད་པ། དཔལ་སྣོད་གནས་སེར་པོ་ཡུམ་ཅན་གཙོ་འཁོར་དགུ་པ། རྒྱ་ནག་པོའི་རིམ་པའི་སྒྲོལ་མ་སྔོན་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་སློབ་དཔོན་རྟག་པའི་རྡོ་རྗེས་མཛད་པ། རྡོ་རྗེ་སྒྲོལ་མ་གསེར་མདོག་ཅན་ཞལ་བཞི་ཕྱག་བརྒྱད་མ་གཙོ་འཁོར་བཅུ་གཅིག གཤིན་རྗེ་གཤེད་མཐར་བྱེད་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཡུམ་ཅན། གཙུག་གཏོར་རྣམ་རྒྱལ་ཞལ་གསུམ་ཕྱག་བརྒྱད་པ། སོ་སོར་འབྲང་མ་ཆེན་མོ་སེར་མོ་ཞལ་བཞི་ཕྱག་བརྒྱད་མ། འཕགས་མ་རྨ་བྱ་ཆེན་མོ་ལྗང་ཁུ་ཞལ་གསུམ་ཕྱག་དྲུག་པ། སྟོང་ཆེན་མོ་རབ་ཏུ་འཇོམས་པ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་དྲུག་མ། གསང་སྔགས་ཆེན་མོ་རྗེས་སུ་འཛིན་མ་ནག་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། བསིལ་བའི་ཚལ་ཆེན་མོ་དམར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། འཕགས་མ་ཕྱིར་བཟློག་པ་ཆེན་མོ་ནག་མོ་ཞལ་གཅིག་ཕྱག་དྲུག་མ། རྒྱལ་མཚན་རྩེ་མོའི་དཔུང་རྒྱན་སྐུ་མདོག་ནག་པོ་ཞལ་གསུམ་ཕྱག་བཞི་མ། འཕགས་མ་རྣམ་པར་རྒྱལ་མ་སེར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། རྡོ་རྗེ་གནྡྷ་རི་ཉིད་སྐུ་མདོག་ནག་མོ་ཞལ་དྲུག་ཕྱག་བཅུ་གཉིས་མ། གུར་མགོན་ཞལ་གཅིག་ཕྱག་གཉིས་པ། རྡོ་རྗེ་ལྕགས་སྒྲོག་སྒྲོལ་མ་སྔོན་མོ་ཞལ་གསུམ་ཕྱག་དྲུག་མ།

བཅོམ་ལྡན་འདས་སྟོབས་པོ་ཆེ་དམར་པོ་ཞལ་གཅིག་ཕྱག་བཞི་པ། འཕགས་མ་གདུགས་དཀར་མོ་ཅན་དཀར་མོ་ཞལ་གསུམ་ཕྱག་དྲུག་མ། དཔལ་རྟ་མགྲིན་དམར་པོ་ཞལ་གསུམ་ཕྱག་དྲུག་པ། དཔལ་ཧེ་རུ་ཀ་ཡབ་ཡུམ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། བདག་མེད་མ་སྔོན་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཡང་བཅོམ་ལྡན་འདས་བདག་མེད་མ་སྔོན་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། གསེར་མདོག་ཅན་གྱི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། རྟོགས་པ་ལས་རྒྱུད་པའི་འོད་ཟེར་ཅན་སེར་མོ་ཞལ་གསུམ་ཕྱག་བཅོ་བརྒྱད་མ། རྡོ་རྗེ་ཕག་མོ་དམར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། འཕགས་མ་རིག་བྱེད་མ་དམར་མོ་ཞལ་གཅིག་ཕྱག་བཞི་མ། སྒྲོལ་མ་དཀར་མོ་ཞལ་གསུམ་ཕྱག་དྲུག་མ་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ། ཧ་ལ་ཧ་ལ་དམར་པོ་ཞལ་གསུམ་ཕྱག་དྲུག་པ། འཕགས་པ་གནོད་པ་གནས་ཆུ་དབང་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། རྡོ་རྗེ་ཧཱུཾ་མཛད་ནག་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། རྣམ་རྒྱལ་གཙོ་འཁོར་དགུ་པ། འོད་ཟེར་ཅན་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། བསྐུལ་བྱེད་མ་དཀར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་རྣམས་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བ། ཡང་རྟོགས་པ་ལས་གསུངས་པའི་རིག་བྱེད་མ་དཀར་མོ། རིག་པ་འཛིན་པའི་སྡེ་སྣོད་ལས་བསྡུས་པའི་འཇམ་དབྱངས་ཀྱི་སྒྲུབ་ཐབས་རྣམས་ལ་ཆོས་ཚན་བརྒྱ་ཐམ་པ་ཡོད་པ་ཐོབ་པའི་ལུང་གི་བརྒྱུད་པ་ནི། བཻ་རོ་ཙ་ན་རཀྵི་ཏ། ཏ་ཐཱ་ག་ཏ་རཀྵི་ཏ། རྡོ་རྗེ་གདན་པ། བ་རི་ལོ་ཙཱ་བ། མཆིམས་བརྩོན་སེང༌། མཆིམས་བརྩོན་འགྲུས་བླ་མ། རྩི་མཆིམས་འདུལ་འཛིན། ངོར་རྗེ་ཆོས་རྡོར། རོང་པ་རྒ་ལོ། གཅེ་བློ་ལྡན། བླ་མ་དགེ་སློང༌། གཅེ་དབང་རྡོར། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན། ཆོས་རྗེ་ཉི་མ་རྒྱལ་མཚན། ཆོས་ཀྱི་རྗེ་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ལུང་དང་རྗེས་གནང་གཉིས་ཀའི་བརྒྱུད་པ་ནི། སངས་རྒྱས།

ཛེ་ཏཱ་རི། རྡོ་རྗེ་གདན་པ་གསུམ། བ་རི་ལོ་ཙཱ་བ། ས་ཆེན་ཀུན་སྙིང༌། སློབ་དཔོན་བསོད་རྩེ། རྗེ་བཙུན་གྲགས་རྒྱལ། ཆོས་རྗེ་ས་པཎ། ཆོས་རྒྱལ་འཕགས་པ། སྣུར་ཡེར་རྒྱལ། སྣུར་ཆོས་འཕགས། གར་ཐིག་པ་སིང་རྒྱལ། མཁན་ཆེན་ཤེས་རབ་རྡོ་རྗེ། ཕག་གྲུ་མཐིལ་གྱི་ཆོས་སྒོ་བ། ཕག་གྲུ་མཐིལ་གྱི་མཁན་པོ་གྲགས་རྡོར། བླ་མ་བྱང་ཆུབ་རྡོ་རྗེ། འཇམ་དབྱངས་ཆོས་རྗེ། ཆོས་རྗེ་ཉི་མ་རྒྱལ་མཚན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་ཀླུ་དབང་རྒྱལ་པོའི་སྒྲུབ་ཐབས་ཀྱི་བརྒྱུད་པ་ནི། སློབ་དཔོན་ཀླུ་སྒྲུབ། ཙནྡྲ་ཀཱིརྟི། རིག་པའི་ཁུ་བྱུག ཨ་ཏི་ཤ། འབྲོམ་སྟོན་པ། སྤྱན་མངའ་བ། རྣལ་འབྱོར་པ། བྱ་ཡུལ་བ། དགེ་བཤེས་ཀླུ་མེན་པ། དེས་སྣར་ཐང་པ་ནས་བརྒྱུད་དེ། མཁན་ཆེན་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་སྒྲོལ་མ་ཡིད་བཞིན་འཁོར་ལོའི་བརྒྱུད་པ་ནི། རྗེ་བཙུན་མ་སྒྲོལ་མ། ངག་དབང་གྲགས་པ། གསེར་གླིང་པ། ཇོ་བོ་རྗེ། འབྲོམ་སྟོན་པ། སྤྱན་སྔ་བ། བྱ་ཡུལ་བ། དྭགས་པོ་ལྷ་རྗེ། ཞང་སུམ་ཐོག་པ། བཀའ་སྩལ་པ། བླ་མ་ལྷ་བ། རིན་པོ་བཟང་པོ། མཁན་ཆེན་བདེ་བ། མཁན་ཆེན་བསོད་ནམས་གྲགས་པ། ཆོས་རྗེ་བསོད་རྒྱལ་བ། དོན་ཡོད་རྒྱལ་མཚན། མཁན་ཆེན་རིན་རྒྱལ་བ། རྒྱལ་བ་ཕྱག་ན། ཆོས་རྗེ་ཀུན་བློ་བ། དེས་བདག་ལའོ།། ༈ །།ཡང་སྙིང་པོ་དོན་གསུམ་བརྒྱུད་པ་ནི། སྤྱན་རས་གཟིགས། བྱང་སེམས་བློ་གྲོས་རིན་ཆེན། ཏིལླི་པ། རོལ་པའི་རྡོ་རྗེ། མི་ཏྲ་ཛྭ་ཀི། ཁྲོ་ཕུ་ལོ་ཙཱ་བ། འཛམ་བུ་བ་བསོད་དབང༌། ཁྲོ་ཕུ་རིན་པོ་ཆེ། མཁན་ཆེན་ཀ་བཞི་པ། བསོད་གྲགས་པ། གྲགས་རྒྱལ་བ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་བ། རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་ཐུགས་རྗེ་ཆེན་པོ་ཕྱག་བཞི་པ་རིག་གཏད་ཀྱི་བརྒྱུད་པ་ནི།

མགོན་པོ་ཚེ་དཔག་མེད། སྤྱན་རས་གཟིགས། གྲུབ་ཆེན་ར་ཧུ་ལ། དཔལ་ལྡན་ཨ་ཏི་ཤ། གུང་ཐང་པ། ནག་ཚོ་ཚུལ་ཁྲིམས་རྒྱལ་བ། དེས་རོང་ལག་སོར་བ། བག་ཡུལ་བ། དེས་འཕགས་གཅེ་སྒོ་དང༌། མངའ་རིས་པ་ཤེས་རབ་རྒྱལ་མཚན་གཉིས་ལ་གནང༌། དེ་གཉིས་ཀ་ལ་སངས་རྒྱས་སྐྱེར་སྒང་པས་གསན། དེས་སངས་རྒྱས་གཉན་སྟོན། དེས་འགྲོ་མགོན་ཆོས་རྗེ་སྟོན་པ། དེས་མཁས་གྲུབ་ཤངས་སྟོན། དེས་མཁས་བཙུན་ཡོན་ཏན་རྒྱ་མཚོ། དེས་ཁྱུང་པོ་ཚུལ་ཁྲིམས་མགོན། དེས་བྱ་བྲལ་བ་རི་ཁྲོད་རིན་ཆེན། དེས་འཇམ་དབྱངས་ཤངས་པ་དཀར་པོ། དེས་མཉམ་མེད་སངས་རྒྱས་དཔལ་བཟང་པོ། དེས་ཆོས་རྗེ་ནམ་མཁའ་རྒྱལ་མཚན། དེས་རིན་ཆེན་འབྱུང་གནས་ཞི་བ། དེས་ཆོས་རྗེ་ཀུན་དགའ་བློ་གྲོས་པ། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །འདི་ནི་རིམ་པ་ལྟར་ཞལ་མཐོང་བ་བར་མ་ཆད་པའི་བརྒྱུད་པའོ། །ཡང་དྲན་སྟོན་གྱི་བདེན་བདར་གྱི་ལུང་གི་བརྒྱུད་པ་ནི། བླ་མ་ཤཱཀྱ་དཔལ། མཁན་ཆེན་རྒྱལ་མཆོག་པ། རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་དེས་བདག་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། བླ་མ་ལྔ་བཅུ་པའི་ཆོས་ཚན་ཐོབ་པའི་རིམ་པ་ནི། ཀླུ་དབང་རྒྱལ་པོའི་སྒྲུབ་ཐབས་འཕགས་པ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ། ཀླུའི་དབང་ཕྱུག་རྒྱལ་པོའི་གཟུངས། རྣམ་པར་རྒྱལ་མའི་སྒྲུབ་ཐབས་འཕགས་པས་མཛད་པ། འོད་ཟེར་ཅན་མའི་སྒྲུབ་ཐབས་དཔལ་ལྡན་བླ་མ་དམ་པ་བསོད་ནམས་རྒྱལ་མཚན་གྱིས་མཛད་པ། མགོན་དཀར་དུར་ཁྲོད་བདག་པོའི་གསང་བ་སྙིང་གི་འཁོར་ལོའི་བརྒྱུད་ལེའུ་བརྒྱད་པ། དུར་ཁྲོད་བདག་པོའི་སྒྲུབ་ཐབས། ཞལ་གདམས་ས་ཆེན་ཀུན་དགའ་སྙིང་པོས་མཛད་པ། མགོན་དཀར་དུར་ཁྲོད་བདག་པོའི་རྒྱ་གཞུང༌། འཛིང་པ་ལོ་ཙཱ་བས་མཛད་པའི་དུར་ཁྲོད་བདག་པོའི་སྒྲུབ་ཐབས་དང་ལས་སྦྱོར། དུར་ཁྲོད་བདག་པོའི་ལས་ཚོགས་ཡར་ཀླུངས་པ་སེང་གེ་རྒྱལ་མཚན་གྱིས་མཛད་པ། དུར་ཁྲོད་བདག་པོའི་མཆོད་གཏོར། རྗེས་གནང་གི་ཆོ་ག་དང་བཅས་པ།

དུར་ཁྲོད་བདག་པོའི་བསྟོད་པ་འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན་གྱིས་མཛད་པ་དང༌། ཡར་ཀླུངས་པ་སེང་གེ་རྒྱལ་མཚན་གྱིས་མཛད་པ་གཉིས། དཔལ་ཕྱག་ན་རྡོ་རྗེས་ཨུ་ཙཱརྻ་ལ་བརྟེན་ནས་སློབ་མ་རྗེས་སུ་གཟུང་བའི་ཆོ་ག་བྱིན་རླབས་ཀྱི་རྒྱ་མཚོ་དང༌། སྒྲུབ་ཐབས་བགེགས་འཇོམས་རྡོ་རྗེ་ཞེས་བྱ་བ་དགེ་སློང་བྱང་སེང་གིས་མཛད་པ། དེའི་ལས་ཚོགས། བཅོམ་ལྡན་འདས་མ་རི་ཁྲོད་ལོ་མ་ཅན་གྱི་སྒྲུབ་ཐབས། གཟུངས་སྒྲུབ་བསྟོད་པ་དང་བཅས་པ་འཕགས་པས་མཛད་པ། ཁྲུས་ཆོག་རྒྱས་པ་དང༌། རིམས་དྲག་པོ་གཅོད་པའི་ཚུལ་དགའ་ལྡན་པ་ཀུན་བསོད་ཀྱིས་མཛད་པ། ཁྲོ་བོ་རྨེ་རྩེགས་ཀྱི་སྒྲུབ་ཐབས་ལས་ཚོགས་དང་བཅས་པ་ས་པཎ་གྱིས་མཛད་པ། དེའི་བསྟོད་པ། གཏོར་ཆོག་དང་བཅས་པ་འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན་པས་མཛད་པ། རྡོ་རྗེ་དྲིལ་བུ་ཕྲེང་བ་གསུམ་གྱི་རྗེས་གནང་རྗེ་བཙུན་གྲགས་པས་མཛད་པ། མི་གཡོ་བ་སྔོན་པོའི་ལས་ཚོགས་ཆོས་དྲུག་ཏུ་གྲགས་པ་འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན་པས་མཛད་པ། དཔལ་དུས་ཀྱི་འཁོར་ལོའི་དབང་གོང་མ་བཞིའི་ཆོ་ག་ཇོ་ནང་པ་ཕྱོགས་ལས་རྣམ་རྒྱལ་གྱིས་མཛད་པ། ཤ་བ་རི་སྦྱོར་བ་ཡན་ལག་དྲུག་གི་རྒྱ་གཞུང༌། པདྨ་དྲག་པོའི་སྒྲུབ་ཐབས། མཁན་ཆེན་དོན་གྲུབ་དཔལ་འབྱོར་བས་མཛད་པ། པདྨ་དྲག་པོའི་དབང་ཆོག་གཏེར་ཁ་གོང་འོག་གཉིས་ཀ་མཐུན་པ། མི་འཁྲུག་པའི་རྒྱུད་འཕགས་པ་ལས་ཀྱི་སྒྲིབ་པ་རྒྱུན་གཅོད་པའི་གཟུངས། མཚན་བརྗོད་བཀླག་པའི་མན་ངག་ཁ་ཆེ་པཎ་ཆེན་གྱིས་མཛད་པ། རྡོ་རྗེ་རྣམ་འཇོམས་ཀྱི་སྒྲུབ་ཐབས་འཕགས་པས་མཛད་པ། མཁའ་འགྲོ་མ་སེང་གེ་གདོང་ཅན་གྱི་སྒྲུབ་ཐབས། ལོ་རྒྱུས་ལས་ཚོགས་དང་བཅས་པ། བདག་གིས་ནི་ཆོས་རྗེ་རིན་པོ་ཆེ་རྒྱལ་བ་ཕྱག་ན་ལ་ཡང་ཞུས་ཤིང་དཔོན་ས་གོང་མ་རིན་པོ་ཆེ་ལ་ཐོབ་པའི་བརྒྱུད་པ་ནི། པཎ་ཆེན་ནགས་ཀྱི་རིན་ཆེན་དེ་ཉིད་ཀྱིས་ཁོང་རང་གི་བླ་མ་མ་ཧཱ་བྷོ་དྷི་བྱ་བ་ལ་ཞུས་འདུག རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་སྲས། ཕྱག་ན་རྡོ་རྗེ་གཏུམ་ཆུང་དང་བཅས་པ་འཁོར་རྟ་བདག་བརྒྱད།

ཀླུ་བརྒྱད། སྡེ་བརྒྱད་དང་བཅས་པ། མགོན་པོ་གསང་སྒྲུབ་ལྗང་ཁུ་ཡབ་ཡུམ། མགོན་པོ་གར་མཁན་མཆོག་ལྗང་ཁུ་ཞལ་གསུམ་ཕྱག་བཅོ་བརྒྱད་པ་གཙོ་འཁོར་བརྒྱད། དྲག་བྱེད་ནག་པོ་ཀླུ་བརྒྱད་ཀྱིས་བསྐོར་བ། མདུང་དམར་ཅན་གསང་བ་གཤིན་རྗེ་ལྟར་སྒྲུབ་པའི་གཙོ་འཁོར་ཉི་ཤུ་རྩ་གཅིག་པ། ནང་མ་མོ་ལྟར་བསྒྲུབ་པ་གཙོ་འཁོར་དགུ་རྣམས་ཀྱི་སྒྲུབ་ཐབས་གཏོར་ཆོག བསྐང་བཤགས། བསྟོད་པ། བསྐུལ། གཏོར་རྫོངས་དང་བཅས་པའི་ལུང་དང༌། རྗེས་གནང༌། པཎ་ཆེན་ཤཱཀྱ་ཤྲཱི་ཡི་བདེ་མཆོག་དཀར་པོའི་ཆོས་བསྐོར་རྣམས་ལ། དཔལ་ལྷན་ཅིག་སྐྱེས་པ་ཡུམ་ཕག་མོ་དམར་མོ་དང་བཅས་པའི་སྒྲུབ་ཐབས། བྱིན་རླབས་འཕོ་བའི་མན་ངག་གཉིས། བྲམ་ཟེ་ཆེན་པོ་ས་ར་ཧས་མཛད་པ། པཎ་ཆེན་མཚུར་ཕུ་བ་ལ་གནང་བའི་ཟབ་ཆོས་སུ་གྲགས་པ་བདེ་མཆོག་དཀར་པོའི་རྫོགས་རིམ་དམིགས་པ་ཟབ་མོ་པཎ་ཆེན་ཉིད་ཀྱིས་མཛད་པ་དེ་ཉིད་འགྲེལ་པ་མཁན་ཆེན་ལྷོ་བྲག་པ་བྱང་ཆུབ་དཔལ་བཟང་པོས་མཛད་པ། བདེ་མཆོག་ལྷན་སྐྱེས་རིམ་ལྔའི་དབང་ཆོག་བདག་ཉིད་ཀྱིས་ཡི་དམ་ལྷ་དབང་ལེན་པའི་གདམས་པ། ཅོག་རོ་ཆོས་རྒྱལ་གྱིས་མཛད་པ། འདི་དག་གི་རྒྱབ་ཆོས་ཀྱི་དབང་བྱིན་རླབས་ལུང་ཚང་མ་དང་བཅས་པ། ནཱ་རོ་པའི་ཆོས་དྲུག་རྡོ་རྗེའི་ཚིག་རྐང༌། མར་པའི་ཚིགས་སུ་བཅད་པ་བརྒྱད་མ། ཤོག་དྲིལ་བཞི། གྲུབ་ཆེན་ཤ་བ་རིའི་དོ་ཧ། ནག་པོ་པའི་དོ་ཧ། ཏིལླི་པའི་དོ་ཧ། བིརྺ་པའི་དོ་ཧ། ཁྱད་པར་རྗེ་བཙུན་བིརྺ་པའི་དོ་ཧའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང༌། བིརྺ་པ། ནག་པོ་པ། ཀརྞ་པ། སྒྲ་མཁན་པ། པུ་ར་པ། བཻ་རོ་ཙ་ན་བཛྲ། ཞང་གཡུ་བྲག་པ། སྒོམ་པ་ཚོང་སྟོན། གཅེ་སྒོམ་རྫོང་བ་ཤེས་རབ་འབར། འདུལ་བ་རྡོ་རྗེ། སྤྲུལ་སྐུ་དཀོན་མཆོག་བཟང་པོ། རྟོགས་ལྡན་རྒྱ་སྟོན་པ། འཛིང་པ་རྒྱལ་མཚན་དཔལ། སྤང་སྒང་པ་གྲགས་པ་དཔལ། ཨཱརྻ་དེ་ཝ། རིན་ཆེན་རྒྱལ་མཚན། བླ་མ་ཆོས་གྲགས་དཔལ། འཇམ་དབྱངས་རིན་ཆེན་དཔལ། འཁྲུལ་ཞིག་གྲགས་པ་བཀྲ་ཤིས་པ། འཇམ་དབྱངས་ཉི་མ་རྒྱལ་མཚན།

མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་རྒྱལ་མཚན་དཔལ་བཟང་པོ་དེས་བདག་ལའོ།། ༈ །།དོ་ཧ་གཞན་རྣམས་ཀྱང་བླ་མ་བཻ་རོ་མན་ཆད་འདི་དང་མཐུན། ཆོས་དྲུག་རྡོ་རྗེའི་ཚིག་རྐང༌། ཚིགས་སུ་བཅད་པ་བརྒྱད་པ། ཤོག་དྲིལ་བཞི་རྣམས་ཀྱི་བརྒྱུད་པ་ནི། རྗེ་མར་པ། ཚུར་དབང་ངེ༌། འཇང་བུ་རྒྱ་གར་བ། སྟོད་ལུང་གར་ར། བླ་མ་ཉི་མ་ལུང་པ། རོག་བདེ་གཤེགས། རྩེ་བ་སྒང་པ། མཉམ་མེད་ཐམས་ཅད་མཁྱེན་པ། གཡུ་སྟོན་རྡོ་རྗེ་དཔལ། བླ་མ་བསོད་ནམས་བཟང་པོ། ཁུ་སྟོན་ཚུལ་མགོན། ཐོད་སྟོན་རྒྱལ་མཚན་གྲུབ། བླ་མ་རིན་ཆེན་རྒྱལ་མཚན། བླ་མ་ཆོས་གྲགས་དཔལ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན། འཇམ་དབྱངས་ཉི་མ་རྒྱལ་མཚན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས་དཔལ་བཟང་པོ། དེས་བདག་ལའོ།། ༈ །།ཡང་ཇོ་བོའི་ལུགས་ཀྱི་མི་འཁྲུགས་པ་ལྷ་བཅུ་གསུམ་གྱི་དཀྱིལ་འཁོར་དུ་དབང་བསྐུར་བ་ཐོབ་པ་དང༌། ཇོ་བོ་རྗེས་མཛད་པའི་སྒྲུབ་ཐབས་ཀྱི་ཆོ་ག དཔའ་བོ་རྡོ་རྗེ་ཐུམ་ཆོག་དང་བཅས་པ་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཀླུ་སྒྲུབ། ཨཱརྻ་དེ་ཝ། ཨ་ཝ་དྷཱུ་ཏི་པ། གསེར་གླིང་པ། ཨ་ཏི་ཤ། འབྲོམ་སྟོན་པ། པུ་ཏོ་བ། ཤ་ར་བ། རྒྱ་ཡི་སྒོམ། གྲུབ་ཐོབ་ནུབ་འབར། ཞང་དགེ་སློང་ཀུན་ཡེ། བླ་མ་ཁུས་སྟོན། འཕགས་འོ། བུ་སྟོན། ཁྱུང་ལྷས་པ། འཇམ་དབྱངས་རིན་ཆེན་རྒྱལ་མཚན་པ། སེམས་དཔའ་ཆེན་པོ་ཟླ་བ་བློ་གྲོས། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས། དེས་བདག་ལའོ།། ༈ །།ཁྲོ་ཕུ་བའི་མན་ངག་བརྒྱ་རྩའི་ལུང་གི་བརྒྱུད་པ་ནི། ཁྲོ་ཕུ་ལོ་ཙཱ་བ་ཚུལ་ཁྲིམས་ཤེས་རབ། ཁྲོ་ཕུ་རིན་པོ་ཆེ་བསོད་ནམས་སེང་གེ། ཚད་མའི་སྐྱེས་བུ་ཚུལ་ཁྲིམས་མགོན་པོ། མཁན་ཆེན་གཞོན་ནུ་མགོན་པོ། མཁན་ཆེན་གྲགས་པ་རྒྱལ་མཚན། མཁན་ཆེན་རིན་ཆེན་རྒྱལ་མཚན། མཁན་ཆེན་རྒྱལ་བ་ཕྱག་ན། མཁན་ཆེན་ཀུན་དགའ་བློ་གྲོས། དེས་བདག་ལའོ། །ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། སྙིང་པོ་དོན་གསུམ་གྱི་ཁྲིད་ཐོབ་པའི་བརྒྱུད་པ་ནི། མན་ངག་བརྒྱ་རྩ་དང་འདྲའོ། །ཡང་མཚན་ཡང་དག་པར་བརྗོད་པའི་ལུང་ཐོབ་པའི་བརྒྱུད་པ་ནི།

རྒྱལ་བ་རྡོ་རྗེ་འཆང༌། འཇམ་དཔལ་དབྱངས། འཇམ་དཔལ་བཤེས་གཉེན། སྒེག་པའི་རྡོ་རྗེ། སུརྻ་སིང་ཧ། དྷརྨ་བཛྲ། མུ་ཏྲི་སིདྷི། དེ་པ་ཤྭ། སྨི་ཏི། ཆང་སྡེ་ངག་དབང༌། ཁམས་པ་ཤེས་རབ་རྡོ་རྗེ། རྔོག་ཆོས་རྡོར། རྔོག་ལྦ་བ། རོང་སྨན་ལྷ་སྟེང༌། ཇོ་འབུམ། ཅོ་མདོ་སྡེ་སེང༌། རོང་པ་རྒ་ལོ། འཇམ་དབྱངས་ལྷོ་པ་གྲུབ་སེང༌། སྟོན་པ་ཆོས་བཟང་པོ། ཐུན་པ་མགོན། བསོད་གྲགས་པ། གྲགས་རྒྱལ་བ། རིན་རྒྱལ་བ། རྒྱལ་བ་ཕྱག་ན། དེས་མཁན་ཆེན་རིན་པོ་ཆེ་ཀུན་དགའ་བློ་གྲོས། དེས་བདག་ལའོ།། ༈ །།ཡང་ཀཱ་རི་ཀ་ལྔ་བཅུ་པའི་ལུང༌། སོ་སོ་ཐར་པའི་འདུལ་བའི་གཞི་གསུམ་ཆོ་ག་ཕྱག་ལེན་དང༌། སོ་ཐར་མདོ་སོགས་རྒྱས་པར་ཐོབ་པའོ།། ༈ །།མཁྱེན་པའི་བདག་ཉིད་བརྩེ་བའི་མངའ་བདག་ལས། །མཛད་པའི་གཞིར་གྱུར་ངུར་སྨྲིག་འཛིན་པའི་དཔལ། །ནུས་པ་མཐར་ཕྱིན་སྨྲ་བའི་དབང་ཕྱུག་གཙོ། །བསྟན་པའི་ཉི་མ་གང་དེ་རྒྱལ་གྱུར་ཅིག ཅེས་སློབ་དཔོན་རིན་པོ་ཆེ་བཀའ་བཅུ་སྨྲ་བ་བློ་གྲོས་བསྟན་པའི་ཞལ་མངའ་ནས་ཀྱི་དྲུང་དུ། དམ་པའི་ཆོས་ཐོབ་པ་ལ། ཐེག་པ་ཆེན་པོ་རྒྱུད་བླ་མའི་ལུང༌། སྤྱོད་འཇུག་གི་ལུང༌། ཤེས་སྤྲིང་གི་ལུང༌། བློ་སྦྱོང་རྒྱས་པ། ཆོས་ཉིད་རྣམ་འབྱེད། དབུས་མཐའ་རྣམ་འབྱེད། སྨན་བླ་སྐོར་གསུམ། མཚན་བརྗོད། ཆ་གསུམ་འབྲིང་པོ། ལམ་འབྲས་རྡོ་རྗེའི་ཚིག་རྐང༌། ནེ་གུ་རྡོ་རྗེའི་ཚིག་རྐང༌། སུམ་པ་ལོ་ཙཱ་བའི་བློ་སྦྱོང་ཞེན་པ་བཞི་བྲལ། མངོན་རྟོགས་རྒྱན། སྤྱོད་བསྡུས་སྒྲོན་མེ། ཤེས་སྤྲིང་ས་བཅད། ཆོས་སྤྱོད་རབ་གསལ། སོ་ཐར། མདོ་རྩ་བ། སེང་གདོང་ཅན་གྱི་སྒྲུབ་ཐབས་ལས་ཚོགས་དང་བཅས་པ། འདུལ་བའི་བསླབ་བྱ་འདུལ་ཁྲིམས་ཉིས་བརྒྱ་ལྔ་བཅུ་རྩ་གསུམ་གྱི་བཤད་པ། གཞི་གསུམ་ཆོ་ག་སོགས་མ་ལུས་རྒྱས་པར་ཐོབ་པའོ། །ལེགས་པའི་དཔལ་ལྡན་རིན་ཆེན་མཆོག་གི་རྒྱན། །བཟང་པོའི་ཕྲིན་ལས་གདུལ་བྱའི་གླིང་ལ་མཛེས། །འདུལ་འཛིན་གཉིས་པ་གང་དེ་རྒྱལ་གྱུར་ཅིག །ཅེས་པས། གསང་སྟེ་སྟོན་པ་སློབ་དཔོན་རིན་པོ་ཆེ་ལེགས་རིན་པའི་དྲུང་དུ་ཆོས་ཐོབ་པ་ལ།

།ཅེས་པས། གྲུབ་ཆེན་རིན་པོ་ཆེ་བློ་གྲོས་དཔལ་ལྡན་པའི་དྲུང་དུ་ཆོས་ཐོབ་པ་ལ། འབའ་ར་བས་མཛད་པའི་ཡིག་ཆའི་སྟེང་ནས་ཚེ་དཔག་མེད་ཀྱི་དབང༌། ཚེ་ཁྲིད་ཀྱི་ལུང་ཐོབ་པའི་བརྒྱུད་པ་ནི། སྣང་བ་མཐའ་ཡས། མ་ཅིག་གྲུབ་པའི་རྒྱལ་མོ། ཏི་ཕུ་གསང་སྔགས་སྡོང་པོ། རས་ཆུང་རྡོ་རྗེ་གྲགས་པ། རྗེ་མི་ལ་ཆེན་པོ། མཉམ་མེད་དྭགས་པོ་ལྷ་རྗེ། རྗེ་ཕག་མོ་གྲུ། རྒྱལ་བ་འཇིག་རྟེན་གསུམ་གྱི་མགོན་པོ། སྤྱན་སྔ་རིན་པོ་ཆེ། ཡང་དགོན་པ། ཐུགས་སྲས་སྤྱན་མངའ་རིན་ལྡན། ཆོས་རྗེ་ཟུར་ཕུག་པ། འབའ་ར་བ་ཆེན་པོ། བླ་མ་རིན་ཆེན་དཔལ་བ། བླ་མ་ཆོས་གྲགས་དཔལ་བ། བྱང་སེམས་ཟླ་བ་བློ་གྲོས། གྲུབ་ཆེན་ཚངས་པའི་རྒྱལ་མཚན། གྲུབ་ཆེན་དཔལ་ལྡན་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་གྲུབ་ཆེན་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། སྐལ་ལྡན་ཐར་ལམ་གྱི་ཡིག་ཆའི་སྟེང་ནས། ནམ་མཁའ་སྒོ་འབྱེད་ཀྱི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། གཞུང་གི་བརྒྱུད་པ། ཐབས་ཀྱི་བརྒྱུད་པ། ཤེས་རབ་ཀྱི་བརྒྱུད་པ་གསུམ་ལས། དང་པོ་ནི་ཐུབ་དབང༌། བྱམས་པ། ཐོགས་མེད། དབྱིག་གཉེན། ཨཱརྻ་དེ་ཝ། དམ་པ་སངས་རྒྱས། སྐྱོ་སྟོན་རྣམ་གཉིས། མ་ཅིག་ལབ་སྒྲོན། རྩི་སྟོན་དར་མ། གླང་ལུང་པ། གཉལ་པ། མལ་གྲོ་བ། གྲུབ་པ་འབར། སྤང་པོ་པ། ཤཱཀྱ་བླ་མ། སངས་རྒྱས་རས་པ། རིན་ཆེན་གཞོན་ནུ། སྒྲོ་སྟོན་ཆེན་པོ། རིན་ཆེན་འོད་ཟེར། ཆོས་ཀྱི་དབང་ཕྱུག་ཞེན་མེད་རས་པ། གཞོན་ནུ་བཟང་པོ། ཡེ་ཤེས་འོད་ཟེར། ཤེས་རབ་སེང་གེ བློ་གསལ་རི་ཁྲོད་པ། འཁྲུལ་ཞིག་ཚངས་པའི་རྒྱལ་མཚན། གྲུབ་ཆེན་རིན་པོ་ཆེ། དེས་བདག་ལའོ། །ཐབས་ཀྱི་བརྒྱུད་པ་ནི། ཐུབ་པ། འཇམ་དབྱངས། ཨཱརྻ་དེ་ཝ། དམ་པ། སྐྱོ་སྟོན་བསོད་ནམས་བླ་མ། མ་ཅིག ཁུ་སྒོམ་ཆོས་སེང༌། དོལ་པ་ཟང་ཐལ། རྒྱ་ནག་གཅེར་བུ། སངས་རྒྱས་རབ་སྟོན། སངས་རྒྱས་དོན་བློང༌། སངས་རྒྱས་རས་པ། སངས་རྒྱས་སྟོན་པ། མཁས་གྲུབ་ཆོས་རྗེ། ཆོས་རྗེ་གསེར་གླིང་པ། རྡོ་རྗེ་དཔལ། ཆོས་དཔལ་ཤེས་རབ། བློ་གསལ་རི་ཁྲོད་པ། འཁྲུལ་ཞིག་ཚངས་པ།

གྲུབ་ཆེན་པ། དེས་བདག་ལའོ།། ༈ །།ཤེས་རབ་ཀྱི་བརྒྱུད་པ་ནི། ཡུམ་ཆེན་མོ། སྒྲོལ་མ་སུ་ཁ་སིདྡྷི། དམ་པ་སངས་རྒྱས། མ་ཅིག་ལབ་སྒྲོན། རྒྱལ་བ་དོན་གྲུབ། དམ་པ་དབུས་པ། རྡོ་རྗེ་བདག གཞོན་ནུ་ཡེ་ཤེས། མི་བསྐྱོད་རྡོ་རྗེ། སངས་རྒྱས་བསྟན་པ། འགྲོ་མགོན་ལུང་ཕྲན་པ། གདན་གཅིག་པ་བསམ་གཏན་དར། མི་ཉག་རིན་རྒྱལ་བ། བློ་གསལ་རི་ཁྲོད་པ། བླ་མ་རིན་པོ་ཆེ། གྲུབ་ཆེན་རིན་པོ་ཆེ། དེས་བདག་ལའོ།། ༈ །།ཡང་བླ་མ་དེའི་དྲུང་དུ། ཨུ་རྒྱན་བསྙེན་སྒྲུབ་ཀྱི་གཞུང་ཆུང་དང༌། ཟླ་བ་སེང་གེའི་ཁྲིད་ཡིག་རྣམས་ཀྱི་ལུང་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྒྱལ་བ་རྡོ་རྗེ་འཆང་ཆེན་པོ། རྡོ་རྗེ་རྣལ་འབྱོར་མ། ཨུ་རྒྱན་པ། ཟླ་བ་སེང་གེ། སེ་རས་པ། བླ་མ་ཚུལ་གྲགས་པ། བླ་མ་རྟོགས་ལྡན་རིན་ཆེན་པ། གྲུབ་ཆེན་པ། དེས་བདག་ལའོ། །ཡང་ན་ཟླ་བ་སེང་གེ་ཡན་འདྲ་ལ། དེས་ཆོས་རྗེ་བྱང་རྩེ་བ། འཇམ་དབྱངས་ཆོས་མགོན་པ། བླ་མ་ཚུལ་གྲགས་པ། དྲུང་ཚངས་པ་བ། གྲུབ་ཆེན་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་གྲུབ་ཆེན་དེ་ཉིད་ཀྱི་དྲུང་དུ། རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་བིརྺ་པའི་བསྲུང་བ་ས་པཎ་གྱིས་མཛད་པ་དང༌། འཕགས་པས་མཛད་པ། རྗེ་བཙུན་གྲགས་པས་མཛད་པ་ཚར་གསུམ་ཁུག་པ་རྣམས་ཀྱི་ལུང་བརྒྱུད་ནི། རྒྱལ་བ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང༌། བདག་མེད་མ། བིརྺ་པ། ས་ཆེན། བསོད་རྩེ། ས་པཎ། ཚོགས་སྒོམ་པ། ཆོས་རྗེ་ཉབ་ཆེན་པ། བྲག་ཕུག་པ། མཁས་བཙུན་དཀོན་མཆོག་རྒྱལ་མཚན། བུ་སྟོན་པ། འཇམ་དབྱངས་གྲགས་པ། ཆོས་རྗེ་ཉི་མ་བ་དང༌། བསམ་འགྲུབ་དཔལ་བཟང་པོ་གཉིས་ཀའི་དྲུང་ནས་གསན། ཡང་ན་པཎྜི་ཏ། འཕགས་པ། ཞང་དཀོན་མཆོག་དཔལ། བྲག་ཕུག་པ། མཁས་བཙུན་དཀོན་མཆོག་རྒྱལ་མཚན། བུ་སྟོན། འཇམ་དབྱངས་གྲགས་པ། བླ་མ་བསམ་འགྲུབ་དཔལ་བཟང་དྲུང་ནས་ཀུན་མཁྱེན་ཆེན་པོའི་གསུང་སྒྲོས། ཤེས་རབ་བླ་མའི་ཞུ་ཡིག་དང༌། དེའི་ལན་ཉེ་གནས་སང་རིན་ལ་གདམས། ཉེ་གནས་ནམ་མཁའ་བྱང་ཆུབ་ལ་གདམས་པ། ཡུལ་མདོ་ཁམས་སུ་སྐྱེས་པའི་དད་པ་ཅན་ལ་གདམས་པ། གེགས་སེལ་ཉམས་ཀྱི་ཡིག་ཆུང༌།

ཡང་སྦྱོར་བ་ཡན་ལག་དྲུག་པའི་ཁྲིད་ལུང༌། རླུང་བུམ་པ་ཅན་རྒྱན་ཆེན་དང་རྒྱན་ཆུང༌། དེའི་རྒྱབ་ཆོས་ལ་ཤ་བ་རིའི་རྒྱ་གཞུང་བཤད་པ། གེགས་སེལ་མན་ངག་ལུས་སྦྱོངས་ནཱ་རོ་དྲུག་སྐོར། རས་ཆུང་དྲུག་སྐོར། བསྲེ་འཕོ་དྲུག་སྐོར། སུམ་ཅུ་སོ་བདུན་མ། རས་ཆེན་པས་མཛད་པའི་གངས་སེང་མ། གཞུང་ཆ་ལག་དང་བཅས་པ། ལུས་སྦྱོངས་བཅུ་གཉིས་མ། གཏུམ་མོ་ཉི་ཤུ་པ། ཡན་ལག་ལྔ་སྦྱོངས། བསྙེན་སྒྲུབ་ཀྱི་འཁྲུལ་འཁོར་ཆ་ལག་རྫོགས་པ། མཚན་བརྗོད་བཀླག་པའི་མན་ངག མཚན་བརྗོད་ཀྱི་ལུང༌། གྲིབ་གཟུགས་བརྟག་པ། སྲོག་གཟུགས་བརྟག་པ། གཞན་གྱི་སྲོག་གཟུགས་བརྟག་པ། དྲི་ཆུ་བར་མའི་བསྲུང་བ་སྒྲུབ་ལུགས་བརྟག་པ། ཡང་རྗེ་རས་ཆུང་པའི་ལུགས་ཀྱི་ཚེ་སྒྲུབ། ཕྱི་སྣང་བ་མཐའ་ཡས་ལྟར་གྱི་བསྐྱེད་པའི་རིམ་པ། ནང་ཚེ་དཔག་ཏུ་མེད་པའི་སྐུ་ལ་བརྟེན་ཏེ་ཚེའི་སྒོ་བསྡམ་པ་སྐུའི་རྫོགས་རིམ། གསུང་རྡོ་རྗེའི་བཟླས་པ་བྱ་བ་ལས། བརྡ་སྒྱུར་རྡོ་རྗེའི་བཟླས་པ། དོན་དམ་རྡོ་རྗེའི་བཟླས་པ། འགོག་པ་རྡོ་རྗེའི་བཟླས་པ། བརྡ་སྒྱུར་རྡོ་རྗེའི་བཟླས་པ་ལ་འཇུག་པ་ཨོཾ་དང་གནས་པ་ཨཱཿལྡང་བ་ཧཱུཾ་དུ་ཤེས་པར་བྱ། ཞེས་པ་ལས། དང་པོར་ཨོཾ་སྒྲུབ་པར་བྱ་སྟེ། དང་པོ་ཡི་གེ་དང་པོར་བཞག །དབྱངས་ཡིག་ལྔ་པས་ཡང་དག་བརྒྱན། །སྡེ་ཚན་ལྔ་པས་མཐའ་ཅར་ཅན། །ཞེས་པས་སོ། །ཨཱཿསྒྲུབ་པ་ལ། སྒྲ་མཚམས་སྦྱོར་ལྔ་པ་ལྟར། མཚུངས་པ་དང་འདྲ་བའི་ཡི་གེ་ལས་རིང་པོར་འགྱུར་ཏེ། ཕ་རོལ་འང་དབྱིའོ་ཞེས་པ་དབྱངས་ཡིག་དང་པོ་གཉིས་ལ་བརྟེན་ཏེ་སྒྲུབ། ཧཱུྂ་སྒྲུབ་པ་ལ། ཡ་ར་ལ་སོགས་ཡི་གེ་དགུ། །ཀླད་སྐོར་ང་རུ་ཀླག་པར་བྱ། །ཞེས་པ་དང༌། ཧ་དང་པོ་ལ་ཀླད་སྐོར་རྒྱག་ཅིང་ཞབས་ཀྱུ་བཏགས། ར་དང་ས་དག་གིས་རྣམ་པར་བཅད་བྱའོ། །ཞེས་པ་ལྟར་རྣམས་རྒྱབ་ནས་འཇོག་པས་སོ། །དེ་ལྟར་བཟླས་པ་སོགས། དོན་དམ་རྡོ་རྗེའི་བཟླས་པ་ནི་ཡིད་བཟླས་སོ། །འགོག་པ་རྡོ་རྗེའི་བཟླས་པ་ནི་རླུང་འགོག་པ་སྟེ། ཡོན་པོ་སྒྱུར་བ་བུམ་པ་ཅན་རབ་འབྲིང་ཐ་གསུམ་གྱིས་བསྒྲུབ་པ།

རྡོ་རྗེའི་བཟླས་པའོ། །དེ་ཉིད་ཀྱིས་གསུང་རྡོ་རྗེའི་བཟླས་པ། ཐུགས་ཀྱི་ངོ་བོ་སྒྲུབ་པའི་རྫོགས་རིམ་ཕྲ་ཐིག་གཉིས་ལས། དང་པོ་ཐིག་ལེའི་རྣལ་འབྱོར། སྙིང་ཁར་ཚེ་ཡི་ཕུར་པ་གདབ་པ་སྟེ། ཧཱུྂ་གི་ས་བོན་བསྐྱེད་པའི་ཞབས་ཀྱུ་ལས། ལྕགས་སྒྲོག་འཕྲོས་ཏེ། སྲིན་ལག་བར་དུ་བསྲིངས་ལ། ས་བོན་གྱི་ཡི་གེ་ཉི་ཟླ་གའུ་ཁ་སྦྱོར་གྱི་ཚུལ་དུ་གནས་པའི་བར་དུ་བཅུག་སྟེ། ཕྲ་མོའི་རྣལ་འབྱོར་དམིགས་རྟེན་དེ་ཉིད་ལ་བརྟེན་ཏེ་ནང་དུ་བསམ་པ་བརྟན་པར་བྱེད་པའོ། །དེ་རྣམས་ཐོབ་པའི་བརྒྱུད་པ་ནི། སངས་རྒྱས་སྣང་བ་མཐའ་ཡས། མ་ཅིག་གྲུབ་པའི་རྒྱལ་མོ། ཏི་ཕུ་གསང་སྔགས་སྡོང་པོ། རས་ཆུང་རྡོ་རྗེ་གྲགས་པ། མི་ལ་རྡོ་རྗེ་རྒྱལ་མཚན། དྭགས་པོ་ཟླ་འོད་གཞོན་ནུ། རྗེ་ཕག་མོ་གྲུ་པ། རིན་པོ་ཆེ་མཚན་དཔལ་བཟང་པོ། སྤྱན་སྔ་གྲགས་པ་འབྱུང་གནས། ཡང་དགོན་པ་རྒྱལ་མཚན་དཔལ། སྤྱན་སྔ་རིན་ཆེན་ལྡན། ཟུར་ཕུག་པ་བཀྲ་ཤིས་དཔལ། འབའ་ར་བ་རྒྱལ་མཚན་དཔལ། གྲུབ་ཆེན་རིན་ཆེན་དཔལ། བླ་མ་ཆོས་ཀྱི་དཔལ་བ། མཁན་ཆེན་རྒྱལ་བ་ཕྱག་ན། དེས་བཀའ་དྲིན་མཉམ་མེད་གྲུབ་ཆེན། རིན་པོ་ཆེ་བློ་གྲོས་དཔལ་ལྡན་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་བླ་མ་དེ་ཉིད་ཀྱི་དྲུང་དུ། མཁའ་འགྲོ་མ་ནི་གུ་མ་ནས་བརྒྱུད་པ་ནི་གུ་ཆོས་དྲུག་གི་ལུང་ཚང་མ། ཁྱུང་པོ་རྣལ་འབྱོར་གྱིས་མཛད་པའི་ཡོན་ཏན་རབ་རྫོགས་མ་སོགས་ཐོབ། ཤངས་པ་བཀའ་བརྒྱུད་ནས་རྒྱུད་པའི་དཔལ་མགོན་ཕྱག་དྲུག་པའི་དབང་ལུང༌། མགོན་པོ་སྐུ་ལྔའི་རྗེས་གནང་ཁྲིད་ལུང་དང་བཅས་པ་ཐོབ།། ༈ །།ཡོངས་ཀྱི་དགེ་བའི་བཤེས་གཉེན་ཡོངས་ཀྱི་མགོན། །མཁས་ཤིང་གྲུབ་པ་བརྙེས་པའི་རྣལ་འབྱོར་པ། །ཐོགས་མེད་སྨྲ་བའི་དབང་ཕྱུག་སྤྱན་སྔ་བ། །མང་ཐོས་ཡོན་ཏན་མངའ་བ་ཁྱོད་ཕྱག་འཚལ། །བསླབ་གསུམ་དྲི་མེད་སྦྲང་རྩི་འཛིན་པའི་དཔལ། །བྱང་ཆུབ་སེམས་ཀྱི་བུམ་པ་རབ་རྒྱས་པ། །གསང་སྔགས་འབྲས་བུའི་བཅུད་ཕྱུང་འཕུར་ལྡིང་ཅན། །སྐུ་གསུམ་རྐང་དྲུག་ལྡན་པ་ཁྱོད་ཕྱག་འཚལ། །རིག་པའི་གནས་རྣམས་མ་ལུས་མངོན་དུ་གྱུར། །མཁྱེན་པའི་འོད་ཟེར་གདུལ་བྱའི་གླིང་ལ་ཕོག

།སྙན་པའི་བ་དན་གྲགས་པའི་དཔལ་ཡོན་ཅན། །རྒྱལ་བའི་རྒྱལ་ཚབ་འཛིན་པ་ཁྱོད་ཕྱག་འཚལ། །ཞེས་པས་རྒྱལ་བ་མ་ལུས་པའི་གློག་ཕྲེང་གཅིག་ཏུ་བསྡུས་ནས་དད་པའི་ས་གཞི་བརླན་པར་མཛད་ཏེ། སྐལ་ལྡན་མཐའ་དག་གི་ལུས་སེམས་ཀྱི་དྲི་མ་སྦྱངས་ནས། མངོན་པར་མཐོ་བ་དང་ངེས་པར་ལེགས་པའི་ལམ་ལ་སྦྱོར་བར་མཛད་པ། ཆོས་ཀྱི་རྗེ་ཆོས་ཀྱི་རྒྱལ་པོ་ཆོས་ཀྱི་སྤྱན་དང་ལྡན་པ། དོན་གྱི་སླད་དུ་མཚན་ནས་སྨོས་ན་གསུང་རབ་མང་ཐོས་པའི་ཞལ་མངའ་ནས་ཀྱི་དྲུང་དུ་ཆོས་ཐོབ་པ་ལ། ཐོག་མར་རང་བྱུང་གི་ཡེ་ཤེས་རྟོགས་པར་བྱེད་པ། དུས་གསུམ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་གཤེགས་ཤུལ་དང་ལམ་ལ་གནས་པའི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བགྲོད་པ་གཅིག་པའི་ལམ་མཆོག ཚེ་གཅིག་ལུས་གཅིག་ལ་མཆོག་གི་འབྲས་བུ་སྟེར་བར་བྱེད་པའི་ཉེ་ལམ་དུ་གྱུར་པ། མོས་གུས་གསང་བ་འདུས་པའི་དབང་ཁྲིད། ཞལ་གདམས་དང་བཅས་པ་ལེགས་པར་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཁྱབ་བདག་རྡོ་རྗེ་འཆང༌། རྗེ་ཏིལླི་པ། ནཱ་རོ་ཡེ་ཤེས་དངོས་གྲུབ། མར་པ་ཆོས་ཀྱི་བློ་གྲོས། རྗེ་མི་ལ། མཉམ་མེད་དྭགས་པོ། འགྲོ་མགོན་ཕག་མོ་གྲུ་པ། རྒྱལ་བ་འཇིག་རྟེན་མགོན་པོ། དབོན་བསོད་ནམས་གྲགས་པ། རྒྱལ་བ་གཅུང༌། ཐོག་ཁ་བ་རིན་ཆེན་སེང་གེ། མཚམས་བཅད་པ་གྲགས་པ་བསོད་ནམས། ཆོས་རྗེ་རྡོ་རྗེ་རིན་ཆེན། རྡོ་རྗེ་རྒྱལ་པོ། ཆོས་ཀྱི་རྒྱལ་པོ། དོན་གྲུབ་རྒྱལ་པོ། བླ་མ་ནམ་མཁའ་རྒྱལ་མཚན། འདུལ་འཛིན་ངག་དབང་རྒྱལ་མཚན། མཉམ་མེད་ངག་དབང་གྲགས་པ། མང་ཐོས་གསུང་རབ་པ་དཀོན་མཆོག་རིན་ཆེན་དཔལ་བཟང་པོ། དེས་བདག་ལའོ།། ༈ །།བརྒྱུད་པའི་རིམ་པ་འདི་རྣམས་ཁྲིད་དུ་གསང་བའི་མཚན་འདི་རྣམས་གསང་མཚན་གྱི་བཟླས་པ་བྱེད་དེ། སྲོག་འཁོར་གྱི་བསྙེན་པ་སྔགས་སུ་བྱེད་པ་འདི་ཡིན་ནོ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་དྷྲྀཀ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་པྲཛྙཱ་བྷ་དྲ་སིདྡྷིཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཛྙཱ་ན་སིདྡྷ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་དྷརྨ་མ་ཏི་སིདྡྷི་ཕ་ལ་ཧཱུྂ།

ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་དྷྭ་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་པུཎྱེ་རཏྣ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་ར་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་རཏྣ་ཤྲཱི་བྷ་དྲ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་པུཎྱེ་ཀཱིརྟི་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་ཀཱིརྟི་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་རཏྣ་སིང་ཧ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཀཱིརྟི་པཱུཎྱེ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་རཏྣ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་ར་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་དྷརྨ་ར་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཨཱ་མོ་གྷ་ར་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་བཛྲ་ག་ག་དྷྭ་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཝ་ཀི་ཤྭ་ར་དྷྭ་ཛ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཝ་ཀི་ཤྭ་ར་ཀཱིརྟི་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཏྲ་ཡ་རཏྣ་སིདྡྷི་ཕ་ལ་ཧཱུྂ། ཨོཾ་ཨཱཿགུ་རུ་ཨཱ་ནནྡ་དྷརྨ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་སིདྡྷྀ་ཕ་ལ་ཧཱུྂ། ༈ ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། རྒྱལ་བ་ཐམས་ཅད་ཀྱི་ཐུགས་ཀྱི་ཉིང་ཁུ། ཕྱག་རྒྱ་ཆེན་པོ་ལྷན་ཅིག་སྐྱེས་སྦྱོར་ཉམས་ཁྲིད་དུ་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང༌། ཏིལླི། ནཱ་རོ། མར་པ། དེ་ཕྱིར་སྤྱི་དང་འདྲ་ཡང༌། རྡོ་རྗེ་འཆང༌། ཡེ་ཤེས་ཀྱི་མཁའ་འགྲོ་མ། ཏིལླི་ཕྱིན་འདྲ། ཡང་ན་ནི། རྡོ་རྗེ་འཆང༌། བྱང་ཆུབ་སེམས་དཔའ་བློ་གྲོས་རིན་ཆེན། ས་ར་ཧ། ཤ་བ་རི། རྒྱལ་བ་མངའ་བདག་མཻ་ཏྲི་པ། མར་པ། དེ་ཕྱིན་སྤྱི་དང་འདྲ། རིང་བརྒྱུད་ནི། ལྷོ་ཕྱོགས་ནས་སྒྱུ་ལུས་བར་དོའི་བཀའ་བབ། ཕྱག་ན་རྡོ་རྗེ། ཀླུ་སྒྲུབ། འཕགས་པ་ལྷ། ཟླ་བ་གྲགས་པ། མ་ཏཾ་གི། བྱང་ཕྱོགས་ནས་བྱ་སྤྱོད་རྨི་ལམ་གྱི་བཀའ་བབ་ནི། ལུ་ཧི་པ། ཌེཾ་གི་པ། དྷ་རི་ཀ་པ། སུ་ཀ་དྷ་རི། ཤར་ཕྱོགས་གཏུམ་མོའི་བཀའ་བབ་ནི། སྐལ་བ་བཟང་མོ། ཐང་ལོ་པ། ཤིང་ལོ་པ། པརྞ་རི་པ། ནུབ་ཕྱོགས་ནས་འོད་གསལ་གྱི་བཀའ་བབ་ནི། ཌོཾ་བྷི་པ། བི་ན་པ། ལྭ་བ་པ། ཨིནྡྲ་བྷུ་ཏི་རྣམས་སོ། །དེ་ནས་མར་པ།

མི་ལ། དྭགས་པོ། དུས་གསུམ་མཁྱེན་པ། འགྲོ་མགོན་རས་ཆེན་པ། སྤོམ་བྲག་པ། ཀརྨ་པ། ཨུ་རྒྱན་པ། རང་བྱུང་རྡོ་རྗེ། གཡུང་སྟོན་པ། རོལ་པའི་རྡོ་རྗེ། མཁའ་སྤྱོད་དབང་པོ། དེ་བཞིན་གཤེགས་པ། རཏྣ་བྷ་དྲ། མཐོང་བ་དོན་ལྡན། འཇམ་དཔལ་བཟང་པོ། མ་ཕམ་ཆོས་རྗེ། རབ་འབྱམས་སྨྲ་བ་གསུང་རབ་མང་ཐོས་དཀོན་མཆོག་རཏྣ། དེས་བདག་ལའོ།། ༈ །།ཡང་ན། ཕག་མོ་གྲུ་པ། སྟག་ལུང་ཐང་པ། རིན་ཆེན་མགོན་པོ། ཤེས་རབ་བླ་མ། བཀྲ་ཤིས་བླ་མ། སངས་རྒྱས་དཔལ་བཟང༌། རཏྣ་གུ་རུ། རཏྣ་ཀ་ར། ཀུན་སྤངས་ཆེན་པོ། བཀྲ་ཤིས་དཔལ་བརྩེགས། བྱང་ཆུབ་རྒྱ་མཚོ། རྒྱལ་དབང་ངག་གི་དབང་པོ། གསུང་རབ་རིན་པོ་ཆེ། དེས་བདག་ལའོ། །ཡང་ན། གླིང་རས་པ། རྒྱ་རས། རྒོད་ཚང་པ། ཡང་དགོན་པ། སྤྱན་སྔ་རིན་ལྡན། ཟུར་ཕུག་པ། རྒྱལ་མཚན་འབུམ། ཤེས་རབ་འབུམ། རིན་ཆེན་གཞོན་ནུ། སྤྱན་སྔ་བསོད་ནམས་རྒྱལ་མཚན། ཆོས་རྗེ་བཀྲ་ཤིས་བློ་གྲོས། རབ་འབྱམས་གསུང་རབ་མང་ཐོས་པ། དེས་བདག་ལའོ། །ཡང་རིན་པོ་ཆེའི་དེའི་དྲུང་དུ། རྔོག་ལུགས་ཀྱི་དགྱེས་པ་རྡོ་རྗེ་ལྷ་དགུའི་དཀྱིལ་འཁོར་དུ་དབང་བསྐུར་བ་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང༌། ཏིལླི། ནཱ་རོ། མར་པ། རྔོག་སྟོན་ཆོས་ཀྱི་རྡོ་རྗེ། རྔོག་ཞེ་སྡང་རྡོ་རྗེ། རྔོག་ཀུན་དགའ། གཟི་བརྗིད་གྲགས་པ། རྔོག་རིན་ཆེན་བཟང་པོ། རྔོག་ཆོས་ཀྱི་གྲགས་པ། རྔོག་དོན་གྲུབ་དཔལ། རྔོག་སངས་རྒྱས་ཡོན་ཏན། རྔོག་སྟོན་བྱང་ཆུབ་དཔལ། རྔོག་རིན་ཆེན་དཔལ། རྔོག་བྱང་ཆུབ་དཔལ་གྲུབ། རབ་འབྱམས་གསུང་རབ་མང་པོ་ཐོས་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། བདག་མེད་མ་ལྷ་མོ་བཅོ་ལྔའི་དཀྱིལ་འཁོར་དུ་དབང་བཞི་ཡོངས་རྫོགས་སུ་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྔོག་བྱང་ཆུབ་དཔལ་ཡན་གོང་དང་འདྲ། དེ་ནས་ལོ་ཆེན་བསོད་ནམས་རྒྱ་མཚོ། ཐམས་ཅད་མཁྱེན་པ་ཆོས་ཀྱི་གྲགས་པ་ཡེ་ཤེས་དཔལ་བཟང་པོ། རབ་འབྱམས་སྨྲ་བ་གསུང་རབ་མང་ཐོས་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། མོས་གུས་སྲོག་གི་འཁོར་ལོ་ཐོབ་པའི་བརྒྱུད་པ་ནི།

རྡོ་རྗེ་འཆང༌། ཏིལླི། ནཱ་རོ་པ། མར་པ། མི་ལ་རས་པ། སྒམ་པོ་པ། ཕག་མོ་གྲུ་པ། རྒྱལ་བ་འཇིག་རྟེན་མགོན་པོ། དབོན་བསོད་ནམས་གྲགས་པ། ཆོས་རྗེ་རྡོ་རྗེ་གྲགས་པ། རིན་ཆེན་སེང་གེ། གྲགས་པ་བསོད་ནམས། རྡོ་རྗེ་རིན་ཆེན། རྡོ་རྗེ་རྒྱལ་པོ། རཏྣ་ར་ཛ། དྷརྨ་རཱ་ཛ། ནམ་མཁའ་རྒྱལ་མཚན། ངག་དབང་རྒྱལ་མཚན། མཉམ་མེད་ངག་དབང་གྲགས་པ། གསུང་རབ་རིན་པོ་ཆེ། དེས་བདག་ལའོ།། ༈ །།ཡང་ཕག་མོ་ལྷན་སྐྱེས་ཀྱི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། མཐོང་བ་དོན་ལྡན་ཡན་ཕྱག་ཆེན་རྒྱ་གཞུང་གི་བརྒྱུད་པ་ཕྱི་མ་དང་འདྲའོ། །དེ་ནས་དཔལ་འབྱོར་དོན་གྲུབ། ཐམས་ཅད་མཁྱེན་པ་ཞྭ་དམར་ཅོད་པན་འཛིན་པ་བཞི་པ། གསུང་རབ་རིན་པོ་ཆེ། དེས་བདག་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེའི་དེའི་དྲུང་དུ། གུ་རུ་དྲག་པོའི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས་མཁའ་འགྲོ་ཡེ་ཤེས་མཚོ་རྒྱལ། ཆོས་རྒྱལ་ཁྲི་སྲོང་ལྡེ་བཙན། མངའ་བདག་མྱང༌། ནམ་མཁའ་འོད་ཟེར། ནམ་མཁའ་དཔལ། སྤྲུལ་སྐུ་རྫོང་པ་སྐུ་མཆེད། དཱེ་བ་བྷཱ་དྲ། དྷརྨ་ར་ཛ། སུམ་ལྡན་ཡབ་སྲས། འཇམ་དབྱངས་མགོན། དགའ་རབ་རྡོ་རྗེ། མཁས་གྲུབ་དཔལ་ལྡན་འོད། ཆོས་རྗེ་ཀུན་དགའ་བཟང་པོ། ཆོས་རྗེ་པདྨ་རྒྱལ་པོ། གསུང་རབ་རིན་པོ་ཆེ། དེས་བདག་ལའོ། །ཡང་གོང་གི། ཕག་མོ་ལྷན་སྐྱེས་ཀྱི་དབང་གི་བརྒྱུད་ལུགས་གཅིག་ལ། རྡོ་རྗེ་འཆང༌། རྡོ་རྗེ་ཕག་མོ། ཏིལླི། ནཱ་རོ། མར་པ། མི་ལ། སྒམ་པོ་པ། ཕག་མོ་གྲུ་པ། རྒྱལ་བ་འཇིག་རྟེན་མགོན་པོ། སངས་རྒྱས་དབོན། རྒྱལ་བ་གཅུང༌། རྒྱལ་སྲས་ཐོག་ཁ་བ། སྤྱན་སྔ་མཚམས་བཅད་པ། སངས་རྒྱས་བཅུ་གཉིས་པ། ཆོས་རྗེ་རྡོ་རྗེ་རྒྱལ་པོ། འཛམ་གླིང་ཆོས་ཀྱི་རྒྱལ་པོ། ཆོས་རྗེ་དོན་གྲུབ་རྒྱལ་པོ། བླ་མ་ནམ་མཁའ་རྒྱལ་མཚན། འདུལ་འཛིན་ངག་དབང་རྒྱལ་མཚན། མཉམ་མེད་ངག་དབང་གྲགས་པ། གསུང་རབ་མང་ཐོས་པ། དེས་བདག་ལའོ།། ༈ །།ཡང་ན་རྗེ་སྒམ་པོ་པ་ཡན་གོང་དང་འདྲ་ལ། དེས་ཀརྨ་པ་དུས་གསུམ་མཁྱེན་པ།

རས་ཆེན་པ། སྤོམ་བྲག་པ། ཀརྨ་པཀྵི། ཨུ་རྒྱན་པ། རང་བྱུང་བ། གཡུང་སྟོན་པ། རོལ་པའི་རྡོ་རྗེ། མཁའ་སྤྱོད་དབང་པོ། དེ་བཞིན་གཤེགས་པ། རཏྣ་བྷ་དྲ། མཐོང་བ་དོན་ལྡན། དཔལ་འབྱོར་དོན་གྲུབ། ཐམས་ཅད་མཁྱེན་པ་ཆོས་ཀྱི་གྲགས་པ་ཡེ་ཤེས་དཔལ་བཟང་པོ། གསུང་རབ་རིན་པོ་ཆེ་ཏྲ་ཡཱ་བྷ་དྲ། དེས་བདག་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། བྲམ་ཆེན་པོ་ས་ར་ཧའི་དོ་ཧ་ཐོབ་པ་ལ། རྒྱལ་པོ་དོ་ཧ། བཙུན་མོ་དཽ་ཧ། དམངས་དོ་ཧ། བཤད་པ་ཟིན་བྲིས་དང་བཅས་པ། ཀྱཻ་རྡོ་རྗེའི་རྒྱུད་ཀྱི་རྣམ་བཤད། ཟིན་ཐུན། རྫོགས་རིམ་གྱི་རྣམ་བཤད། ཀྱཻ་རྡོ་རྗེའི་ཆོ་ག་ཤིས་བརྗོད་དང་བཅས་པ་ཐོབ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། གུ་རུ་དྲག་པོའི་དབང་ཆོག་གི་ལུང༌། བྱད་བཟློག་གི་སྲུང་འཁོར་སྒོམ་བཟླས་ཀྱི་དམིགས་པ་སོགས་རྒྱས་པར་ཐོབ་པར་བྱས་སོ།། ༈ །།ན་མོ་གུ་རུ། སློབ་དཔོན་པདྨས་རྗེས་གཟུང་ཞིང༌། །ཡང་དག་དོན་ལ་ངེས་ཤེས་སྐྱེས། །མཁས་དང་གྲུབ་པ་བརྙེས་པ་ཡིས། །ཟླ་བའི་མཚན་ཅན་ལ་ཕྱག་འཚལ། །གསང་སྔགས་གསར་རྙིང་མ་ལུས་ཐུགས་ལ་འཁྲུངས། །ཚེ་གཅིག་མཆོག་གི་སྒྲུབ་པ་ལྷུར་ལེན་ཅིང༌། རིམ་གཉིས་བརྟན་པ་ཐོབ་པའི་རྣལ་འབྱོར་བ། །ཡོངས་རྫོགས་འགྲོ་བའི་མགོན་པོ་ལ་ཕྱག་འཚལ། །ཟབ་གསལ་གཉིས་མེད་བདེ་བའི་གར་བསྒྱུར་ཞིང༌། །ལྷུན་གྲུབ་འབྲས་བུའི་དཔལ་ལ་ལོངས་སྤྱོད་ནས། །འགྲོ་བའི་རེ་སྐོང་སྤྲུལ་པའི་སྐུར་སྟོན་པ། །རྒྱལ་བ་སྐུ་བཞིའི་དབང་པོ་ལ་ཕྱག་འཚལ། །ཁྱོད་ཀྱི་སྐུ་གསུང་ཐུགས་ཀྱི་གཟི་འོད་ཀྱི། །བདག་སོགས་གདུལ་བྱའི་པད་ཚལ་རབ་རྒྱས་པ། །ཆགས་དང་ཐོགས་པ་མེད་པའི་ཤེས་རབ་ཀྱིས། །སྨིན་གྲོལ་དབང་བཞིའི་རྟོགས་པ་བསྐྱེད་དུ་གསོལ། །ཞེས་བསྟོད་པ་དང་སྨོན་ལམ་བཏབ་ནས། །བླ་མ་རིན་པོ་ཆེ་སྦས་པའི་བརྟུལ་ཞུགས་བརྙེས་པ། དོན་གྱི་སླད་དུ་མཚན་ནས་སྨོས་པ་གསང་སྔགས་འཛིན་པ་ཟླ་འོད་ཀྱི་ཞལ་མངའ་ནས་ཀྱི་དྲུང་དུ་ཆོས་ཐོབ་པ་ལ་ཨུ་རྒྱན་གྱི་རྒྱལ་པོ་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པའི་གཅོད་རྙིང་མའི་ཁྲིད་ཐོབ་པའི་བརྒྱུད་པ་ནི།

ཆོས་སྐུ་ཀུན་ཏུ་བཟང་པོ། ཡུམ་ཆེན་ལོངས་སྤྱོད་རྫོགས་སྐུ། སྤྲུལ་སྐུ་བཅོམ་ལྡན་འདས་ཤཱཀྱ་ཐུབ་པ། འཕགས་པ་རྗེས་འཛིན་ཀུན་དགའོ། །འཕགས་མཆོག་ཀླུ་སྒྲུབ་མགོན་པོ། འཆི་མེད་རིག་འཛིན་ཐོད་པ་རྩལ། ཆོས་རྒྱལ་ཁྲི་སྲོང་སྤྲུལ་པའི་སྐུ། ཡུམ་ཆེན་ཤེས་རབ་མཚོ་རྒྱལ། གཏེར་སྟོན་སངས་རྒྱས་གླིང་པ། དེ་སྲས་པདྨ་དབང་ཆེན། གུ་རུ་སངས་རྒྱས་རྡོ་རྗེ། གྲུབ་ཐོབ་དཔལ་ལྡན་རྡོ་རྗེའི་ཞབས། ག་ག་རཏྣའི་ཞབས། དེ་སྲས་ག་ག་བཤེས་གཉེན་དང༌། ཆོས་ཀྱི་རྗེ་ཟླ་བའི་མཚན་ཅན། དེ་གཉིས་ཆར་གྱིས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།གཅོད་རྙིང་མ་ཚང་མའི་ལུང་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཆོས་སྐུ་ཀུན་ཏུ་བཟང་པོ། ལོངས་སྐུ་ཡུམ་ཆེན་མོ། སྤྲུལ་སྐུ་ཤཱཀྱ་ཐུབ་པ། འཕགས་པ་རྗེས་འཛིན་ཀུན་དགའ་བོ། འཕགས་མཆོག་ཀླུ་སྒྲུབ། འཆི་མེད་རིག་འཛིན་ཐོད་པ་རྩལ། ཆོས་རྒྱལ་ཁྲི་སྲོང་ལྡེའུ་བཙན། ཤེས་རབ་ཡུམ་ཆེན་མཚོ་རྒྱལ། གཏེར་སྟོན་སངས་རྒྱས་གླིང་པ། པདྨ་དབང་ཆེན། གུ་རུ་སངས་རྒྱས་རྡོ་རྗེ། གྲུབ་ཐོབ་དཔལ་ལྡན་རྡོ་རྗེ། དྲིན་ཅན་ག་ག་རཏྣ། འཁྲུལ་ཞིག་ཀུན་དགའ་བཤེས་གཉེན། ཆོས་རྗེ་ཟླ་བ་འོད། དེས་དགེ་སློང་རྡོ་རྗེ་འཛིན་པ་ཨཱ་ནནྡ་རཏྣ་དཔལ་བཟང་པོ་ལའོ། །ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། ཚེ་ཁྲིད་རས་ཆུང་ལུགས་ཐོབ་པའི་བརྒྱུད་པ་ནི། སངས་རྒྱས་འོད་དཔག་མེད། མ་ཅིག་གྲུབ་པའི་རྒྱལ་མོ། ཏི་ཕུ་གསང་སྔགས་སྡོང་པོ། རས་ཆུང་རྡོ་རྗེ་གྲགས་པ། མི་ལ་བཞད་པའི་རྡོ་རྗེ། མཉམ་མེད་སྒམ་པོ་པ། བདེ་གཤེགས་ཕག་མོ་གྲུ་པ། རྒྱལ་བ་འཇིག་རྟེན་གསུམ་མགོན། རྒྱལ་བ་ཡང་དགོན་པ། ཆོས་རྗེ་ཟུར་ཕུག་པ། ཐུགས་སྲས་ཆོས་རྗེ་སྤྱན་སྔ་རིན་ལྡན། མཁས་བཙུན་ཡོན་ཏན་རྒྱ་མཚོ། རྟོགས་ལྡན་འཇམ་དཔལ་རྒྱ་མཚོ། ཆོས་རྗེ་ཟླ་བ་འོད། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། གུ་རུ་མཚོ་སྐྱེས་རྡོ་རྗེའི་དབང་ཁྲིད་ལུང་དང་བཅས་པ་ཐོབ་པའི་བརྒྱུད་པ་ནི།

ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས། མཁའ་འགྲོ་ཡེ་ཤེས་མཚོ་རྒྱལ། མངའ་བདག་མྱང་རལ་པ་ཅན། སྲས་མི་བསྐྱོད་རྡོ་རྗེ། འཇམ་དབྱངས་དབང་ཕྱུག་རྒྱལ་མཚན། རྒྱ་སྟོན་ཀུན་དགའ་བརྩོན་འགྲུས། རྣལ་འབྱོར་ཡེ་ཤེས་རྡོ་རྗེ། བླ་མ་དད་པ་དཔལ་བཟང༌། བླ་མ་མཚམས་པ། ཆོས་རྗེ་ཟླ་བ་འོད། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །ཡང་ན་ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས། མཁའ་འགྲོ་ཡེ་ཤེས་མཚོ་རྒྱལ། མྱང་རལ་པ་ཅན། སྲས་མི་བསྐྱོད་རྡོ་རྗེ། མདོ་བོ་རས་པ། མེ་ལོང་རྡོ་རྗེ། མཚན་ལྡན་ཧ་རི་རྡོ་རྗེ། སྣུབས་སྟོན་སངས་རྒྱས། དེས་བླ་མ་དད་པ་བཟང་པོ་ལའོ། །དེ་ཕྱིན་གོང་བཞིན་ནོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོའི་རྒྱུད་ཀྱི་ལུང་ཐོབ་པའི་རིམ་པ་ནི། ཆོས་སྐྱོང་སྤྱིའི་རྒྱུད། རྩ་བའི་རྒྱུད། བཤད་རྒྱུད། མན་ངག་གི་རྒྱུད་དོ། དང་པོ་སྤྱི་རྒྱུད་ལ་བཞི་སྟེ། གྲི་གུག་དང་མཐུན་པ་གྲི་གུག་བརྩེགས་པའི་རྒྱུད། རྟ་མང་དང་མཐུན་པ་མངོན་པར་འབྱུང་བའི་རྒྱུད། བེང་མ་དང་མཐུན་པ་བེང་ཆེན་རོལ་པའི་རྒྱུད། མགོན་པོ་ཐམས་ཅད་དང་མཐུན་པ་རིན་ཆེན་དཔུང་རྒྱན་གྱི་རྒྱུད། འདིའི་སྤྱི་རྒྱུད་ལ་གསུམ་ལས། སྐྱེ་རྒྱུད་གཙོ་བོར་སྟོན་པ་སྲིན་པོ་ཐུགས་རྗེ་འབྱུང་བའི་རྒྱུད་དང༌། དབང་དང་དམ་ཚིག་གཙོ་བོར་སྟོན་པ་བྱ་རོག་བརྩེགས་པ་སྤྱིའི་རྒྱུད་དང༌། ཏིང་ངེ་འཛིན་དང་ལས་ཚོགས་གཙོ་བོར་སྟོན་པ་བུམ་པ་བཟང་པོ་སྤྱིའི་རྒྱུད། རྩ་བའི་རྒྱུད་གསུམ་ནི། སྤུ་གྲི་ཀླད་གཟེར་ནག་པོ་ཡབ་ཀྱི་རྩ་རྒྱུད་དང༌། དྲག་པོ་མཚོན་ཆ་འབར་བ་ལས་མགོན་གྱི་རྩ་རྒྱུད་དང༌། དུག་རླངས་ཁོལ་མ་ཡུམ་གྱི་རྩ་རྒྱུད། བཤད་པའི་རྒྱུད་གསུམ་ནི། ཁྲག་མཚོ་ཁོལ་མ་ཡབ་རྒྱུད་དང༌། སྲོག་གཟན་ནག་པོ་ལས་མགོན་གྱི་རྒྱུད་དང༌། ལྷ་མོ་ཙཎྜི་ཀ་ཡུམ་གྱི་བཤད་རྒྱུད་དོ། །མན་ངག་གི་རྒྱུད་གསུམ་ནི།

ལས་གསང་བ་མན་ངག་སྦས་ཡབ་དང༌། གཞི་སྙིང་སྦས་པའི་རྒྱུད་ལས་མགོན་གྱི་དང༌། མ་མོ་འཇིག་རྟེན་གཙུག་གི་རྒྱུད་ཡུམ་གྱི་མན་ངག་གཙོ་བོར་སྟོན་པའི་རྒྱུད་དོ། །ཐུན་མོང་གི་རྒྱུད་གསུམ་ནི། མཁའ་འགྲོ་མ་སྙིང་རྗེ་ཁྲོས་པའི་རྒྱུད་དང༌། ཁྲོས་མ་སྙིང་པོ་བསྡུས་པའི་རྒྱུད་དང༌། ཁྲོས་མ་ལས་ཚོགས་ཕྲེང་བའི་རྒྱུད་དོ། །བདེ་མཆོག་ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད། དཔལ་ནག་པོ་ཆེན་པོ་བྱ་རོག་གདོང་ཅན་ལུང་བསྟན་པ་ཐུགས་རྗེ་འབྱུང་བའི་རྒྱུད་ལེའུ་བཅུ་བདུན་པ། དཔལ་ནག་པོའི་རྒྱུད་ཆེན་པོ་ཁྲག་མཚོ་ཁོལ་མ་ཞེས་བྱ་བ་ལེའུ་བརྒྱད་ཅུ་གྱ་གཅིག་པ། ཁྲག་མཚོ་ཁོལ་མ་ནང་གི་རྒྱུད་ཆུང་བ་ལེའུ་ཉི་ཤུ་རྩ་གཉིས། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོ་ཕྱག་བཞི་པའི་རྒྱུད། དངོས་གྲུབ་འབྱུང་བ་ལེའུ་ཉི་ཤུ་རྩ་ལྔ་པ། དཔལ་ནག་པོ་ཆེན་པོ་ལས་གསང་བའི་རྒྱུད་ལེའུ་བཅོ་ལྔ་པ། དཔལ་ཡེ་ཤེས་མགོན་པོའི་རྒྱུད་བྱ་རོག་རྩེགས་པ་ལེའུ་ཉི་ཤུ་རྩ་གཅིག་པ། འཕགས་པ་སྤུ་གྲི་ཀླད་གཟེར་ནག་པོ་དངོས་གྲུབ་འབྱུང་བའི་བཀའ་རྒྱུད་ཆེན་པོ་ལེའུ་བཅུ་བདུན་པ།། ༈ །།དཔལ་ནག་པོ་ཆེན་པོ་བྱ་རོག་གདོང་ཅན་གྱི་རྒྱུད་གསང་བ་ཆེན་པོའི་ལེའུ་བཅུ་པ། འཇིག་རྟེན་ལས་ཀྱི་མགོན་པོ་བྱ་རོག་གདོང་ཅན་གྱི་སྲོག་གཟན་ནག་པོ་ནང་གི་རྒྱུད་ལེའུ་བཅུ་བདུན་པ། འཇིག་རྟེན་ལས་ཀྱི་མགོན་པོ་བྱ་རོག་གདོང་ཅན་གྱི་གཞི་སྙིང་སྦས་པ་སྲོག་གཟེར་གསང་བའི་རྒྱུད་ལེའུ་བཅུ་བདུན་པ། ལས་ཀྱི་མགོན་པོ་ཐོད་པ་ནག་པོའི་རྒྱུད་ལེའུ་བཅོ་ལྔ་པ། དཔལ་ནག་པོ་ཆེན་པོ་བྱ་རོག་གདོང་ཅན་གསོད་བྱེད་སྲོག་གི་སྤུ་གྲིའི་རྒྱུད་ལེའུ་ཉི་ཤུ་རྩ་གསུམ་པ། དཔལ་ཤན་པ་ནག་པོའི་རྒྱུད་ལེའུ་བཅུ་བདུན་པ། མཁའ་འགྲོ་མ་བཅུད་འདུས་ཀྱི་རྒྱུད་ལེའུ་དྲུག་པ། གསོད་བྱེད་དམར་པོ་རྩ་འཇིབ་རྐང་འདག་མའི་རྒྱུད་ལེའུ་བཅོ་ལྔ་པ། སློབ་དཔོན་ཀླུ་སྒྲུབ་ཀྱིས་མཁའ་འགྲོ་མ་ལ་དེ་ཕྱི་ལ་ཞུས་པའོ། །སློབ་དཔོན་རྩྭ་མི་སངས་རྒྱས་གྲགས་པས་ཚངས་པ་ཀླུ་འདུལ་གྱི་ལྷ་ཁང་ནས་གཏེར་ནས་ཕྱུང་བ་ཡིན་ནོ། །རེ་མ་ཏིའི་རྒྱུད་ལེའུ་གསུམ་པ།

གྲི་གུག་བརྩེགས་པའི་རྒྱུད་ལེའུ་ཉི་ཤུ་རྩ་གཅིག་པ་རྣམས་ལེགས་པར་ཐོབ་པའི་རྒྱུད་པའི་རིམ་པ་ནི། ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོ། བཅོམ་ལྡན་འདས་ཕྱག་ན་རྡོ་རྗེ། ཨིནྡྲ་བྷུ་ཏི། ལྷ་ལྕམ་ཀླུའི་བུ་མོ། ར་ཧུ་ལ་བྷ་དྲ། ཀླུ་སྒྲུབ། ཨཱརྻ་དེ་ཝ། ཀླུ་བྱང༌། དུས་འཁོར་བ་ཆེ་ཆུང༌། རྡོ་རྗེ་གདན་པ་ཆེ་ཆུང༌། ཨཱ་བྷ་ཡཱ་ཀ་ར། རྩྭ་མི། རྒ་ལོ། རྗེ་ཕག་མོ་གྲུ་པ། འཇིག་རྟེན་གསུམ་གྱི་མགོན་པོ། རྗེ་སྤྱན་མངའ། རྒྱལ་བ། བཅུ་གཉིས་རྙིང་མ་བ། འོ་ལ་བ། བྲག་པ་རིན་རྒྱལ་བ། རྒྱུད་བླ་བ། མགོན་གཞོན་པ། རྡོར་ཡེ་བ། གཞོན་ཤེ་བ། རཏྣ་བྷ་དྲ། མཁན་ཆེན་བསོད་ནམས་བཀྲ་ཤིས། རྟོགས་ལྡན་ངག་དབང་བ། ཡང་ན་པཎ་པ། ལྷ་གཟིག་པ། རཏྣ་བྷ་དྲ། སྤྲུལ་སྐུ་ཤར་སྨད་པ། ཡང་བྲག་པ་རིན་ཆེན་རྒྱལ་བ། ཚེས་བཞི་རྙིང་མ་བ། བཅུ་གཉིས་གསར་མ་བ། ཚེས་བཞི་གསར་མ་བ། ཀུན་སྤངས་པ། ལྷ་གཟིགས་རས་པ། རཏྣ་བྷ་དྲ། མཁན་ཆེན་ཤར་སྨད་པ་བསོད་ནམས་བཀྲ་ཤིས། རྟོགས་ལྡན་ངག་དབང༌། ཡང་ན་བླ་མ་ཞང༌། རིན་ཆེན་གདན་ས་བ། རྟོགས་ལྡན་གཞུང་པ། རིན་པོ་ཆེ་རྒྱལ་སྲས་བ། ཆོས་དབྱིངས་པ། དཀོན་མཆོག་རྡོ་རྗེ། ཆོས་ཀྱི་རྒྱ་མཚོ། རྡོར་ཡེ་བ། གཞོན་ཤེས་པ། རཏྣ་བྷ་དྲ། མཁན་ཆེན་ཤར་སྨད་པ་བསོད་ནམས་བཀྲ་ཤིས། ཡང་རྡོར་ཡེ་བ། པཎ་པ། འཇམ་དབྱངས་པ། རཏྣ་བྷ་དྲ། མཁན་ཆེན་ཤར་སྨད་པ་བསོད་ནམས་བཀྲ་ཤིས། རྟོགས་ལྡན་སངས་རྡོར་བ། བླ་མ་བཟོད་པ་ལེགས་ལྡན་པ། ཆོས་རྗེ་ཟླ་བ་འོད། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་ཡོངས་སུ་རྫོགས་པར་གནང་བ་ཡིན་ནོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། ཡི་དམ་འཁོར་ལོ་སྡོམ་པ་དཔལ་རྒ་ལོ་ལྷ་ལྔའི་དབང་ཐོབ་པའི་རྒྱུད་པ་ནི། རྒྱལ་བ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོ། མགོན་པོ་ཀླུ་སྒྲུབ། ཨཱརྻ་དེ་ཝ། ཨཱ་བྷ་ཡཱ་ཀ་ར། རྩྭ་མི་སངས་རྒྱས་གྲགས་པ། དཔལ་ཆེན་རྒ་ལོ། གྲུབ་ཐོབ་ཁམས་པ་རྒྱ་གར་བ། སྟོན་པ་བཀྲ་ཤིས་འབུམ། བླ་མ་གྲགས་པ་སེང་གེ། བླ་མ་གཞོན་ནུ་ཡེ་ཤེས།

བླ་མ་དམ་རིན་ཆེན། བླ་མ་ཚུལ་ཁྲིམས་དཔལ་བ། བླ་མ་ཤཱཀྱ་སེང་གེ། སྟོན་པ་ངག་དབང་གྲགས་པ། སྤྲུལ་སྐུ་བསོད་ནམས་རྒྱ་མཚོ། བླ་མ་བཟོད་པ་ལེགས་ལྡན་པ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེ་ནས་བདག་གི་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་དགྱེས་ཤིང་བསྩལ་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང༌། དཔལ་རྡོ་རྗེ་ནག་པོ་ཆེན་པོ་ཕྱག་བཞི་པའི་ལས་རིམ་བསྟོད་འབུམ། བླ་མ་ཡི་དམ་གཏེར་མགོན་དང་བཅས་པའི་བསྐང་ཡིག ཧ་སྒྲོག་མ། སྡེ་དཔུང་རིན་ཆེན་སྤུངས་པ་རྒྱས་པ། སྡེ་དཔུང་ཆུང་བ་བསྟན་པ་རིན་ཆེན་རྒྱས་བསྐང༌། བེང་གི་བསྐང་ཡིག ཨ་ཕྱིའི་བསྐང་ཡིག་ནོར་བུ་ཕྲེང་བ་ཆེ་ཆུང༌། ཚེ་རིང་མའི་བསྐང་ཡིག དམ་ཅན་གྱི་བསྐང་ཡིག བཟློག་བསྐང་རིན་ཆེན་སྤུངས་པ་རྣམས་ཐོབ་པའི་བརྒྱུད་པ་ཡང༌། རྒྱལ་བ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང༌། འཕགས་པ་ཀླུ་སྒྲུབ། ཨཱརྻ་དེ་ཝ། ཨཱ་བྷ་ཡཱ་ཀ་ར། སངས་རྒྱས་གྲགས་པ། དཔལ་ཆེན་རྒ་ལོ། གྲུབ་ཐོབ་ཁམས་པ་རྒྱ་གར་བ། སྟོན་པ་བཀྲ་ཤིས་འབུམ། བླ་མ་གྲགས་པ་སེང་གེ། བླ་མ་དར་མ་རིན་ཆེན། བླ་མ་གཞོན་ནུ་ཡེ་ཤེས། བླ་མ་ཤཱཀྱ་སེང་གེ། བླ་མ་ཚུལ་དཔལ་བ། སྟོན་པ་ངག་དབང་གྲགས་པ། སྤྲུལ་སྐུ་བསོད་རྒྱམ་པ། བླ་མ་བཟོད་པ་ལེགས་ལྡན་པ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་བསྩལ་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། ལྷ་མོ་རེ་མ་ཏི་ལྕགས་ཕུར་བདུན་མའི་རྗེས་སུ་གནང་བ་སྒྲུབ་ཡིག་གི་བཟླས་ལུང་ཐོབ་པའི་རྒྱུད་པ་ནི། སློབ་དཔོན་པདྨ་འབྱུང་གནས། སྤུ་རང་པ་དར་མ་རིན་ཆེན། ལ་སྟོད་ཀྱི་ཁྱུང་པོ་སྟག་ཤམ་ཅན། སྐྱུ་ར་རྒྱལ་བ་ཡེ་ཤེས། ཕྱག་བླ་ཆེན་མགོན་པོ། སྟོང་དཔོན་ཤེས་མགོན་པོ། དུས་གསུམ་མཁྱེན་པ། རབ་ཆེན་པ། སྟོང་དཔོན་ཤེས་རབ་འབར། ཀརྨ་པ་དུས་གསུམ་མཁྱེན་པ། ཀརྨ་པཀྵི། དེ་བཞིན་གཤེགས་པ། མཐོང་བ་དོན་ལྡན། རང་བྱུང་རྡོ་རྗེ། ཆོས་རྗེ་རོལ་པའི་རྡོ་རྗེ།

རྗེ་ཆོས་གྲགས་རྒྱ་མཚོ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན་དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་ཐུགས་དགྱེས་ཤིང་གནང་བའོ། །ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོ་བེར་ནག་ཅན་གྱི་དབང་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཡེ་ཤེས་ཀྱི་མགོན་པོ་བེར་ནག་ཅན། སློབ་དཔོན་ཌོཾ་བྷི་ཧེ་རུ་ཀ སློབ་དཔོན་ཨརྞ། མལ་གྱོ་ལོ་ཙཱ་བ། དམྱལ་བ་སུརྻ་ཛྭ་ལ། དམྱལ་བ་ཇོ་སྲས། འགྲོ་མགོན་རས་ཆེན་པ། སྤོམ་བྲག་པ། ཀརྨ་པ། སྙན་རས་དགེ་འདུན་འབུམ། ཆོས་ཀྱི་རྗེ་རང་བྱུང་རྡོ་རྗེ། རྟོགས་ལྡན་རིན་ཆེན་སིངྷ་ཀཱིརྟི། མཁས་གྲུབ་དར་མ་རྒྱལ་མཚན། རྟོགས་ལྡན་མཁའ་སྤྱོད་དབང་པོ། ཆོས་ཀྱི་བཟང་པོ། ཆོས་དཔལ་ཡེ་ཤེས། མཐོང་བ་དོན་ལྡན། དཔལ་འབྱོར་དོན་གྲུབ། ཆོས་གྲགས་རྒྱ་མཚོ། ཟླ་བའི་མཚན་ཅན། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །ཡང་ན་ཡེ་ཤེས་ཀྱི་མགོན་པོ། སློབ་དཔོན་ཌོཾ་བྷི་ཧེ་རུ་ཀ སློབ་དཔོན་ཨརྞ། མལ་གྱོ་ལོ་ཙཱ་བ། དམྱལ་བ་སུརྻ་ཛྭ་ལ། དམྱལ་བ་ཇོ་སྲས། འགྲོ་མགོན་རས་ཆེན་པ། རྒྱལ་སྲས་སྤོམ་བྲག་པ། ཀརྨ་པ། གཉན་རས་དགེ་འདུན་འབུམ། ཆོས་རྗེ་རང་བྱུང་རྡོ་རྗེ། བླ་མ་མགོན་པོ་རྒྱལ་མཚན། ཆོས་རྗེ་རོལ་པའི་རྡོ་རྗེ། ཀུན་སྤངས་ཀུན་དགའ་བློ་གྲོས། རཏྣ་བྷ་དྲ། མཐོང་བ་དོན་ལྡན། དཔལ་འབྱོར་དོན་གྲུབ། རྒྱལ་བའི་དབང་པོ་ཀརྨ་པ་ཆོས་གྲགས་རྒྱ་མཚོ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །ཡང་ན་ནི་རྒྱལ་བ་རྡོ་རྗེ་འཆང་ཆེན་པོ། ཌོཾ་བྷི་ཧེ་རུ་ཀ མལ་གྱོ་ལོ་ཙཱ་བ་བློ་གྲོས་གྲགས་པ། དམྱལ་བ་ཉི་མ་འབར། དམྱལ་བ་ཇོ་སྲས། འགྲོ་མགོན་སངས་རྒྱས་གྲགས་པ། རྒྱལ་སྲས་པུཎྱ་བཛྲ། གྲུབ་ཆེན་ཀརྨ་པཀྵི། གཉན་རས་སངྒ་ལཀྵ། ཆོས་ཀྱི་རྗེ་རང་བྱུང་རྡོ་རྗེ། རྟོགས་ལྡན་ཀུན་དགའ་གཞོན་ནུ། མཁས་གྲུབ་བྱམས་པ་རྒྱལ་མཚན། སྤྲུལ་སྐུ་ཡོན་ཏན་བཀྲ་ཤིས། རྒྱས་སྲས་ཀུན་དགའ་འོད།

ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་སྤྱན་སྔ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་ཐུགས་དགྱེས་ཤིང་གནང་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། སྐྱེས་བུ་དམ་ཅན་གྱི་རྗེས་གནང་ཐོབ་པའི་བརྒྱུད་པ་ནི། བཅོམ་ལྡན་འདས། གསང་བའི་བདག་པོ་ཕྱག་ན་རྡོ་རྗེ། སློབ་དཔོན་པདྨ་འབྱུང་གནས་གཏེར་སྟོན་སངས་རྒྱས་གླིང་པ། སྤྲུལ་སྐུ་མཐོང་བ་དོན་ལྡན། གུ་ཤྲཱི་དཔལ་འབྱོར་དོན་གྲུབ། རྒྱལ་དབང་ཆོས་གྲགས་རྒྱ་མཚོ། ཆོས་ཀྱི་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེ་ཉིད་ཀྱི་དྲུང་དུ། འཇམ་དཔལ་ནཱ་ག་རཀྵའི་རྗེས་སུ་གནང་བ་ཐོབ་པའི་བརྒྱུད་པ་ནི། འཕགས་པ་འཇམ་དཔལ། སློབ་དཔོན་པདྨ་འབྱུང་གནས། ཆོས་རྒྱལ་ཁྲི་སྲོང་ལྡེའུ་བཙན། གཏེར་སྟོན་རིན་ཆེན་གླིང་པ། སྤྲུལ་སྐུ་ནམ་མཁའ་འོད་ཟེར། རི་ཁྲོད་པ་ནམ་མཁའ་ལེགས་པ། བླ་མ་ཆོས་དབྱིངས་པ། རྣལ་འབྱོར་པ་དཔལ་བཟང་པོ། སློབ་དཔོན་ཕུན་ཚོགས་པ། སློབ་དཔོན་རྒྱལ་བ་རིན་ཆེན། སློབ་དཔོན་བླ་མ་རྒྱལ་མཚན། ཞྭ་དམར་ཅོན་པཎ་འཛིན་པ་ཆོས་ཀྱི་གྲགས་པ་ཡེ་ཤེས་དཔལ་བཟང་པོ། ངོ་གྲོ་བ་མཁས་གྲུབ་དབང་ཕྱུག་དཔལ་ལྡན་པ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་གནང་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། དཔལ་མགོན་པོ་ལེགས་ལྡན་གྱི་དབང༌། དེའི་ཕྱི་སྒྲུབ། ནང་སྒྲུབ། གསང་སྒྲུབ། ཆོ་ག་སོགས་ཀྱི་ལུང༌། ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོ་བེང་གི་རྗེས་གནང་སོགས་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྒྱལ་བ་རྡོ་རྗེ་འཆང་ཆེན་པོ། ཀླུ་སྒྲུབ་སྙིང་པོ། ཨཱརྻ་དེ་ཝ། ཨཱ་བྷ་ཡཱ་ཀ་ར། རྩྭ་མི་སངས་རྒྱས་གྲགས་པ། དཔལ་ཆེན་རྒ་ལོ། གྲུབ་ཐོབ་ཁམས་པ་རྒྱ་གར་བ། སྟོན་པ་བཀྲ་ཤིས་འབུམ། བླ་མ་གྲགས་པ་སེང་གེ། བླ་མ་གཞོན་ནུ་ཡེ་ཤེས། བླ་མ་དར་མ་རིན་ཆེན། བླ་མ་སྐྱིད་ཤོད་པ་ཚུལ་དཔལ།

བླ་མ་ཤཱཀྱ་སེང་གེ། སྟོན་པ་ངག་དབང་གྲགས་པ། སྤྲུལ་སྐུ་བསོད་ནམས་རྒྱ་མཚོ། བླ་མ་བཟོད་པ་ལེགས་ལྡན་པ། རྩ་བའི་བླ་མ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་གནང་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། ཇོ་མོ་བཀྲ་ཤིས་ཚེ་རིང་མའི་སྒྲུབ་ཡིག་གི་ལུང་རྗེས་གནང་སོགས་ཐོབ་པའི་བརྒྱུད་རིམ་ནི། རྒྱལ་བ་རྡོ་རྗེ་འཆང་ཆེན་པོ། རྗེ་ཏིལླི་པ། ནཱ་རོ་པ། མར་པ། མི་ལ་བཞད་པའི་རྡོ་རྗེ། ལྷ་རས་པ་ནས་བླ་མ་རྣམས་སྤྱི་འགྲོས་ལྟར་ལ། བླ་མ་སྟག་ལུང་རིན་པོ་ཆེ་ངག་དབང་གྲགས་པ། བླ་མ་འཇམ་དཔལ་རྒྱ་མཚོ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ།། ༈ །།ཡང་རིན་པོ་ཆེའི་དེའི་དྲུང་དུ། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོ་ཕྱག་བཞི་པའི་ལུས་དཀྱིལ་གྱི་དབང་གི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང་ཆེན་པོ། ཀླུ་སྒྲུབ་སྙིང་པོ། ཨཱརྻ་དེ་ཝ། ཨཱ་བྷ་ཡཱ་ཀ་ར། སངས་རྒྱས་གྲགས་པ། དཔལ་ཆེན་རྒ་ལོ། གྲུབ་ཐོབ་ཁམས་པ་རྒྱ་གར་བ། སྟོན་པ་བཀྲ་ཤིས་འབུམ། བླ་མ་གྲགས་པ་སེང་གེ། བླ་མ་གཞོན་ནུ་ཡེ་ཤེས། བླ་མ་དར་མ་རིན་ཆེན། བླ་མ་ཚུལ་ཁྲིམས་དཔལ་བ། བླ་མ་ཤཱཀྱ་སེང་གེ་སྟོན་པ་ངག་དབང་གྲགས་པ། སྤྲུལ་སྐུ་བསོད་ནམས་རྒྱ་མཚོ། བླ་མ་བཟོད་པ་ལེགས་ལྡན་པ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །ཡང་ན་ནི། དཔལ་ཆེན་རྒ་ལོ་ཡན་གོང་དང་འདྲ་ལ། རྒ་ལོས། ཕག་མོ་གྲུ་པ་ལ། དེས་རྒྱལ་བ་འཇིག་རྟེན་གསུམ་གྱི་མགོན་པོ། སྐྱི་ཤོད་པ་བཀྲ་ཤིས་འབུམ། བླ་མ་བྱ་ཚུལ་བ། བླ་མ་མགོན་གཞོན་པ། ཡར་ཀླུངས་པ་སེང་གེ་རྒྱལ་མཚན། དབོན་བོ་རྒྱལ་མཆོག བདག་ཆེན་གྲགས་པ་བློ་གྲོས། མཉམ་མེད་ངག་དབང་གྲགས་པ། སློབ་དཔོན་རིན་པོ་ཆེ་ཐམས་ཅད་མཁྱེན་པ་དཔལ་ལྡན་རིན་ཆེན་པ། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཨཱ་ནནྡ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལའོ། །ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ།

གུ་རུ་ཞི་བ་རྩ་གསུམ་དྲིལ་སྒྲུབ་གུ་རུ་ཆོས་དབང་གིས་བྲག་གནམ་སྐས་ཅན་ནས་བཏོན་པ། མངའ་བདག་ཉང་གི་སྲས་མི་བསྐྱོད་རྡོ་རྗེ་ལ་གནང་བའི་དབང་ལུང་ཐོབ་པའི་བརྒྱུད་པ་རིང་བརྒྱུད་ཀྱི་དབང་དུ་བྱས་པ་ནི། ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས། མཁའ་འགྲོ་ཡེ་ཤེས་མཚོ་རྒྱལ། རྒྱལ་པོ་ཁྲི་སྲོང་ལྡེའུ་བཙན། གཏེར་སྟོན་གུ་རུ་ཆོས་དབང༌། དེས་ཉང་གི་སྲས་མི་བསྐྱོད་རྡོ་རྗེ། མདོ་བོ་རས་པ་སངས་རྒྱས་བཀྲ་ཤིས། དཀོན་མཆོག་བཤེས་གཉེན། རྟོགས་ལྡན་ཀུན་གཞོན་པ། བྱམས་པ་རྒྱལ་ཡོན་ཏན་བཀྲ་ཤིས། རྒྱལ་སྲས་ཀུན་དགའ་འོད། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་གནང་བའོ། ༈ །ཉེ་བརྒྱུད་ནི། ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས། མཁའ་འགྲོ་ཡེ་ཤེས་མཚོ་རྒྱལ། སྲས་མི་བསྐྱོད་རྡོ་རྗེ། མཉམ་མེད་མདོ་བོ་རས་པ། གྲུབ་ཐོབ་མེ་ལོང་རྡོ་རྗེ། མཚན་ལྡན་ཧ་རི་རྡོ་རྗེ། རྣལ་འབྱོར་ཡེ་ཤེས་རྡོ་རྗེ། བླ་མ་དད་པ་དཔལ། བླ་མ་རིན་པོ་ཆེ་མཚམས་ཆེན་པ། ཆོས་རྗེ་ཟླ་བ་འོད་པ། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་ཐུགས་དགྱེས་པ་ཆེན་པོའི་སྒོ་ནས་གནང་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། ཕག་མོ་ཞལ་གཉིས་མའི་དབང་ལེགས་པར་ཐོབ་པར་བྱས་པའི་བརྒྱུད་རིམ་ནི། རྒྱལ་བ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོ། རྒྱལ་སྲས་ཏིལླི་པ། མཁས་མཆོག་ནཱ་རོ་པ། མར་སྟོན་ཆོས་ཀྱི་བློ་གྲོས། མི་ལ་བཞད་པའི་རྡོ་རྗེ། མཉམ་མེད་ཟླ་འོད་གཞོན་ནུ། རྗེ་དུས་གསུམ་མཁྱེན་པ། ཀརྨ་པཀྵི། དེ་བཞིན་གཤེགས་པ། མཐོང་བ་དོན་ལྡན། དཔལ་འབྱོར་དོན་གྲུབ། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཨཱ་ནནྡ་རཏྣ་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་བསྩལ་བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ།

ཕག་མོ་ལྷན་ཅིག་སྐྱེས་མ་ཐོབ་པའི་བརྒྱུད་པ་ནི། རྡོ་རྗེ་འཆང་ཆེན་པོ། རྡོ་རྗེ་རྣལ་འབྱོར་མ། ཏིལླི་པ། ནཱ་རོ་པ། མར་པ། མི་ལ། དྭགས་པོ་ལྷ་རྗེ། དུས་གསུམ་མཁྱེན་པ། འགྲོ་མགོན་རས་ཆེན་པ། སྤོམ་བྲག་པ། གྲུབ་ཆེན་ཀརྨ་པ། མཁས་གྲུབ་ཨུ་རྒྱན་པ། རང་བྱུང་རྡོ་རྗེ། རྒྱལ་བ་གཡུང༌། ཆོས་རྗེ་རོལ་པའི་རྡོ་རྗེ། རྟོགས་ལྡན་མཁའ་སྤྱོད་དབང་པོ། ཆོས་རྗེ་དེ་བཞིན་གཤེགས་པ། རིག་པའི་རལ་གྲི། ཀུན་མཁྱེན་མཐོང་བ་དོན་ལྡན། གུ་ཤྲཱི་དཔལ་འབྱོར་དོན་གྲུབ། རྒྱལ་སྲས་ཀུན་དགའ་འོད། ཆོས་རྗེ་ཟླ་བའི་མཚན་ཅན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་སྤྱན་སྔ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་གནང་བའོ།། ༈ །།ཡང་བླ་མ་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། བདེ་མཆོག་ནམ་མཁའ་མཉམ་རྒྱུད་ཆོས་སྐྱོང་གི་རྒྱུད་འབུམ་གྱི་དབུ་ན་བཞུགས་པའི་རྒྱུད་རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་ཨཱ་བྷ་ཡཱ་ཀ་ར་དང༌། བོད་ཀྱི་ལོ་ཙཱ་བ་རྩྭ་མི་སངས་རྒྱས་གྲགས་ཀྱིས་བསྒྱུར་བ་ཐོབ་པའི་བརྒྱུད་པ་ནི། གོང་དང་མཚུངས་སོ། །ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། གཤིན་རྗེ་ཚེ་བདག་ནག་པོ་ཡང་བཟློག་མེའི་སྤུ་གྲི་ཐོབ་པའི་བརྒྱུད་རིམ་ནི། རྒྱལ་བ་རྡོ་རྗེ་འཆང་ཆེན་པོ། བསྡུ་བ་པོ་དགའ་རབ་རྡོ་རྗེ། འཇམ་དཔལ་བཤེས་གཉེན། གཙུག་ལག་དཔལ་དགེ། ཤནྟིཾ་གརྦྷ། དཱ་ན་ཧཱུ་ལ། ཁྲག་འཐུང་ནག་པོ། བ་སུ་དྷ་ར། ཇོ་བོ་སངས་རྒྱས། ཁུ་ལུང་ཡོན་ཏན་རྒྱ་མཚོ། ཡེ་ཤེས་རྒྱ་མཚོ། སྟོ་སྟོན་ལྕགས་པ། མེས་སྟོན་དྲན་ཆུང༌། དྲན་སྟོན་མཐའ་བྲལ། མེས་སྟོན་དཀོན་རིན། སློབ་སྟོན་བཀྲ་ཤིས་རྒྱལ་མཚན། སློབ་དཔོན་ལྷ་སྒོམ། བླ་མ་གངས་པ། བླ་མ་བྱ་བྲལ་བ། རྒྱལ་བ་གཡུང་སྟོན། སློབ་དཔོན་ཆེན་པོ་གུའི་གུང་པ། མཁས་གྲུབ་བྱམས་པ་རྒྱལ་མཚན། སྤྲུལ་སྐུ་ཡོན་ཏན་བཀྲ་ཤིས། རྒྱལ་སྲས་ཀུན་དགའ་འོད། དེས་རྩ་བའི་བླ་མ་ཟླ་འོད་པ་ལ་གནང༌། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་བསྩལ་

བའོ།། ༈ །།ཡང་རིན་པོ་ཆེ་དེའི་དྲུང་དུ། གུ་རུ་དྲག་པོ་མངའ་བདག་མྱང་གི་གཏེར་ཁའི་དབང་ལུང་ཐོབ་པའི་བརྒྱུད་པ་ནི། ཆོས་སྐུ་སྣང་བ་མཐའ་ཡས། ལོངས་སྐུ་ཐུགས་རྗེ་ཆེན་པོ། སྤྲུལ་སྐུ་པདྨ་འབྱུང་གནས། མངའ་བདག་ཉི་མ་འོད་ཟེར། སྲས་མི་བསྐྱོད་རྡོ་རྗེ། མདོ་བོ་རས་པ། གུ་རུ་སངས་རྒྱས་བཀྲ་ཤིས། བླ་མ་དཀོན་མཆོག་བཤེས་གཉེན། རྟོགས་ལྡན་ཀུན་དགའ་གཞོན་ནུ། མཁས་གྲུབ་བྱམས་པ་རྒྱལ་མཚན། སྤྲུལ་སྐུ་ཡོན་ཏན་བཀྲ་ཤིས། རྒྱལ་སྲས་ཀུན་དགའ་འོད། དེས་ཆོས་རྗེ་ཟླ་བའི་མཚན། དེས་བདག་རྡོ་རྗེ་འཛིན་པ་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ལ་གནང་བའོ། ༈ །ཡང་བླ་མ་དེའི་དྲུང་དུ། སློབ་དཔོན་ཆེན་པོ་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པའི་སྔགས་བཟློག་ཟབ་ཁྱད་ཅན་གྱི་བཟླས་ལུང་ཐོབ་པར་བྱས་སོ། །དེ་ལྟར་གསན་ཡིག་ནོར་བུའི་ཕྲེང་བ་འདི། །བཀོད་པའི་རྣམ་དཀར་དགེ་བ་དྲི་མེད་དེས། །སངས་རྒྱས་བསྟན་པ་ཕྱོགས་བཅུར་རྒྱས་པ་དང༌། །སྒྲུབ་པ་བསམ་གཏན་ཀློག་པ་ཐོས་པ་དང༌། །བསམ་སོགས་ཆོས་སྤྱོད་བཅུ་ཡིས་བསྒྲུབས་པ་དག །ནམ་མཁའི་མཐའ་ཀླས་མཉམ་པའི་བསོད་ནམས་དེ། །འཁོར་བ་མ་སྟོངས་བར་དུ་འཕེལ་གྱུར་ཅིག །གང་ཞིག་སྨོན་ལམ་གྲུབ་པ་ཞེས། །བླ་མ་དང་ནི་ལྷག་པའི་ལྷ། །ཕྱོགས་བཅུའི་སངས་རྒྱས་ཆོས་དང་འཕགས་ཚོགས་བྱང་སེམས་རང་རྒྱལ་ཆེ་རྣམས་དང༌། །དངོས་གྲུབ་སྟེར་དང་བར་ཆད་སེལ་མཛད་དཔའ་བོ་མཁའ་འགྲོ་ཆོས་སྐྱོང་སྲུང་མའི་ཚོགས། །མ་ལུས་ལུས་པ་མེད་དང་ཞིང་ཁམས་ཀུན་སྣང་རྟེན་སྐུ་ཡེ་ཤེས་མཛད་པ་ཕྲིན་ལས་འབད་རྩོལ་བྲལ་བ་ལྷུན་གྱིས་གྲུབ་པ་རྣམས། །བདག་སོགས་ལུས་ཅན་ནམ་མཁའི་མཐའ་ཀླས་བསོད་ནམས་ཞིང་དང་སྨོན་ལམ་གྲུབ་པའི་རྟེན་དཔང་ཉིད་དུ་གསོལ་འདེབས་ན། །ཡིད་བཞིན་རེ་བ་བསྐོང་ཕྱིར་དང་པོའི་ཐུགས་བསྐྱེད་ཡོངས་རྫོགས་སྙེལ་བ་མི་མངའ་འགྲོ་དོན་ཕྱིར། །ཐབས་སྟོབས་སྨོན་ལམ་ཡེ་ཤེས་མཛད་པའི་དགོངས་པ་མྱུར་འགྲུབ་ཤོག

།ཡང་དག་དོན་ལ་སྦྱོར་དང་འགལ་བའི་ཕྱོགས་འཛིན་བརྩོན་པ་ལྷུར་བླངས་པ། །ཆོག་པར་མི་འཛིན་བྱེད་པའི་གོ་བགོས་བདུད་ཀྱི་རྒྱུ་འདུལ་བརྩོན་པ་མྱུར་འགྱུར་ཤོག །གཟུགས་མཐོང་འོག་ཏུ་བསམ་པ་བརྟན་བྱེད་ལྷག་པ་མཐོང་བ་ཐོབ་བྱེད་པ། །ཟབ་དོན་ཆགས་བྲལ་ཞི་བ་དོན་དམ་སྙིང་པོ་མྱུར་ཐོབ་ཤོག །རྣམ་པར་འབྱེད་པའི་ཤེས་རབ་བདེན་གཉིས་ཟུང་འཇུག་རྟོགས་བྱེད་པ། །ཆོས་ཀྱི་རང་བཞིན་མ་འདྲེས་སོ་སོར་རྟོག་པའི་གོ་འཕང་མཆོག་ལ་སྦྱོར་བར་ཤོག །སྟོབས་བཅུའི་རྒྱུ་འབྱུང་ཐབས་དང་སྨོན་ལམ་སྟོབས་ལྡན་ཡེ་ཤེས་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་པས། ཚོགས་སྦྱོར་མཐོང་སྤང་སྒོམ་སློབ་མཐར་ཕྱིན་ལམ་ལྔ་ཡོངས་ཐོབ་ཤོག །རྒྱུ་འབྲས་གཞི་ལམ་བསྐྲུན་པའི་མཐུ་ལས་རབ་དགའ་དྲི་མེད་འོད་ལྡན་དང༌། །བཞི་པ་འོད་འཕྲོ་སྦྱང་དཀའ་དག་དང་མངོན་གྱུར་རིང་འགྲོ་བ། །བརྒྱད་པ་མི་གཡོ་བློ་གྲོས་ཆོས་ཀྱི་སྤྲིན་ཅན་ཀུན་ཏུ་འོད་དང་མ་ཆགས་རྡོ་རྗེ་འཛིན་པ་ས་ལམ་ལེགས་ཐོབ་ཤོག །འདི་ལྟར་བསྒྲུབས་པའི་འབྲས་བུ་རང་གཞན་གཉིས་ཀྱི་དོན་ཐོབ་ཅིང༌། འཁོར་བ་མུ་མཐའ་མེད་པའི་སེམས་ཅན་རྣམས་ལ་སྨིན་གྱུར་ཅིག །བསམ་པའི་སྟོབས་དང་སྨོན་ལམ་བྱེ་བའི་སྟོབས་དག་དང༌། །རྒྱུ་སྟོབས་རྩ་བའི་སྟོབས་བཞི་དག་གིས་བསྒྲུབས་པ་ཡིས། །འབྲས་བུ་སྐུ་བཞིའི་དཔལ་ལ་ལོངས་སྤྱོད་ཟུང་འཇུག་རྫོགས་པའི་སྐུ། །ནམ་མཁའི་མཐས་ཁྱབ་སེམས་ཅན་རྣམས་ཀྱིས་མྱུར་ཐོབ་ཤོག །གང་ཡང་དུས་གསུམ་བསགས་དང་རང་བཞིན་ཡོད་པའི་དགེ་བ་དེས་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་བླ་ན་མེད་པའི་གོ་འཕང་མྱུར་ཐོབ་ཤོག། །ཡེ་དྷརྨ་ཧེ་ཏུ་པྲ་བྷ་ཝ་ཧེ་ཏུནྟེ་ཥཱནྟ་ཐཱ་ག་ཏོ་ཧྱ་ཝདཏ་ཏེ་ཥཱཉྩ་ཡོ་ནི་རོ་དྷ་ཨེ་ཝཾ་བ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎ་ཡེ་སྭཱ་ཧཱ།། དགེའོ།། །།
