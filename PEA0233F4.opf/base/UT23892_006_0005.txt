རྣལ་འབྱོར་པ་གང་ཞིག་ཨ་ཕྱིའི་སྒྲུབ་ཐབས། རིན་པོ་ཆེའི་ཞལ་མངའ་ནས་མཛད་པ། རྣལ་འབྱོར་པས་རྒྱུན་དུ་ཉམས་སུ་ལེན་པ། རིན་པོ་ཆེའི་ཅོད་པན་ལེའུ་བཅོ་ལྔ་མ་ནས་བཏུས་པའོ། །གསོལ་ཁ་བྱེད་པར་འདོད་པས། དུས་བཟང་དང་ཚེས་བཅུ་ལ་སོགས་པ་ལ་དབེན་པའི་གནས་སུ་གཙང་སྦྲ་ལེགས་པར་བྱས་ཏེ། རྟེན་གཏོར་ལ་སོགས་པའི་མདུན་ནམ། མེད་ན་རྗེ་བཙུན་མ་དང་ཨ་ཕྱིའི་ཐང་སྐུ་བཀྲམ་ལ། མཆོད་པ་རྣམ་པ་ལྔ་བཤམས། ཕྱེ་ཕུད་གཙང་མ་ལ་གཏོར་མ། རབ་ལྔ། འབྲིང་གསུམ། ཐ་མ་གཅིག་བྱས་ལ། ཤ་ཁྲག་གིས་ལེགས་པར་བརྒྱན། སིནྡྷུ་ར་དང་དྲི་བཟང་དང༌། ཟས་སྣ་ཚོགས་དང༌། ཤིང་ཐོག་སྣ་ཚོགས་དང༌། ཨམྲྀ་ཏ་རྣམས་དང༌། མདའ་དར་སྣ་ལྔ་ཞུགས། ཆབ་དང་ཆང་ཕུད་གཙང་མ་བྲན། དེའི་མདུན་དུ་ལུག་ཤ་ལག་པ་གཡས་པ། ཟས་རྣམ་པ་སྣ་ཚོགས་པ་གང་འཛོམས་ཀྱི་ཚོགས་ཆས། སྤོས་དཀར་ནག་སོགས་མཆོད་གཏོར་ཐམས་ཅད་བསགས། གསང་སྔགས་ཀྱི་རྫས་རྣམས་ལེགས་པར་ཚོགས་པར་བྱས་ལ། དེ་ནས། རང་ཉིད་རྡོ་རྗེ་རྣལ་འབྱོར་ཞལ་གཉིས་མ་གྲི་གུག་དང་ཐོད་ཁྲག་འཛིན་པ། གཡས་བསྐུམས་ཤིང་གཡོན་བརྐྱངས་པའི་སྟབས་ཀྱིས་སྣ་ཚོགས་པདྨ་དང་ཉི་མ། རོའི་གདན་ལ་གནས་པ་སྐད་ཅིག་གིས་ཡོངས་སུ་རྫོགས་པར་གསལ་བར་གྱུར། ཞེས་བརྗོད་ནས། ནང་མཆོད་བྱིན་གྱིས་བརླབ་པ་ནི། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལཱི་ཧ་ན་ཧ་ན་ཧཱུྂ་ཕཊ྄། ཨོཾ་སྭ་བྷཱ་ཝ་ཤུདྡྷཿསརྦ་དྷརྨཱཿསྭ་བྷཱ་ཝ་ཤུདྡྷོ྅ཧཾ། སྟོང་པ་ཉིད་དུ་གྱུར། སྟོང་པའི་ངང་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ། ཨ་ལས་ཐོད་པ་ཕྱི་དཀར་ལ་ནང་དམར་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ། ཤར་དུ་གོ་ལས་བ་གླང་གི་ཤ ལྷོར་ཀུ་ལས་ཁྱིའི་ཤ ནུབ་ཏུ་དྷ་ལས་རྟའི་ཤ བྱང་དུ་ཧ་ལས་གླང་པོ་ཆེའི་ཤ དབུས་སུ་ན་ལས་མིའི་ཤ དེ་རྣམས་ཀྱི་སྟེང་དུ། ཤར་དུ་བྷྲཱུཾ་ལས་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས། ལྷོར་ཨཾ་ལས་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས། ནུབ་ཏུ་ཛྲཱིཾ་ལས་སོ་སོར་རྟོགས་པའི་ཡེ་ཤེས། བྱང་དུ་ཁཾ་ལས་བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས།

དབུས་སུ་ཧཱུྂ་ལས་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས། དེ་རྣམས་ཀྱི་སྟེང་དུ་ཨོཾ་དཀར་པོ་མགོ་ཐུར་དུ་བསྟན་པ། དེ་ལས་འོད་ཟེར་འཕྲོས། རླུང་གཡོས། མེ་སྦར། ཐོད་པའི་ནང་གི་རྫས་རྣམས་ཞུ་ཞིང་ཁོལ་བའི་རྣམ་པ་ཅན། དཀར་ལ་དམར་བའི་མདངས་ཆགས་པར་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ་ཧ་ཧོཿཧྲིཿ ལན་གསུམ་བརྗོད། མཆོད་པ་ལ། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལཱི་ཧ་ན་ཧ་ན་ཧཱུྂ་ཕཊ྄། ཀྱིས་བསངས། ཨོཾ་སྭ་བྷཱ་ཝ་ཤུདྡྷཿསརྦ་དྷརྨཱཿསྭ་བྷཱ་ཝ་ཤུདྡྷོ྅ཧཾ། གིས་སྦྱངས། སྟོང་པ་ཉིད་དུ་གྱུར། སྟོང་པའི་ངང་ལས་ཨོཾ་ལས་རིན་པོ་ཆེའི་སྣོད་ཡངས་ཤིང་རྒྱ་ཆེ་བ་རྣམས་ཀྱི་ནང་དུ། ཨོཾ་འོད་དུ་ཞུ་བ་ལས་བྱུང་བའི་ལྷ་རྫས་ལས་གྲུབ་པའི་མཆོད་ཡོན། ཞབས་བསིལ། མེ་ཏོག བདུག་སྤོས། མར་མེ། དྲི་ཆབ། ཞལ་ཟས། རོལ་མོ་རྣམས་དྭངས་ཤིང་ཐོགས་པ་མེད་པ་ནམ་མཁའ་དང་མཉམ་པར་གྱུར། ཨོཾ་ཨརྒྷཾ་ཨཱཿཧཱུྂ། ཞེས་པ་ནས། ཤཔད འི་བར་དང༌། གཏོར་མ་ལ། ཨ་མྲྀ་ཏ ས་བསངས། སྭ་བྷཱ་ཝ ས་སྟོང་པར་སྦྱངས། སྟོང་པའི་ངང་ལས་ཀཾ་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་ལ་གནས་པའི་ཨ་ལས་ཐོད་པ་ཕྱི་དཀར་ལ་ནང་དམར་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ། ཤ་ལྔ་དང་བདུད་རྩི་ལྔའི་རང་བཞིན་དུ་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ། ཧ་ཧོཿཧྲཱིཿ ཞེས་ལན་གསུམ་དང༌། ཕེཾ་༌། བླ་མ་ཡི་དམ་སངས་རྒྱས་བྱང་སེམས་དཔའ་བོ་མཁའ་འགྲོ་ཆོས་སྐྱོང་སྲུང་མའི་ཚོགས་དང་བཅས་པ་སྤྱན་དྲངས། བཛྲ་ས་མཱ་ཛཿ དྲིན་ཅན་རྩ་བའི་བླ་མ་ ནས་བརྒྱུད་པའི་བླ་མའི་མཚན་གྱི་མཇུག་ཏུ། ཨི་དཾ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི ས་འབུལ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཨི་དཾ། ཞེས་གོང་བཞིན་འགྲེ། ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ། བཛྲ་ཧ་རི་ནི་ས་མ་མ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཏཱ་ར་དྷེ་མ་ཌཱ་ཀི་མ་མ་བྷ་ཛཾ་ཨི་དཾ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སོགས་ཀྱིས་ཨ་ཕྱི་གཙོ་འཁོར་ལ་འབུལ། དེ་ནས། ལྷ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཀླུ་ཡི་ཚོགས་ཚོགས་རྣམས་མ་ལུས་དང༌། །གནོད་སྦྱིན་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྲིན་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །འབྱུང་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཡི་དྭགས་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཤ་ཟའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྨྱོ་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །བརྗེད་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །མཁའ་འགྲོ་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་མོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་ལུས་པ་ནི་ཐམས་ཅད་དག

།འདིར་གཤེགས་བདག་ལ་དགོངས་སུ་གསོལ། །མཆོད་སྦྱིན་གཏོར་མ་འདི་བསྣོམས་ལ། །རྣལ་འབྱོར་བདག་ཅག་འཁོར་བཅས་ལ། །ནད་མེད་ཚེ་དང་དབང་ཕྱུག་དང༌། །དཔལ་དང་གྲགས་དང་སྐལ་པ་བཟང༌། །ལོངས་སྤྱོད་རྒྱ་ཆེན་ཀུན་གྲུབ་ཤོག །ཞི་དང་རྒྱས་ལ་སོགས་པ་ཡི། །ལས་ཀྱི་དངོས་གྲུབ་བདག་ལ་སྩོལ། །སྲུང་མས་རྟག་ཏུ་གྲོགས་མཛོད་ཅིག །དུས་མིན་འཆི་དང་ནད་རྣམས་དང༌། །གདོན་དང་བགེགས་རྣམས་མེད་པར་མཛོད། །རྨི་ལམ་ངན་དང་མཚན་མ་ངན། །བྱད་སྟེམས་ངན་པ་མེད་པར་མཛོད། །འཇིག་རྟེན་བདེ་ཞིང་ལོ་ལེགས་དང༌། །འབྲུ་རྣམས་རྒྱས་ཞིང་ཆོས་འཕེལ་དང༌། །བདེ་ལེགས་ཕུན་སུམ་ཚོགས་པ་དང༌། །ཡིད་ལ་འདོད་པ་ཀུན་འགྲུབ་ཤོག ཞེས་སོགས་འདོད་དོན་དུ་གསོལ་བ་བཏབ། དེ་ནས་གཞི་བདག་གི་གཏོར་མ་ལ་དམིགས་ཏེ། ཨ་མྲྀ་ཏ ས་བསངས། སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་པ་ཉིད་དུ་གྱུར། ཟག་པ་མེད་པའི་བདེ་བ་ཁྱད་པར་ཅན་བསྐྱེད་པ་རྣམས་སུ་གྱུར། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱཿཎཱཾ་ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ་ཨོཾ་ཨཱཿཧཱུྂ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རིན་ཆེན་མང་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་པ་གཟུགས་མཛེས་དམ་པ་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་པ་སྐུ་འཇམ་ཀླས་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་འཇིགས་པ་ཐམས་ཅད་དང་ཡོངས་སུ་བྲལ་བ་ལ་ཕྱག་འཚལ་ལོ། ན་མཿསརྦ་ཏ་ཐཱ་ག་ཏ་ཛྙཱ་ན་ཨ་ཝ་ལོ་ཀི་ཏེ་ཨོཾ་སམྦྷ་ར་སམྦྷ་ར་ཧཱུྂ། མཆོད་སྦྱིན་གྱི་གཏོར་མ་རྒྱན་ཁ་དོག་དྲི་རོ་རེག་བྱ་ནུས་པ་ཕུན་སུམ་ཚོགས་པ་འདོད་པའི་ཡོན་ཏན་ལྔ་དང་ལྡན་པ་འདི་འབུལ་ཞིང་བསྔོ་བའི་གནས་སུ་གྱུར་པ་ནི། ས་བདག་སའི་ལྷ་མོ་བསྟན་མས་གཙོ་བྱས་པའི། སྟོད་མངའ་རིས་སྐོར་གསུམ། བར་དབུས་གཙང་རུ་བཞི། སྨད་མདོ་ཁམས་སྒང་དྲུག སྲིད་པ་ཆགས་པའི་ལྷ་དགུ། རྗེ་ཡི་མགུར་ལྷ་བཅུ་གསུམ། བོད་ཁམས་སྐྱོང་བའི་བསྟན་མ་བཅུ་གཉིས། མཚོ་མ་ཕམ། ཙཱ་རི་ཙཱ་གོང༌། ལ་ཕྱི་ཆུ་བར། གསེར་གྱི་རི་བདུན། རོལ་བའི་མཚོ་བདུན། རྒྱ་མཚོའི་མཐའ་དབུས།

རི་རབ་ཀྱི་སྟེང་ཤོད། སྟོང་གསུམ་འཛམ་གླིང་ན་གནས་པའི་གཞི་བདག་གཉུག་མར་གནས་པ་རྣམས་དང༌། ཁྱད་པར་དུ་ཡང༌། ཆོས་རྗེའི་དགེ་བསྙེན་བར་ལྷ་ལྷ་དང་བཅས་པ། སོག་ར་བཙན་དང་བཅས་པ། ཏི་སྒྲོ་སྨན་དང་བཅས་པ། ཆུ་འཕེན་ཀླུ་དང་བཅས་པ། ཙོག་སྣའི་རྣམ་རྒྱལ་དཀར་པོ། ས་རི་ས་སྐོར། ལུང་པ་ཕུ་མདའ། ས་ཕྱོགས་ཡུལ་སྐོར་འདི་ཀུན་ཏུ་གནས་པའི་གཞི་བདག་གཉུག་མར་གནས་པ་དཀར་པོའི་ཕྱོགས་ལ་མངོན་པར་དགའ་བ་རྣམས་ལ་མཆོད་གཏོར་འབུལ་ལོ། །བཞེས་སུ་གསོལ་ལོ། །བཞེས་ནས་ཀྱང་རྣལ་འབྱོར་པ་བདག་ཅག་འཁོར་དང་བཅས་པ་འགྲོ་ན་མ་བཤོལ། འདུག་ན་མ་བསྒུལ། འཚོ་བ་སྦྱོར། ཡོན་བདག་གྱིས། ཆོས་དང་མཐུན་པའི་སྡོང་གྲོགས་རྒྱ་ཆེན་པོ་མཛད་དུ་གསོལ། དེ་ནས་བགེགས་གཏོར་ལ་དམིགས་ལ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རིན་ཆེན་མང་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་པ་གཟུགས་མཛེས་དམ་པ་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་པ་སྐུ་འཇམ་ཀླས་ལ་ཕྱག་འཚལ་ལོ། དེ་བཞིན་གཤེགས་འཇིགས་པ་ཐམས་ཅད་དང་ཡོངས་སུ་བྲལ་བ་ལ་ཕྱག་འཚལ་ལོ། ན་མཿསརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་ཝ་ལོ་ཀི་ཏེ་ཨོཾ་སཾ་བྷ་ར་སཾ་བྷ་ར་ཧཱུྂ། ལན་གསུམ་བརྗོད་ལ། བགེགས་ཀྱི་རྒྱལ་པོ་བི་ནཱ་ཡ་ཀས་གཙོ་བྱས་པའི་བགེགས་རིགས་སྟོང་ཕྲག་བརྒྱད་ཅུ། བྱིས་པའི་གདོན་ཆེན་བཅོ་ལྔ། མི་འདོད་པ་བསྐྱེད་པར་བྱེད་པ། དགེ་བ་ལ་བར་དུ་གཅོད་པར་བྱེད་པའི་སྟེང་གདོན་གཟའ། འོག་གདོན་ཀླུ། བར་གདོན་བཙན་ལ་སོགས་པ་གདོན་བགེགས་ཀྱི་རིགས་སུ་གྱུར་པ་ཐམས་ཅད་ལ་འབུལ་ལོ། འདོད་པའི་ཡོན་ཏན་ལྔ་དང་ལྡན་པ་འདིས་ཚིམ་པར་གྱིས་ལ་གནོད་པ་དང་འཚེ་བའི་བསམ་སྦྱོར་ངན་པ་ཐམས་ཅད་ཞི་ནས་ཕན་པ་བདེ་བ་བྱང་ཆུབ་ཀྱི་སེམས་དང་ལྡན་པར་གྱུར་ཅིག སརྦ་བྷཱུ་ཏ་གཙྪ། ཞེས་བརྗོད། ཛཿའདི་ལྟར་ལྷ་མོ་ཡ་མཚན་ཅན། །ལྷ་མོའི་ལུས་ལ་ཁྭ་གདོང་མ། །བཛྲ་ཀཾ་ཀ་ཞེས་བྱ་བའི། །ཤར་སྒོ་སྲུང་བའི་ཕྲ་མེན་མ། །ཛཿཕྲ་མེན་རྡོ་རྗེའི་དུས་ལ་བབ། །ཤར་སྒོ་སྲུང་བའི་ལས་མཛོད་ཅིག

།ཛཿའདི་ལྟར་ལྷ་མོ་ཡ་མཚན་ཅན། །ལྷ་མོའི་ལུས་ལ་འུག་གདོང་མ། །བཛྲ་ཨུལླུ་ཞེས་བྱ་བའི། །བྱང་སྒོ་སྲུང་བའི་ཕྲ་མེན་མ། །ཛཿཕྲ་མེན་རྡོ་རྗེའི་དུས་ལ་བབ། །བྱང་སྒོ་སྲུང་བའི་ལས་མཛོད་ཅིག །ཛཿའདི་ལྟར་ལྷ་མོ་ཡ་མཚན་ཅན། །ལྷ་མོའི་ལུས་ལ་ཁྱི་གདོང་མ། །བཛྲ་ཤྭ་ན་ཞེས་བྱ་བའི། །ནུབ་སྒོ་སྲུང་བའི་ཕྲ་མེན་མ། །ཛཿཕྲ་མེན་རྡོ་རྗེའི་དུས་ལ་བབ། །ནུབ་སྒོ་སྲུང་བའི་ལས་མཛོད་ཅིག །ཛཿའདི་ལྟར་ལྷ་མོ་ཡ་མཚན་ཅན། །ལྷ་མོའི་ལུས་ལ་ཕག་གདོང་མ། །བཛྲ་སུ་ཀ་ཞེས་བྱ་བའི། །ལྷོ་སྒོ་སྲུང་བའི་ཕྲ་མེན་མ། །ཛཿཕྲ་མེན་རྡོ་རྗེའི་དུས་ལ་བབ། །ལྷོ་སྒོ་སྲུང་བའི་ལས་མཛོད་ཅིག །ཛཿའདི་ལྟར་ལྷ་མོ་ཡ་མཚན་ཅན། །སྒོ་མ་རྣམས་ཀྱིས་སྒོ་སྲུངས་ལ། །མཚམས་མ་རྣམས་ཀྱིས་མཚམས་སྲུངས་ཤིག །མཚམས་བཅད་གནས་འདི་ཚུན་ཆད་དུ། །བགེགས་རྣམས་འདའ་བར་མ་བྱེད་ཅིག ། ཅེས་མཚམས་བཅད། དེ་ནས་སྐད་ཅིག་གིས་སྟོང་པའི་ངང་ལས། བྷྲཱུཾ་ཡོངས་སུ་གྱུར་པ་ལས། འོག་ཟངས་མའི་འཁོར་ལོ་རྩིབས་བརྒྱད་ལྟེ་བར་རྡོ་རྗེ་རྒྱ་གྲམ། དེའི་འོག་ཏུ་མུན་ནག་གི་གྲུ་གསུམ་ཟབ་པ་གཏིང་རིང་བ་ཞིག་བསྒོམ། དེའི་ནང་དུ་ཡི་གེ་ནྲྀ་ཡོངས་སུ་གྱུར་པ་ལས། བྱད་མའི་གཟུགས་རིད་ཅིང་སྐེམ་པ་ཞིག་བསམ། གདོན་བགེགས་ཐམས་ཅད་ཁྲོ་བོའི་ང་རྒྱལ་གྱིས་བཀུག་ལ་བསྟིམ་ནས་འགུལ་བསྐྱོད་མི་ནུས་པར་བསམ། དེའི་སྟེང་དུ་རྡོ་རྗེའི་ལྟེ་བ་ནས་ཡར་ཀ་བ་བསླང་བའི་ཚུལ་ལྟ་བུ། ཤེལ་གྱི་རྡོ་རྗེའི་ལྟེ་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བ་ཞིག་བསམ། དེའི་དབུས་སུ་རང་ཉིད་རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མ་ཚོན་གང་པ་རྒྱན་ཆ་ལུགས་མངོན་རྟོགས་དང་མཐུན་པ། ཡེ་ཤེས་ཀྱི་མེ་དཔུང་གི་དབུས་ན་བཞུགས་པར་བསྒོམ། དེའི་ཐུགས་ཀར་ཨ་ཕྱིའི་ས་བོན་ཧྲཱིཿདཀར་པོ་ཚེག་དྲག་དང་ལྡན་པའི་གཡས་བསྐོར་ལ་རང་གི་མིང་གིས་བསྐོར་བར་བསྒོམ། ཤེལ་གྱི་རྡོ་རྗེའི་སྟེང་གསེར་གྱི་འཁོར་ལོ་རྩིབས་བརྒྱད་པ་ཞིག་གིས་ཐོག་ཕུབ་པ་བཞིན་བསྒོམས་པ་དང༌།

ཟངས་མའི་འཁོར་ལོ་གཉིས་ཁྲིག་གིས་བསྒྲིགས་པའི་བར་དང་གསེང་ལ་རླུང་སེར་བུ་ཙམ་ཡང་མི་ཐར་བར་བསྒོམས་པའི་ནང༌། ཤེལ་གྱི་རྡོ་རྗེ་ལ་བསྐོར་བ་བྱེད་པའི་སྒྲ་དི་རི་རི་སྒྲོགས་པར་བསམ། བསྲུང་བའི་སྙིང་ཏིག་ཤིན་ཏུ་གཅེས་པ་རྣམས་ཤེལ་གྱི་རྡོ་རྗེ་དང་མཁའ་འགྲོ་མའི་བར་དེར་བསྒོམ། རྡོ་རྗེ་དང་རིན་པོ་ཆེ་དང༌། པདྨ་དང༌། རལ་གྲི་དང༌། མེ་རི་རྣམས་ཀྱི་གུར་ཁང་བསྒོམ། མེ་རིའི་གུར་ཁང་གི་ནང་དུ་ལས་ཀྱི་མཁའ་འགྲོའི་ཚོགས་དང༌། ཤ་ཟ་མཁའ་འགྲོ་བསམ་གྱིས་མི་ཁྱབ་པས་མཚོན་ཆ་སྣ་ཚོགས་ཐོགས་ནས་བྱད་མ་དགྲ་བགེགས་གནོད་བྱེད་བཀུག་ཚར་བཅད་བདེ་བའི་རོས་གང་སྟེ། བདག་མེད་ངང་ལ་བཞག འགྲོ་རྣམས་བསྒྲལ་འདོད་བསམ་པ་ཡིས། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ལ། །བྱང་ཆུབ་སྙིང་པོར་མཆིས་ཀྱི་བར། །རྟག་པར་བདག་ནི་སྐྱབས་སུ་མཆི། །ཤེས་རབ་སྙིང་རྗེ་དང་བཅས་པས། །བརྩོན་པས་སེམས་ཅན་དོན་དུ་བདག །སངས་རྒྱས་མདུན་དུ་གནས་བཅས་ཏེ། །རྫོགས་པའི་སངས་རྒྱས་ཐོབ་པར་བགྱིའོ། ། ལན་གསུམ་བརྗོད་ནས། སྭ་བྷཱ་ཝ་ དང༌། ཤཱུ་ནྱ་ཏཱ། བརྗོད་པས། སྟོང་པའི་ངང་ལས་རང་ཉིད་རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མ་སྐུ་མདོག་དམར་མོ་ཞལ་གཉིས་ཕྱག་གཉིས་མ། སྐུ་ལ་རུས་པའི་རྒྱན་དྲུག་གིས་བརྒྱན་པ། ཐོད་པ་སྐམ་པོ་ལྔས་དབུ་བརྒྱན་ཞིང་རློན་པ་ལྔ་བཅུའི་དོ་ཤལ་ཅན། དབུ་སྐྲ་སིལ་བུའི་ཚོགས་གྲོལ་པ། རྣ་ལྟག་གཡས་སུ་ཕག་ཞལ་ནག་པོ། ཕྱག་གཡས་ན་ཟངས་ཀྱི་གྲི་གུག་བསྣམས་པ། གཡོན་པ་ཐོད་པ་ཁྲག་གིས་གང་བ་བསྣམས་པ། ཞབས་བརྐྱང་བསྐུམ་གྱི་ཚུལ་དུ་བཞུགས་པ། སྣ་ཚོགས་པདྨ་དང་ཉི་མ་རོ་ཡི་གདན་ལ་སྐུ་གར་སྟབས་ཀྱིས་བཞུགས་པར་བསྐྱེད། དེའི་མདུན་དུ་སའི་དཀྱིལ་འཁོར་སེར་པོ་ཟླུམ་པོ། དེའི་དབུས་སུ་འཁོར་ལོ་དཀར་པོ་རྩིབས་བཞི་པ་ལྟེ་བ་མུ་ཁྱུད་དང་བཅས་པ་ཡོངས་སུ་གྱུར་པ་ལས། ཤར་སྔོ། ལྷོ་སེར། ནུབ་དམར། བྱང་ལྗང༌། དབུས་དཀར་བ། དེའི་སྟེང་དུ་སྣ་ཚོགས་པདྨ་ཉི་མ་རོ་ཡི་གདན་ལ།

ཡི་གེ་ཧྲཱིཿདཀར་པོ་ཡོངས་སུ་གྱུར་པ་ལས། ཨ་ཕྱི་སྣ་ནམ་བཟའ་དབང་ཕྱུག་ཆོས་ཀྱི་སྒྲོན་མ། སྐུ་མདོག་དཀར་ལ་དམར་བའི་མདངས་ཆགས་པ། ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཕྱག་གཡས་ན་དངུལ་དཀར་གྱི་མེ་ལོང་འདས་མ་འོངས་ད་ལྟ་དུས་གསུམ་གསལ་བར་འཆར་ཞིང༌། གཡོན་པས་ཐོད་པ་དངོས་གྲུབ་ཐམས་ཅད་འབར་བའི་ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེས་གང་བ་བསྣམས་པ། དར་སྣ་ཚོགས་ཀྱི་ན་བཟའ་སྐུ་ལ་གསོལ་བ། ཞི་བའི་མདངས་དང་ལྡན་པ། རྒྱས་པའི་བྱིན་ཆགས་པ། དབང་སྡུད་དྲག་པོ་ལས་ཀྱི་རྩལ་མངའ་བ། དྲག་པོ་དགྲ་བགེགས་བསྒྲལ་བའི་ལས་ལ་མྱུར་བ། མཐུ་སྟོབས་ཕྲིན་ལས་ལ་ཐོགས་པ་མེད་པ། བདུད་བཞིའི་གདན་གྱི་སྟེང་ན་བཞུགས་པ། མདུན་གྱི་རྩིབས་ཀྱི་གདན་ལ་ཤ་ཟ་ལས་ཀྱི་མཁའ་འགྲོ་སྐུ་མདོག་མཐིང་ནག་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཕྱག་གཡས་རལ་གྲི་ནམ་མཁའ་ལ་འཕྱར་བས་དགྲ་བགེགས་ཀྱི་སྲོག་རྩ་གཅོད་ཅིང༌། གཡོན་པས་དམ་ཉམས་ཀྱི་སྙིང་ཁྲག་གཙོ་མོའི་ཞལ་དུ་སྟོབ་པ། གླང་ཆེན་གྱི་པགས་པ་སྐུ་ལ་གསོལ་ཞིང༌། དུར་ཁྲོད་ཀྱི་ཆས་ཀྱིས་བརྒྱན་པ། གཡས་ཀྱི་རྩིབས་ཀྱི་སྟེང་དུ། ཡེ་ཤེས་ཀྱི་མཁའ་འགྲོ་མ་གཙུག་ན་ནོར་བུ། སྐུ་མདོག་དཀར་སེར་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཕྱག་གཡས་པ་གསེར་གྱི་ཌཱ་མ་རུ་དྲག་ཏུ་འཁྲོལ་ཞིང༌། གཡོན་པས་ཐོད་པ་རིན་པོ་ཆེས་གང་བ་ཐུགས་ཀར་འཛིན་པ། རིན་པོ་ཆེའི་རྒྱན་སྣ་ཚོགས་ཀྱིས་སྐུ་ལ་བརྒྱན་པ། དར་ལྗང་ཁུའི་ན་བཟའ་ལྷུབ་ལྷུབ་ཏུ་གསོལ་བ། གཡོན་གྱི་རྩིབས་ཀྱི་སྟེང་དུ་དམ་ཚིག་མཁའ་འགྲོ་ཡིད་བཞིན་ནོར་བུ་སྐུ་མདོག་དམར་སྐྱ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། ཕྱག་གཡས་པ་དབང་གི་རྐྱལ་བ་བསྣམས་ཤིང༌། གཡོན་ན་ཐོད་པ་ཡིད་བཞིན་གྱི་ནོར་བུས་གང་བ་འཛིན་པ། དར་དཀར་གྱི་ན་བཟའ་སྐུ་ལ་གསོལ་ཞིང༌། རིན་པོ་ཆེའི་རྒྱན་སྣ་ཚོགས་ཀྱིས་སྤྲས་པ། རྒྱབ་ཀྱི་རྩིབས་ཀྱི་སྟེང་དུ་དབང་གི་མཁའ་འགྲོ་སྐུ་མདོག་དམར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། གཡས་པས་ཟངས་ཀྱི་གྲི་གུག་ནམ་མཁའ་ལ་འཕྱར་བས་དགྲ་བགེགས་སྲོག་རྩ་གཅོད་ཅིང༌།

གཡོན་པས་ཟས་ནོར་ཕྱྭ་གཡང་ཐམས་ཅད་འགུགས་པར་བྱེད་པའི་ལྕགས་ཀྱུ་འཛིན་པ། དར་སྣ་ཚོགས་ཀྱི་ན་བཟའ་གསོལ་ཞིང་རིན་པོ་ཆེའི་རྒྱན་གྱིས་བརྒྱན་པ། མཁའ་འགྲོ་མ་བཞི་པོ་ཐམས་ཅད་ཀྱང་གཙོ་མོ་ལ་འདུད་པའི་ཚུལ་གྱིས་བཞུགས་ཤིང༌། བཀའ་ཉན་ཞིང་གསུང་སྒྲུབ་པའོ། །གཙོ་འཁོར་ཐམས་ཅད་ཀྱི་སྤྱི་བོར་ཨོཾ་དཀར་པོ། མགྲིན་པར་ཨཱཿདམར་པོ། ཐུགས་ཀར་ཧཱུྂ་སྔོན་པོ། ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་འཕྲོས་པས་ཨུ་རྒྱན་གྱི་གནས་དང༌། བཀའ་དང་དམ་ཚིག་གི་གཞལ་ཡས་ནས་རྗེ་བཙུན་མ་དང་ཨ་ཕྱི་སྣ་ནམ་བཟའ་མཆེད་ལྔ་ལ་མཁའ་འགྲོ་མ་བྱེ་བའི་ཚོགས་ཀྱིས་བསྐོར་བ། ཌཱ་མ་རུ་དང་དྲིལ་བུའི་སྒྲ་འུར་འུར་སིལ་སིལ་སྐད་ཅིག་གིས་མདུན་གྱི་ནམ་མཁར་སྤྱན་དྲངས། བཛྲ་ས་མཱ་ཛཿ ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་པཱ་དྱཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་པུཥྤེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་དྷཱུ་པེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་ཨཱ་ལོ་ཀེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་གནྡྷེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་ནཻ་ཝི་དྱེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཛྙཱ་ན་ས་མ་ཡ་ཤཔད་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་པཱ་དྱཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་པུཥྤེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་དྷཱུ་པེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། །ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཨཱ་ལོ་ཀེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་གནྡྷེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ནཻ་ཝི་དྱ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ།

ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཤཔད་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། 

དེ་ནས་བསྟོད་པ་ནི། ཧཱུྂ་༌། སྟོང་ཉིད་སྙིང་རྗེའི་སྙིང་པོ་ཅན། །ཁམས་གསུམ་པ་ཡི་ངོ་བོ་ཉིད། །བསྐལ་པ་མེ་ལྟར་འབར་བའི་འོད། །རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་བསྟོད། །དཔལ་ལྡན་རྡོ་རྗེ་མཁའ་འགྲོ་མ། །མཁའ་འགྲོ་མ་ཡི་འཁོར་ལོས་བསྒྱུར། །ཡེ་ཤེས་ལྔ་དང་སྐུ་གསུང་ཐུགས། །འགྲོ་བ་སྐྱོབ་ལ་ཕྱག་འཚལ་བསྟོད། །བདེ་བའི་རང་བཞིན་མ་སྐྱེས་པ། །ཇི་བཞིན་ཉིད་དུ་རྟོགས་གྱུར་པའི། །གཞན་དོན་མཐར་ཕྱིན་མཛད་པ་ཡི། །རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་བསྟོད། །གཡོན་པ་ཆགས་སོགས་ཁྲོ་མོའི་ཞལ། །གཡས་པ་རྟོག་མེད་ཕག་གདོང་ཞལ། །བདག་མེད་སྐྱེ་མེད་ཆོས་ཀྱི་དབྱིངས། །ཕག་མོ་ཡུམ་ལ་ཕྱག་འཚལ་བསྟོད། །ཇི་སྙེད་རྡོ་རྗེ་མཁའ་འགྲོ་མ། །ཀུན་རྟོག་འཆིང་བ་གཅོད་མཛད་ཅིང༌། །འཇིག་རྟེན་བྱ་བར་རབ་འཇུག་པའི། །དེ་སྙེད་རྣམས་ལ་ཕྱག་འཚལ་བསྟོད། །བདེ་གཤེགས་སྐུ་ལ་གཉིས་མེད་འཁྲིལ། །ཟག་མེད་བདེ་བའི་སྐུ་ཉམས་ལྡན། །སེམས་ཅན་རྣམས་ལ་བདེ་སྟེར་མ། །ཕག་མོ་ཡུམ་ལ་ཕྱག་འཚལ་བསྟོད། །ལྟེ་བའི་དབུས་ན་འབྱུང་ཞིང་འཕྲོ། །ཆོས་དང་ལོངས་སྤྱོད་རྫོགས་པར་ལྡན། །མཁའ་འགྲོ་མ་རྣམས་བསྡུད་མཛད་མ། །རྗེ་བཙུན་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཕྱི་ནང་གསང་གསུམ་མཆོད་པའི་ཚོགས། །རྣལ་འབྱོར་མ་ལ་མཆོད་པར་འབུལ། །ཐུགས་དམ་བསྐང་ཕྱིར་བཞེས་སུ་གསོལ། །དངོས་གྲུབ་གནང་ཕྱིར་བཞེས་སུ་གསོལ། །བར་ཆད་སེལ་ཕྱིར་བཞེས་སུ་གསོལ། །འདིར་གཤེགས་བཤམས་པའི་མཆོད་གཏོར་བཞེས། །མཆོག་དང་ཐུན་མོང་དངོས་གྲུབ་སྩོལ། །ཧཱུྂ་༌། ནམ་མཁའ་མཐིང་བཟང་འབར་བའི་ཀློང་དཀྱིལ་ནས། །མུ་མེན་མཐིང་གི་གཡུ་བྲག་ཏི་སྒྲོ་ནས། །ཆེ་ཞེས་ཆེན་མོ་ཨ་ཕྱི་སྣ་ནམ་བཟའ། །དགྲ་བགེགས་འདུལ་ཕྱིར་གནས་འདིར་གཤེགས་སུ་གསོལ། །དབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་ཆོས་ཀྱི་སྒྲོན། །སྐུ་མདོག་དཀར་དམར་ཞལ་གཅིག་ཕྱག་གཉིས་མ། །བདུད་བཞིའི་གདན་ལ་ཞབས་གཉིས་མཉམ་པར་བཞེངས། །དར་དཀར་ན་བཟའ་བདུད་རིགས་ཟིལ་གྱིས་གནོན། །ཞི་བའི་འཛུམ་ལྡན་རིན་ཆེན་རྒྱན་གྱིས་སྤྲས། །སྲིད་པ་གསུམ་གསལ་དངུལ་དཀར་མེ་ལོང་གཡས། །དགོས་འདོད་བསྒྲུབ་པའི་ནོར་བུ་གཡོན་ན་འཛིན། །མཐུ་སྟོབས་ཐོགས་མེད་ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོན། །ཨུ་རྒྱན་གནས་དང་དམ་ཚིག་གཞལ་ཡས་ནས། །རིན་ཆེན་མཛེས་པའི་གནས་མཆོག་དམ་པ་འདིར། །སྤྱན་འདྲེན་མཆོད་ཕྱིར་གནས་འདིར་གཤེགས་སུ་གསོལ། །ཨོཾ་དྷརྨ་ཏཱ་རེ་པདྨ་སཱུརྻ་མཎྜལ་ཧཱུྂ།

ཧཱུྂ་༌། དཔལ་ལྡན་ལྷ་མོ་གཙུག་ན་ནོར་བུ་ནི། །ཞལ་གཅིག་ཕྱག་གཉིས་སྐུ་མདོག་དཀར་སེར་ཅན། །དར་གྱི་ན་བཟའ་ཨིནྡྲ་ནཱི་ལའི་མདོག །སྐུ་ལ་གསོལ་ཞིང་རྒྱས་པའི་ངང་ཚུལ་ཅན། །ཞབས་གཉིས་མཉམ་བཞེངས་རིན་པོ་ཆེ་ཡིས་བརྒྱན། །ཕྱག་གཡས་གསེར་གྱི་ཌཱ་རུ་དྲག་ཏུ་འཁྲོལ། །གཡོན་པས་ཐུགས་ཀར་ཐོད་པ་རིན་ཆེན་བསྣམས། །པདྨའི་གདན་བཞེངས་གཙོ་ལ་འདུན་པའི་ཚུལ། །ཨུ་རྒྱན་ཡུལ་ནས་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །ཨོཾ་ཛྙཱ་ན་ཏཱ་རེ་པདྨ་མཎྜ་ལ་ཧཱུྂ། ཧཱུྂ་༌། དཔལ་ལྡན་ལྷ་མོ་པདྨ་རཱ་གའི་མདོག །ཞལ་གཅིག་ཕྱག་གཉིས་སྤྱན་གསུམ་གཅེར་བུ་གྲོལ། །ཐོད་པས་བརྒྱན་ནས་པདྨའི་གདན་ལ་བཞེངས། །ཕྱག་གཡས་གྲི་གུག་གཡོན་ན་ལྕགས་ཀྱུ་བསྣམས། །གཙོ་ལ་འདུན་ཕྱིར་དབང་གི་ངང་ཚུལ་ཅན། །དམ་ཚིག་གནས་ནས་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །ཨོཾ་མ་ཏྲ་དྷ་དྷེ་མ་ཙཀྲ་པདྨ་མཎྜ་ལ་ཧཱུྂ། ཧཱུྂ་༌། དཔལ་ལྡན་ལྷ་མོ་ཡིད་བཞིན་ནོར་བུ་ནི། །ཞལ་གཅིག་ཕྱག་གཉིས་སྐུ་མདོག་དམར་སྐྱའི་མདོག །ལེ་བརྒན་དར་དཀར་ན་བཟའ་སྐུ་ལ་གསོལ། །ཕྱག་གཡས་ནེའུ་རྐྱལ་གཡོན་ན་ནོར་བུ་བསྣམས། །སྣ་ཚོགས་ཀྱིས་བརྒྱན་པད་མའི་གདན་ལ་བཞུགས། །གཙོ་ལ་འདུན་ཞིང་སེམས་ཅན་དབུལ་བ་སེལ། །དམ་ཚིག་གནས་ནས་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །ཨོཾ་ས་མ་ཡ་ཏཱ་རེ་ཨ་ཙཀྲ་པདྨ་མཎྜལ་ཧཱུྂ། ཧཱུྂ་༌། ཤ་ཟ་ལས་ཀྱི་མཁའ་འགྲོ་ཆེན་མོ་ནི། །སྐུ་མདོག་སྔོ་སྐྱ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། །ཕྱག་གཡས་རལ་གྲི་གཡོན་ན་ཐོད་པ་བསྣམས། །ཞིང་ལྤགས་གསོལ་ཞིང་དུར་ཁྲོད་ཆས་ཀྱིས་བརྒྱན། །ཞབས་གཉིས་པུས་བཙུགས་གཙོ་མོའི་བཀའ་ཉན་བྱེད། །དགྲ་བགེགས་ཚར་གཅོད་ལྷ་མོ་སྤྱན་འདྲེན་གྱི། །བར་ཆད་སེལ་ཕྱིར་གནས་འདིར་གཤེགས་སུ་གསོལ། ཨོཾ་བྷ་ཤ་ཌཱ་ཀི་ནཱི་ཏི་ར་མཎྜལ་ཧཱུྂ། ཞེས་བརྗོད་ལ་སྤྱན་དྲངས། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་པཱ་དྱཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ།

ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་པུཥྤེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་དྷཱུ་པེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། །ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཨཱ་ལོ་ཀེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་གནྡྷེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ནཻ་ཝི་དྱེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ་ཤཔད་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། རང་གི་མདུན་དུ་བསྐྱེད་པའི་ཨ་ཕྱི་སྣ་ནམ་བཟའ་དབང་ཕྱུག་ཆོས་ཀྱི་སྒྲོན་མའི་སྐུ་དཔག་ཏུ་མེད་པ་བྱོན་ནས། སླ་ངའི་ནང་དུ་མར་ཁུ་ཞུ་བ་བཞིན་དུ། སིབ་སིབ་གཉིས་སུ་མེད་པར་བསམ། འཁོར་རྣམས་ལ་ཡང་རང་རང་གཉིས་སུ་མེད་པར་བསྟིམ་ལ། ཛཿཧཱུྂ་བཾ་ཧོཿ དགུག་གཞུག་བརྟན་པར་བྱས་ལ། ཡི་གེ་གསུམ་གྱིས་བྱིན་གྱིས་བརླབས་པར་བསམ། སླར་ཡང་ཐུགས་ཀའི་ས་བོན་ལས་འོད་ཟེར་འཕྲོས་པས། རང་གི་ལྷ་རྣམས་སྤྱན་དྲངས། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨརྒྷཾ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པཱ་དྱཾ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པུཥྤེ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་དྷཱུ་པེ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨཱ་ལོ་ཀེ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་གནྡྷེ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ནཻ་ཝི་དྱེ་ཨཱཿཧཱུྂ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཤཔད་ཨཱཿཧཱུྂ། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བདག་ལ་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། དེ་རྣམས་ཀྱིས། ཇི་ལྟར་བལྟམས་པ་ཙམ་གྱིས་ནི། །ལྷ་རྣམས་ཀྱིས་ནི་ཁྲུས་གསོལ་ལྟར། །ལྷ་ཡི་ཆུ་ནི་དག་པ་ཡི། །དེ་བཞིན་བདག་གིས་དབང་བསྐུར་རོ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་བྷི་ཥེ་ཀ་ཏ་ས་མ་ཡ་ཤྲཱི་ཡེ་ཧཱུྂ། །ཞེས་དབང་བསྐུར་དྲི་མ་དག་ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས། བདག་བསྐྱེད་ཀྱི་གཙོ་མོ་ལ་མི་བསྐྱོད་པས་དབུ་བརྒྱན་ཞིང༌། མདུན་བསྐྱེད་ཀྱི་གཙོ་མོ་ལ་རྡོ་རྗེ་རྣལ་འབྱོར་མ་དང༌། མཁའ་འགྲོ་མ་བཞི་ལ་རིན་ཆེན་འབྱུང་ལྡན་གྱིས་དབུ་བརྒྱན་ཞིང༌།

དབང་གི་ལྷ་རྣམས་ཀྱང་དེ་རྣམས་ལ་ཐིམ་པར་གྱུར་པའི་རང་གི་ཐུགས་ཀར་ཡི་གེ་བཾ་དམར་པོའི་མཐའ་སྐོར་དུ། ཨོཾ་ཨོཾ་ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ་བཛྲ་ཝརྞ་ནཱི་ཡེ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཕཊ྄་ཕཊ྄་ཕཊ྄་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧ་རི་ནི་ས་ཧཱུྂ་ཕཊ྄། མདུན་བསྐྱེད་ཀྱི་གཙོ་མོའི་ཐུགས་ཀར། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཏཱ་རེ་མ་མ་འདུས། སརྦ་འདུས། ཡར་འདུས། མ་མ་ཛྙཱ་ན་ཏཱ་ར་འདུས། གྷ་མུ་ས་མ་ཡ་ཏཱ་ར་དྷེ་མ་ཌཱ་ཀི་ནཱི། མ་མ་ཨཱ་ཡུཿཔུཎྱཻ་པུཥྚིཾ་ཀུ་རུ་ཡེ་འདུས། ཧཱུྃ་ཧཱུྂ་ཕཊ྄་ཕཊ྄། སྭཱ་ཧཱ། མགལ་མེའི་འཁོར་ལོ་ལྟ་བུ་དང༌། མཁའ་འགྲོ་མ་བཞིས་ཀྱང་བཟླས་པ་བྱེད་པར་བསྒོམས་ལ། སྔགས་གོང་འོག་གཉིས་མ་སྐྱོ་བའི་བར་ལ་བཟླས། དེ་ནས་ནང་མཆོད་བྱིན་གྱིས་བརླབ་པ་ནི། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལཱི་ཧ་ན་ཧ་ན་ཧཱུྂ་ཕཊ྄། ཨོཾ་སྭ་བྷཱ་ཝ་ཤུདྡྷཿསརྦ་དྷརྨཱཿསྭ་བྷཱ་ཝ་ཤུདྡྷོ྅ཧཾ། སྟོང་པ་ཉིད་དུ་གྱུར། སྟོང་པའི་ངང་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ། ཨ་ལས་ཐོད་པ་ཕྱི་དཀར་ལ་ནང་དམར་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ། ཤར་དུ་གོ་ལས་བ་གླང་གི་ཤ ལྷོར་ཀུ་ལས་ཁྱིའི་ཤ ནུབ་ཏུ་ད་ལས་རྟའི་ཤ བྱང་དུ་ཧ་ལས་གླང་པོ་ཆེའི་ཤ དབུས་སུ་ན་ལས་མིའི་ཤ དེ་རྣམས་ཀྱི་སྟེང་དུ། ཤར་དུ་བྷྲཱུྂ་ལས་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས། ལྷོར་ཨཾ་ལས་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས། ནུབ་ཏུ་ཛྲིཾ་ལས་སོ་སོར་རྟོགས་པའི་ཡེ་ཤེས། བྱང་དུ་ཁཾ་ལས་བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས། དབུས་སུ་ཧཱུྂ་ལས་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས། དེ་རྣམས་ཀྱི་སྟེང་དུ་ཨོཾ་དཀར་པོ་མགོ་ཐུར་དུ་བསྟན་པ། དེ་ལས་འོད་ཟེར་འཕྲོས། རླུང་གཡོས། མེ་སྦར། ཐོད་པའི་ནང་གི་རྫས་རྣམས་ཞུ་ཞིང་ཁོལ་བའི་རྣམ་པ་ཅན། དཀར་ལ་དམར་བའི་མདངས་ཆགས་པར་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ་ཧ་ཧོཿཧྲིཿ ལན་གསུམ་བརྗོད། གཡོན་པའི་མཐེབ་སྲིན་གྱིས་ཐབས་བདེ་བའི་བརྡར་ཟླུམ་སྐོར་གཅིག་བྱ། དེ་ནས་ཤེས་རབ་སྟོང་པའི་བརྡར་གྲུ་གསུམ་ཉིས་བརྩེགས་བྲི།

དེ་ནས་ལག་པ་གཉིས་པོའི་མཐིལ་དུ་ཉི་མ་དང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཨཱ་ལི་ཀཱ་ལིའི་ཡིག་འབྲུ་རྣམས་དམིགས་ཏེ། ཨཱ་ལི་ཀཱ་ལི་ཟླ་བ་དང་བཅས་པ་ཆང་ལ་ཐིམ་པས། དམ་ཚིག་པ་དང་དབྱེར་མེད་དུ་བྱིན་གྱིས་བརླབས་ནས། ལྷ་ཚོགས་ལ་སྨན་མཆོད་རྒྱས་པར་འབུལ་བ་ནི། ཨོཾ་ཨཱཿཧཱུྂ། མཆོད་དོ་ཕྱོགས་བཅུ་དུས་གསུམ་གྱི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས། ཡོན་ཏན་ཕྲིན་ལས་ཐམས་ཅད་ཀྱི་ངོ་བོ་གྱུར་པ། ཆོས་ཀྱི་ཕུང་པོ་སྟོང་ཕྲག་བརྒྱད་ཅུ་རྩ་བཞིའི་འབྱུང་གནས། འཕགས་པའི་དགེ་འདུན་ཐམས་ཅད་ཀྱི་མངའ་བདག རྒྱལ་བ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཞེས་འགྲེ། ཡི་དམ་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཡེ་ཤེས་ཀྱི་མཁའ་འགྲོ་ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོན་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཨ་མེས་ཚུལ་ཁྲིམས་རྒྱ་མཚོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། གཡུང་དྲུང་རྒྱ་མཚོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ནམ་མཁའ་དབང་ཕྱུག་གི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། དཔེ་ཀ་དབང་རྒྱལ་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། བསོད་ནམས་དཔལ་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཀ་ཐུང་གྲུ་བཞིའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། མཁན་པོ་དར་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། དཀོན་མཆོག་རིན་ཆེན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། བཙུན་པ་འབར་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཡབ་རྣལ་འབྱོར་པ་རྡོ་རྗེའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཆོས་རྗེ་འཇིག་རྟེན་མགོན་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། སངས་རྒྱས་དབོན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྒྱལ་བ་གཅུང་གི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྒྱལ་སྲས་ཐོག་ཁ་བའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། སྤྱན་སྔ་མཚམས་བཅད་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། སངས་རྒྱས་བཅུ་གཉིས་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཆོས་རྗེ་རྡོ་རྗེ་རྒྱལ་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། འཛམ་གླིང་ཆོས་ཀྱི་རྒྱལ་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། བཤེས་གཉེན་དོན་གྲུབ་རྒྱལ་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རིན་ཆེན་དཔལ་གྱི་རྒྱལ་མཚན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ།

ཆོས་རྒྱལ་རིན་ཆེན་དཔལ་བཟང་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། མཉམ་མེད་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྒྱལ་དབང་ཀུན་དགའ་རིན་ཆེན་ཆོས་ཀྱི་རྒྱལ་མཚན་དཔལ་བཟང་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། དངོས་གཞི་ལག་པའི་མཆོད་པ་ནི། རང་གི་ལག་པ་གཡོན་པའི་མཐིལ་དུ། གཡས་པའི་མཐེ་བོང་དང་སྲིན་ལག་སྤྲད་ལ། བདུད་རྩི་ལ་རེག་ནས་ཟླུམ་བསྐོར་དང་ཆོས་འབྱུང་ཉིས་བརྩེགས་བྲིས་ལ། ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ། དབུས་སུ་བཾ་ལས་ཕག་མོ་དམར་མོ། ཆོས་འབྱུང་གི་ཟུར་རྣམས་སུ། ཧཾ་ཡོཾ་ལས་གཤིན་རྗེ་མ་སྔོན་མོ། ཧྲཱིཾ་མོཾ་ལས་རྨོངས་བྱེད་མ་དཀར་མོ། ཧྲིཾ་ཧྲིཾ་ལས་སྐྱོད་བྱེད་མ་སེར་མོ། ཧཱུྂ་ཧཱུྂ་ལས་སྐྲག་བྱེད་མ་ལྗང་ཁུ། ཕཊ྄་ཕཊ྄་ལས་གཏུམ་པ་མོ་དུད་ཁ་རྣམས་རིམ་པ་བཞིན་བསྐྱེད། ཡང་མཐེ་བོང་ནས་བརྩམས་ཏེ། ཨོཾ་ཧ་ལས་རྡོ་རྗེ་སེམས་དཔའ། ན་མ་ཧི་ལས་རྣམ་པར་སྣང་མཛད། སྭ་ཧཱ་ཧུ་ལས་པདྨ་གར་དབང༌། བོ་ཥ་ཊ་ཧེ་ལས་ཧེ་རུ་ཀ་མཐིང་ཁ། ཧཱུྂ་ཧཱུྂ་ཧོ་ལས་རྡོ་རྗེ་ཉི་མ་སེར་པོ། ཕཊ྄་ཕཊ྄་ཧཾ་ལས་དོན་གྲུབ་ལྗང་ཁུ་རྣམས་རིམ་པ་བཞིན་གསལ་བར་བསྒོམས་ཏེ། བདུད་རྩི་གཡོན་ཐབས་རྫས་གཡས་སུ་བྱས་ནས་འབུལ། ཨོཾ་ཨོཾ་ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ་བཛྲ་ཝརྞ་ནཱི་ཡེ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཕཊ྄་ཕཊ྄་ཕཊ྄་སྭཱ་ཧཱ། ཨོཾ་བཾ། ཧཾ་ཡོཾ། ཧྲཱིཾ་མོཾ། ཧྲིཾ་ཧྲཱིཾ། ཧུྂ་ཧཱུྂ། ཕཊ྄་ཕཊ྄། ཨོཾ་ཧ། ན་མ་ཧི། སྭཱ་ཧཱ་ཧུ། བཽ་ཥ་ཊ་ཧེ། ཧུཾ་ཧཱུྂ་ཧོ། ཕཊ྄་ཕཊ྄་ཧཾ་སྭཱ་ཧཱ། དེ་ནས་ཕྱི་ནང་གསང་གསུམ་གྱི་མཆོད་པ་འབུལ་བ་ནི། ཡོན་ཆབ་མཐའ་ཡས་བླ་ན་མེད་པ་འདི། །རབ་འབྱམས་ཆོས་ཀྱི་དབྱིངས་རྣམས་བཀང་ནས་ནི། །དུས་གསུམ་བླ་མ་རྣམས་དང་དཀོན་མཆོག་མཆོད། །ཐུགས་རྗེས་འགྲོ་བའི་དོན་དུ་བཞེས་སུ་གསོལ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། རྒྱལ་བའི་སྐུ་ལ་ཉོན་མོངས་མི་མངའ་ཡང༌། །སེམས་ཅན་ལུས་ཀྱི་དྲི་མ་དག་བྱའི་ཕྱིར། །སྐུ་གསུང་ཐུགས་ལ་ཁྲུས་ཆབ་འདི་གསོལ་བས། །ཁམས་གསུམ་གནས་པའི་དྲི་མ་དག་གྱུར་ཅིག

ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་པཱ་དྱཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། དཀྱིལ་འཁོར་ཀུན་ཏུ་རྒྱལ་བ་མཉེས་བྱེད་པའི། །ཁ་དོག་དབྱིབས་ལེགས་རིན་ཆེན་མེ་ཏོག་འདི། །ནམ་མཁའ་གང་བར་སེམས་ཀྱིས་མངོན་སྤྲུལ་ནས། །གཞལ་ཡས་ཁང་དུ་སྤྲིན་བཞིན་གཏིབས་གྱུར་ཅིག །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་པུཥྤེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཚུལ་ཁྲིམས་དྲི་ལྡན་སྤོས་མཆོག་དམ་པ་འདི། །ཏིང་འཛིན་སྔགས་དང་ཕྱག་རྒྱས་བྱིན་བརླབས་པའི། །སངས་རྒྱས་ཞིང་དུ་སྤོས་དྲི་ངད་ལྡན་པས། །རྒྱལ་བ་རྒྱ་མཚོའི་ཚོགས་རྣམས་མཉེས་གྱུར་ཅིག །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་དྷཱུ་པེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཡེ་ཤེས་སྒྲོན་མེ་མ་རིག་མུན་སེལ་བའི། །ཤེས་རབ་གཟི་ལྡན་དཔལ་གྱི་སྒྲོན་མེ་འདི། །སྤྲིན་མེད་མཁའ་ལ་ཉི་ཤར་ཇི་བཞིན་དུ། །ཡེ་ཤེས་སྣང་བས་ཕྱོགས་ཀུན་གསལ་གྱུར་ཅིག །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་ཨཱ་ལོ་ཀེ་པྲ་ཏིའཙྪ་སྭཱ་ཧཱ། སྟོང་གསུམ་ཀུན་ཏུ་དྲི་མཆོག་འབྱུང་གྱུར་བའི། །སྤོས་དང་སྨན་སྦྱར་ཆུ་གཙང་རྒྱལ་བའི་ཆུ། །ཡེ་ཤེས་མཁའ་ལ་བདུད་རྩིའི་ཆར་འབབ་པ། །མི་གཙང་དྲི་མའི་ཚོགས་རྣམས་དག་གྱུར་ཅིག །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་གནྡྷེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། རྒྱལ་བའི་སྐུ་ལ་བཀྲེས་པ་མི་མངའ་ཡང༌། །འགྲོ་བའི་དོན་དུ་ལྷ་རྫས་བདུད་རྩིའི་བཤོས། །དག་ཅིང་གཙང་མར་སྦྱར་ཏེ་འབུལ་ལགས་ན། །ཡེ་ཤེས་ཐུགས་རྗེ་ཅན་གྱིས་བཞེས་སུ་གསོལ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་ནཻ་ཝི་དྱེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། དྲིལ་ཆེན་གསུང་སྙན་སྒྲ་སྒྲོག་ཚངས་པའི་དབྱངས། །རྔ་བོ་ཆེ་སོགས་དུང་དང་སྒྲ་སྙན་དང༌། །གསུང་དབྱངས་ངག་གིས་བསྟོད་དེ་གུས་འབུལ་ན། །སེམས་ཅན་རྣམས་ལ་ཆོས་སྒྲ་སྒྲོག་པར་ཤོག །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་ཝཱ་རཱ་ཧཱི་ཤཔད་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། །ཧཱུྂ་༌། ནུབ་ཕྱོགས་ཨུ་རྒྱན་གཞལ་ཡས་ནས། །རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མ། །སྐྱིལ་ཀྲུང་ཕྱེད་པ་རོ་ཡི་གདན། །གཅེར་བུ་སྐྲ་གྲོལ་སྐུ་མདོག་དམར། །གཡས་པ་རྟོག་མེད་ཕག་གདོང་ཞལ། །གཡོན་པ་ཁྲོ་ཞལ་སྤྱན་གསུམ་ལྡན། །ཉོན་མོངས་གཅོད་མཛད་གྲི་གུག་གཡས། །ཐོད་པ་དང་ནི་ཁ་ཊྭཱཾ་གཡོན། །མི་མགོ་རློན་པའི་དོ་ཤལ་ཅན། །མི་རུས་ཕྲེང་བའི་རྒྱན་དང་ལྡན། །ཞིང་ཆེན་རུས་རྒྱན་སྐུ་ལ་གསོལ།

ཁྱོད་ནི་སྟོང་ཉིད་སྙིང་རྗེའི་བདག །རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཨོཾ་བཾ་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །དམར་མོ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །རབ་མཛེས་བདེ་གཤེགས་དགྱེས་མཛད་མ། །རྡོ་རྗེ་ཕག་མོ་ཕྱག་འཚལ་བསྟོད། །ཧཾ་ཡོཾ་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །སྔོན་མོ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །སྒེག་མོའི་སྟོབས་ཀྱིས་གར་ཉམས་མཛད། །གཤིན་རྗེ་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཧྲཱིཾ་མོཾ་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །དཀར་མོ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །རབ་མཛེས་རྫུ་འཕྲུལ་སྟོབས་འཆང་མ། །རྨོངས་བྱེད་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཧྲིཾ་ཧྲཱིཾ་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །སེར་མོ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །སྡིག་པ་འཇོམས་མཛད་འོད་ཟེར་འབར། །སྐྱོད་བྱེད་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཧུྂ་ཧཱུྂ་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །ལྗང་ཁུ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །དཔའ་མོའི་མཚན་ཅན་དྲག་ཤུལ་ཅན། །སྐྲག་བྱེད་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ཕཊ྄་ཕཊ྄་ལས་སྐྱེས་པའི་མཁའ་འགྲོ་མ། །ནག་མོ་གྲི་གུག་ཐོད་ཁྲག་འཛིན། །རབ་འཇིགས་ཁྲོ་མོའི་ཆ་ལུགས་ཅན། །ཙཎྜི་ཀ་ལ་ཕྱག་འཚལ་བསྟོད། །ཨོཾ་ཧ་ས་བོན་དཀར་པོ་ལས། །ཞལ་གསུམ་ཕྱག་དྲུག་མཛེས་པའི་སྐུ། །སྤྱན་གསུམ་མཚོན་ཆ་སྣ་ཚོགས་བསྣམས། །རྡོ་རྗེ་སེམས་དཔའ་ཕྱག་འཚལ་བསྟོད། །ན་མ་ཧི་ནི་སེར་པོ་ལས། །དཔའ་བོ་རལ་པའི་ཐོར་གཙུག་ཅན། །ཡ་མཚན་དར་དང་རུས་པས་བརྒྱན། །རྣམ་པར་སྣང་མཛད་ཕྱག་འཚལ་བསྟོད། །སྭཱ་ཧཱ་ཧུ་ལས་བྱུང་བ་ཡི། །པདྨ་རཱ་གའི་མདོག་ལྟར་དམར། །སྤྱན་གསུམ་མཆེ་གཙིགས་གདུག་པ་འདུལ། །པདྨ་གར་དབང་ཕྱག་འཚལ་བསྟོད། །བཽ་ཥཊ་ཧེ་ནི་མཐིང་ཁ་ལས། །གཡས་བརྐྱང་ཉི་མའི་གདན་ལ་བཞུགས། །ཕྱག་རྒྱ་དྲུག་གིས་རབ་ཏུ་བརྒྱན། །ཧེ་རུ་ཀ་ལ་ཕྱག་འཚལ་བསྟོད། །ཧུཾ་ཧཱུྂ་ཧོ་ནི་དཔའ་བོའི་གཟུགས། །ཐབས་མཆོག་སྒེག་པའི་གར་ཉམས་ལྡན། །འཁོར་ལོ་ནོར་བུའི་དོ་ཤལ་མཛེས། །རིན་ཆེན་ཉི་མ་ཕྱག་འཚལ་བསྟོད། །ཕཊ྄་ཕཊ྄་ཧཾ་ལས་བྱུང་བ་ཡི། །ལྗང་ཁུ་ཕྲིན་ལས་གྲུབ་མཛད་ཅིང༌། །གྲི་ཐོད་རལ་གྲི་ཌ་མ་རུ། །རབ་བསྣམས་དཔའ་བོར་ཕྱག་འཚལ་བསྟོད།

༈ ཀྱཻ་བླ་མ་རྡོ་རྗེ་འཛིན་པ་ཆེན་པོ་རྣམས་དང༌། ཕྱོགས་བཅུ་དུས་གསུམ་ན་བཞུགས་པའི་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་དང༌། རྡོ་རྗེ་རྣལ་འབྱོར་མའི་ལྷ་ཚོགས་རྣམས་བདག་ལ་དགོངས་སུ་གསོལ། བདག་མིང་འདི་ཞེས་བགྱི་བ་ལ། དེ་བཞིན་ཉིད་ཀྱི་ཡེ་ཤེས་ནམ་མཁའ་ལྟ་བུ་དང༌། རྒྱ་མཚོ་ལྟ་བུ་དང༌། ཡིད་བཞིན་གྱི་ནོར་བུ་རིན་པོ་ཆེ་དབང་གི་རྒྱལ་པོ་ལྟ་བུས་གང་བར་མཛད་དུ་གསོལ། ཡོངས་སུ་གང་བར་མཛད་དུ་གསོལ། ཤིན་ཏུ་གང་བར་མཛད་དུ་གསོལ། རྒྱས་པར་མཛད་དུ་གསོལ། ཡོངས་སུ་རྒྱས་པར་མཛད་དུ་གསོལ། ཤིན་ཏུ་རྒྱས་པར་མཛད་དུ་གསོལ། སྨིན་པར་མཛད་དུ་གསོལ། ཡོངས་སུ་སྨིན་པར་མཛད་དུ་གསོལ། ཤིན་ཏུ་སྨིན་པར་མཛད་དུ་གསོལ། 

ཞེས་ལན་གསུམ་བརྗོད། དུས་གསུམ་གྱི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས་མཉམ་པ་ཉིད་ལྷུན་གྲུབ་རྒྱུན་ཆད་མེད་པའི་གཏེར་དུ་གྱུར་པས། མ་ནམ་མཁའ་དང་མཉམ་པའི་སེམས་ཅན་ཐམས་ཅད་ཀྱི་ལུས་ངག་ཡིད་གསུམ་རྣམ་པར་དག་ཅིང༌། དུས་གསུམ་གྱི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས་མཉམ་ཉིད་ལྷུན་གྲུབ་རྒྱུན་ཆད་མེད་པར་མཛད་དུ་གསོལ། ཞེས་ལན་གསུམ་བརྗོད། ལག་པའི་ལྷ་ཚོགས་རྣམས་རང་གི་སྤྱི་གཙུག་ཏུ་ཐིམ། ཨོཾ ཞེས་ཕྱག་རྒྱ་སྤྱི་བོར་བཞག་པས་ལུས་བྱིན་གྱིས་བརླབས། ཨཱཿ ཞེས་ཕྱག་རྒྱ་མགྲིན་པར་གུང་མོ་གཉིས་རྩེ་སྤྲད་པས་ངག་བྱིན་གྱིས་བརླབས། ཧཱུྂ ཞེས་རྡོ་རྗེ་བསྡམས་པའི་ཕྱག་རྒྱས་སྙིང་ཁར་སྦྱར་བས་ཡིད་བྱིན་གྱིས་བརླབས། སླར་ཡང་མས་ཡར་ལྡོག་ནས་ཕྱག་རྒྱ་དང་བཅས་ཏེ། ཧཱུྂ་ཨཱཿཨོཾ། ཞེས་སྲུང་བར་འགྱུར་རོ། འབྲས་རྒུན་བུ་རམ་བཅོས་པའི་ཆང༌། །ཤེས་རབ་མ་ད་ནང་གི་མཆོད་པ་འདི། །དབང་བཞི་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་དབང་བཞིའི་དོན་རྟོགས་ཤོག །བ་ལ་ཤ་ལྔ་འབྲས་འཁུར་དང༌། །ཐབས་གཏི་མུག་མཾ་སའི་མཆོད་པ་འདི། །རྟོག་མེད་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་ཆོས་སྐུ་ཐོབ་པར་ཤོག །ཐིག་ལེ་རྐང་དང་ཁུ་བ་ཞག །ཚོགས་དང་རཀྟའི་ལོངས་སྤྱོད་འདི། །བདེ་སྟོང་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་ལོངས་སྐུ་ཐོབ་པར་ཤོག །དབང་པོ་ལྔ་ཡི་མཆོད་པ་དང༌། །སྣ་ཚོགས་ཀེང་རུས་མཆོད་པ་འདི། །གསལ་སྟོང་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་སྤྲུལ་སྐུ་ཐོབ་པར་ཤོག །སྙིང་ཞོ་ཤ་ལ་སོགས་དམ་ཚིག་རྫས། །པཉྩ་ཨ་མྲྀ་ཏ་ཡི་མཆོད་པ་འདི། །དམ་ཚིག་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་བསླབ་གསུམ་དོན་རྟོགས་ཤོག །བུ་རམ་ལ་སོགས་མངར་གསུམ་དང༌། །དཀར་གསུམ་ཨརྒྷཾ་མཆོད་པ་འདི། །ཟག་མེད་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་སྐྱེ་མེད་རྟོགས་པར་ཤོག །དྲི་ཞིམ་སྨན་གྱི་མཆོད་པ་དང༌། །གཟུགས་སྡུག་སྒྲ་སྙན་འདོད་ཡོན་ཚོགས། །ཕྱི་ནང་གསང་བའི་མཆོད་པ་དང༌། །རྩ་བརྒྱད་སྟོང་ལ་སྦྱར་བ་འདི། །རིགས་ལྔའི་བདུད་རྩིར་བྱིན་གྱིས་རློབས། །ལོངས་སྤྱད་པས་འཁོར་འདས་མ་ལུས་པའི། །ཡོན་ཏན་རིན་ཆེན་ཐོབ་གྱུར་ཅིག

ཨོཾ་ཨཱཿཧཱུྂ། བརྗོད་པ་ཨོཾ་གྱི་འོད་ཟེར་གྱིས། །ཞེ་སྡང་ལས་གྱུར་ཚོགས་ཀྱི་རྫས། །དཀར་ལ་མདངས་ཆགས་བདུད་རྩིར་གྱུར། །སྐུ་བདེ་བ་ཆེན་པོར་གྱུར་བའི་རྫས། །ལོངས་སྤྱད་པས་རིན་ཆེན་ཆོས་སྐུ་ལ། །ཉམས་རྟོགས་དབྱེར་མེད་འཕེལ་གྱུར་ཅིག །གནོད་སེམས་ཞེ་སྡང་འཁྲུག་པ་ཀུན། །བྱམས་པ་ཆེན་པོར་འགྱུར་བར་ཤོག ཨོཾ་ཨཱཿཧཱུྂ། བརྗོད་པ་ཨཱཿཡི་འོད་ཟེར་གྱིས། །འདོད་ཆགས་ལས་གྱུར་ཚོགས་ཀྱི་རྫས། །དམར་ལ་མདངས་ཆགས་བདུད་རྩིར་གྱུར། །འགག་མེད་གསུང་སྒྲ་སྒྲོག་པའི་རྫས། །ལོངས་སྤྱད་པས་གསུང་དབྱངས་ཡན་ལག་དྲུག་ཅུ་ཡིས། །གདུལ་བྱ་མ་ལུས་ཚིམ་གྱུར་ཅིག །ཟག་བཅས་ཉོན་མོངས་འདོད་ཆགས་ཀུན། །ཟག་མེད་བདེ་བར་འགྱུར་བར་ཤོག ཨོཾ་ཨཱཿཧཱུྂ། བརྗོད་པ་ཧཱུྂ་གི་འོད་ཟེར་གྱིས། །གཏི་མུག་ལས་གྱུར་ཚོགས་ཀྱི་རྫས། །མཐིང་ལ་མདངས་ཆགས་བདུད་རྩིར་གྱུར། །ཐུགས་མཉམ་ཉིད་ཆོས་སྐུར་གྱུར་པའི་རྫས། །ལོངས་སྤྱད་པས་སེམས་རྒྱུད་ཡེ་ཤེས་ལྔ། །རྒྱུན་ཆད་མེད་པར་འཕེལ་གྱུར་ཅིག །རྨོངས་ཤིང་མ་རིག་གཏི་མུག་ཚོགས། །འོད་གསལ་ཆེན་པོར་འགྱུར་བར་ཤོག །མཁྱེན་པའི་ཡེ་ཤེས་ལྔ་ལྡན་བའི། །རྟོག་བྲལ་བདུད་རྩིའི་ཚོགས་འཁོར་འདི། །དཔའ་བོ་མཁའ་འགྲོ་མཉེས་པའི་རྫས། །ལྔ་རྫོགས་བདུད་རྩི་ཡིན་པའི་ཕྱིར། །བདེ་གཤེགས་རྡོ་རྗེ་འཆང་ལ་སོགས། །དུས་གསུམ་བདེ་གཤེགས་དཔའ་བོ་དང༌། །ཕྱི་ནང་མཁའ་འགྲོས་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། ཧཱུྂ་༌། རྣལ་འབྱོར་དབང་ཕྱུག་དཔའ་བོ་དང༌། །མཁའ་འགྲོའི་ཚོགས་དང་བཅས་རྣམས་ཀྱིས། །བདག་ཅག་འདིར་ཚོགས་རྣལ་འབྱོར་གྱི། །ཕུང་པོ་ཁམས་དང་སྐྱེ་མཆེད་དང༌། །ལུས་ངག་ཡིད་གསུམ་ཐམས་ཅད་ལ། །ཡེ་ཤེས་ཆེན་པོ་ལྔ་ལྡན་པའི། །བདེ་གཤེགས་རྒྱལ་བའི་དཀྱིལ་འཁོར་དུ། །བྱིན་གྱིས་རློབས་པར་མཛད་དུ་གསོལ། །ལུས་སྤྲུལ་སྐུ་སངས་རྒྱས་སུ་བྱིན་གྱིས་རློབས། །ངག་ལོངས་སྤྱོད་རྫོགས་སྐུར་བྱིན་གྱིས་རློབས། །སེམས་མཉམ་ཉིད་ཆོས་སྐུར་བྱིན་གྱིས་རློབས། །སྐུ་གསུམ་མ་ཐོབ་བར་དུ་ནི། །ཕྱི་ནང་བར་ཆད་མི་འབྱུང་ཞིང༌། །འདུ་བ་རྣམ་བཞིའི་ནད་རྣམས་སོལ། །མི་མཐུན་རྐྱེན་དང་བར་ཆད་སོལ། །མཆོག་དང་ཐུན་མོང་དངོས་གྲུབ་སྩོལ། །

དེ་ནས་ཚོགས་ཕུད་དང་བདུད་རྩི་གཏོར་མ་ལ་བྲན་ཏེ། ཧཱུྂ་༌། གཟུགས་སྒྲ་དྲི་རོ་རེག་ལ་སོགས། །དགྱེས་པ་མཆོག་གི་མཆོད་པ་འདི། །དཔལ་ལྡན་རྩ་བའི་བླ་མ་ལ། །བརྒྱུད་པའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །བཅོམ་ལྡན་འཁོར་ལོ་སྡོམ་པ་ལ། །དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་རྡོ་རྗེ་ཕག་མོ་ལ། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། བཅོམ་ལྡན་བུདྡྷ་ཧེ་རུ་ཀཿ དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་སངས་རྒྱས་མཁའ་འགྲོ་ལ། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། བཅོམ་ལྡན་བཛྲ་ཧེ་རུ་ཀཿ །དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་རྡོ་རྗེ་མཁའ་འགྲོ་ལ། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། བཅོམ་ལྡན་རཏྣ་ཧེ་རུ་ཀཿ །དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་རིན་ཆེན་མཁའ་འགྲོ་ལ། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། བཅོམ་ལྡན་པདྨ་ཧེ་རུ་ཀཿ །དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་པདྨ་མཁའ་འགྲོ་ལ། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌།

བཅོམ་ལྡན་ཀརྨ་ཧེ་རུ་ཀ །དཔའ་བོའི་ཚོགས་ཀྱིས་བསྐོར་བ་དང༌། །ཡུམ་ཆེན་ལས་ཀྱི་མཁའ་འགྲོ་དང༌། །དཔའ་མོའི་ཚོགས་ཀྱིས་བསྐོར་བ་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། དུར་ཁྲོད་ཆེན་པོ་བསིལ་བའི་ཚལ། །འཇིགས་སུ་རུང་བ་བསྐལ་བའི་མེ། །དམར་ནག་འཁྲུག་པའི་ཀློང་དཀྱིལ་ནས། །དཔལ་ལྡན་མ་ཧཱ་ཀཱ་ལ་དང༌། །ཡུམ་ཆེན་ཙཎྜི་དབང་ཕྱུག་མ། །འཁོར་ནག་པོ་སྟོང་དང་ཤ་ཟ་འབུམ། །མ་མོ་བྱེ་བའི་འཁོར་དང་བཅས། །མ་ལུས་འཁོར་ཚོགས་ཐམས་ཅད་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། ཕོ་བྲང་མཁའ་སྤྱོད་ཙཱ་རི་ནས། །བདུད་མགོན་སེང་གདོང་ཡབ་ཡུམ་ལ། །འཁོར་ནག་པོ་སྟོང་དང་ཤ་ཟ་འབུམ། །མ་མོ་བྱེ་བའི་ཚོགས་དང་བཅས། །མ་ལུས་འཁོར་ཚོགས་ཐམས་ཅད་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། །ཧཱུྂ་༌། འབྲི་གུང་དཔལ་གྱི་ནགས་ཁྲོད་ནས། །ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོན་མ་ལ། །མཁའ་འགྲོ་བྱེ་བ་འབུམ་གྱིས་བསྐོར། །མ་ལུས་འཁོར་ཚོགས་ཐམས་ཅད་ལ། །ཚོགས་དང་གཏོར་མ་རྒྱ་ཆེན་འདི། །གཉིས་མེད་ངང་དུ་འབུལ་ལགས་ན། །བླང་དོར་མེད་པར་བཞེས་སུ་གསོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་རོལ། །ཨ་ལ་ལ་ཏེ་དགྱེས་པར་མཛོད། ། ནང་མཆོད་བྱིན་གྱིས་བརླབས་ནས་གཏོར་མ་འབུལ་བ་ནི། ༈ ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལཱི་ཧ་ན་ཧ་ན་ཧཱུྂ་ཕཊ྄། ཨོཾ་སྭ་བྷཱ་ཝ་ཤུདྡྷཿསརྦ་དྷརྨཱཿསྭ་བྷཱ་ཝ་ཤུདྡྷོ྅ཧཾ། སྟོང་པ་ཉིད་དུ་གྱུར། སྟོང་པའི་ངང་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ། ཨ་ལས་ཐོད་པ་ཕྱི་དཀར་ལ་ནང་དམར་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ།

ཤར་དུ་གོ་ལས་བ་གླང་གི་ཤ ལྷོར་ཀུ་ལས་ཁྱིའི་ཤ ནུབ་ཏུ་ད་ལས་རྟའི་ཤ བྱང་དུ་ཧ་ལས་གླང་པོ་ཆེའི་ཤ དབུས་སུ་ན་ལས་མིའི་ཤ དེ་རྣམས་ཀྱི་སྟེང་དུ། ཤར་དུ་བྷྲཱུཾ་ལས་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས། ལྷོར་ཨཾ་ལས་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས། ནུབ་ཏུ་ཛྲིཾ་ལས་སོ་སོར་རྟོགས་པའི་ཡེ་ཤེས། བྱང་དུ་ཁཾ་ལས་བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས། དབུས་སུ་ཧཱུྂ་ལས་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས། དེ་རྣམས་ཀྱི་སྟེང་དུ་ཨོཾ་དཀར་པོ་མགོ་ཐུར་དུ་བསྟན་པ། དེ་ལས་འོད་ཟེར་འཕྲོས། རླུང་གཡོས། མེ་སྦར། ཐོད་པའི་ནང་གི་རྫས་རྣམས་ཞུ་ཞིང་ཁོལ་བའི་རྣམ་པ་ཅན། དཀར་ལ་དམར་བའི་མདངས་ཆགས་པར་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ་ཧ་ཧོཿཧྲིཿ ལན་གསུམ་བརྗོད། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཨི་དཾ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ། བཛྲ་ཧ་རི་ནི་ས་ས་མ་ཡ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། བླ་མ་དམ་པ་རྣམས་དང༌། ཡི་དམ་གྱི་ལྷ་ཚོགས་ ཞེས་རྗེ་བཙུན་མ་ལྷ་ལྔ་ལ་འབུལ། གཏོར་མ་གཉིས་པ་ལ་དམིགས་ཏེ། ཨོཾ་མ་ཧཱ་ཀཱ་ལ་ཧཱུྂ་ཕཊ྄། ཨི་དཾ་བ་ལིཾཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། དཔལ་ཡེ་ཤེས་ཀྱི་མགོན་པོ་ཆོས་སྐྱོང་བའི་སྲུང་མ་འཁོར་དང་བཅས་པ་རྣམས་ལ། དམ་རྫས་བརྒྱན་པའི་གཏོར་མ་འབུལ་ལོ། །མཆོད་དོ། །སངས་རྒྱས་ཀྱི་བསྟན་པ་སྲུངས་ཤིག བདག་ཅག་རྒྱུ་སྦྱོར་བ་འཁོར་དང་བཅས་པ་རྣམས་ཀྱི་ནད་དང་གནོད་པ་མི་མཐུན་པའི་ཕྱོགས་ཐམས་ཅད་ཞི་བར་མཛད་དུ་གསོལ། ཞེས་བརྗོད། གསུམ་པ་ལ་དམིགས་ལ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏེ་ཤྭ་རཱི་དྷརྨ་ཛྙཱ་ན་ས་མ་ཡ། ཏཱ་རེ་དྷེ་མ་ཌཱ་ཀི་ནཱི། སརྦ་མ་མ་བྷ་ཛཾ་ཨི་མཾ་བ་ལིཾ་ཨ་མྲྀ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སྣ་ནམ་གཟའ་དབང་ཕྱུག་ཆོས་ཀྱི་སྒྲོན་མ། ཡེ་ཤེས་ཀྱི་སྒྲོན་མ། དམ་ཚིག་གི་སྒྲོན་མ། དབང་ཕྱུག་མཁའ་འགྲོ་འཁོར་དང་བཅས་པ་རྣམས་ལ་དམ་རྫས་བདུད་རྩིའི་མཆོད་གཏོར་འབུལ་ལོ། །སངས་རྒྱས་ཀྱི་བསྟན་པ་སྲུངས། དཀོན་མཆོག་གི་དབུ་འཕང་བསྟོད། ཆོས་རྗེའི་སྤྱན་སྔར་ཞལ་གྱིས་བཞེས་ནས་དམ་བཅས་པ་བཞིན།

བཀའ་དང་བསྟན་པ་སྲུངས། འཁོར་དང་གང་ཟག་སྐྱོངས། རྣལ་འབྱོར་པ་བདག་ཅག་འཁོར་དང་བཅས་པའི་ནད་དང་གནོད་པ་རྐྱེན་ཕྱི་ནང་གི་བར་ཆད་ཞི་བར་མཛད་དུ་གསོལ། མཆོག་ཐུན་མོང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། ཞེས་བརྗོད་ནས། བཞི་པ་ལ་དམིགས་ལ། ཨོཾ་ཀརྞ་ཝརྟི། ཐུན་མེ་རི་ཨི་དཾ་བ་ལིཾ༴ ཨོཾ་དྷ་ཀ་པ་དྷ་རི་ཙཱ་མུཎྜི་ཏིསྟ་རྦད་ལོ་ཨི་དཾ༴ ཨོཾ་ཨ་ནན་ཧ་ར་གྱི་ནི་མེ་ཐུན་ཕྲེ་ཡེ་ཨི་དཾ༴ ཨོཾ་ནི་ཏི་མ་ཏི་སུ་རུ་མ་ཏི་ཨི་དཾ༴ ཨོཾ་དྷ་ཀི་ཏ་ཀྲ་མ་ནི་དགྲ་བོ་མཱ་ར་ཡ་ཛ་ཛཿཨི་དཾ༴ ཇོ་མོ་བཀྲ་ཤིས་ཚེ་རིང་མ་མཆེད་ལྔ་འཁོར་དང་བཅས་པ་རྣམས་ལ། དམ་རྫས་བདུད་རྩིའི་མཆོད་གཏོར་འབུལ་ལོ། །མཆོད་དོ། །དཔལ་ལྡན་བཞད་པའི་རྡོ་རྗེའི་སྤྱན་སྔར་ཞལ་གྱིས་བཞེས་ཤིང་དམ་བཅས་པ་བཞིན་དུ། སངས་རྒྱས་ཀྱི་བསྟན་པ་སྲུངས། དཀོན་མཆོག་གི་དབུ་འཕང་བསྟོད། ། ཅེས་བརྗོད། ཡང་གཏོར་མ་དེ་རང་བྱིན་གྱིས་བརླབ་ལ། ཨོཾ་དེ་ཝ་རུ་དྲ་མ་ར་ཡ་ཨི་དཾ། བྷི་ཤ་དུར་མི་ཏ་དྷེ་ཌཱ་ཀི་ནཱི་ཨི་དཾ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་ཁཱ། ཤ་ཁྲག་དམར་གྱི་གཏོར་མ་སྨན་རཀ་བདུད་རྩིས་བྲན་པ་འདི། སྲོག་གཅོད་ནག་མོ་འཁོར་དང་བཅས་པ་དང༌། ཕྲིན་ལས་ཤ་ཟ་མཁའ་འགྲོ་འཁོར་དང་བཅས་པ་རྣམས་ལ་འབུལ་ལོ། སངས་རྒྱས་ཀྱི་བསྟན་པ་སྲུངས། དཀོན་མཆོག་གི་དབུ་འཕང་བསྟོད། སེར་ཆགས་ཀྱི་མགོན་སྐྱབས་མཛོད། བདག་ཅག་རྣལ་འབྱོར་པ་རྒྱུ་སྦྱོར་དཔོན་སློབ་འཁོར་དང་བཅས་པ་རྣམས་ལ། བསམ་སྦྱོར་ངན་པ་ཤོམ་པའི་དགྲ་བགེགས་ཐམས་ཅད་ཚར་ཆོད། བཀའ་དང་བསྟན་པ་སྲུངས། འཁོར་དང་གང་ཟག་སྐྱོང༌། ཐུགས་དམ་རྒྱུད་བསྐུལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། ཞེས་བརྗོད་དོ། །ཡང་ལྔ་པ་ལ་དམིགས་ཏེ། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱཿཎཱཾ་ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ་ཨོཾ་ཨཱཿཧཱུྂ། ཨི་དཾ་བ་ལིཾ་ བརྗོད། རྒྱལ་པོ་ཆེན་པོ་སྡེ་བཞི་ལ་སོགས་པ་དཔལ་མགོན་བདུན་ཅུ་འཁོར་དང་བཅས་པ་ལ་འབུལ་ལོ། །སངས་རྒྱས་ཀྱི་བསྟན་པ་སྲུངས། དཀོན་མཆོག་གི་དབུ་འཕང་བསྟོད།

ཅི་བསམ་པའི་དོན་ཐམས་ཅད་ཆོས་བཞིན་དུ་སྒྲུབས། ཞེས་པ་དང༌། དེའི་རྗེས་སུ། ལྷ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཀླུ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། །གནོད་སྦྱིན་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྲིན་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །འབྱུང་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཡི་དྭགས་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཤ་ཟའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྨྱོ་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །བརྗེད་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །མཁའ་འགྲོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་མོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་ལུས་པ་ནི་ཐམས་ཅད་དག །འདིར་གཤེགས་བདག་ལ་དགོངས་སུ་གསོལ། །མཆོད་སྦྱིན་གཏོར་མ་འདི་སྣོམས་ལ། །རྣལ་འབྱོར་བདག་ཅག་འཁོར་བཅས་ལ། །ནད་མེད་ཚེ་དང་དབང་ཕྱུག་དང༌། །དཔལ་དང་གྲགས་དང་སྐལ་བ་བཟང༌། །ལོངས་སྤྱོད་རྒྱ་ཆེན་ཀུན་འགྲུབ་ཤོག །ཞི་དང་རྒྱས་ལ་སོགས་པ་ཡི། །ལས་ཀྱི་དངོས་གྲུབ་བདག་ལ་སྩོལ། །སྲུང་མས་རྟག་ཏུ་སྡོང་གྲོགས་མཛོད། །དུས་མིན་འཆི་དང་ནད་རྣམས་དང༌། །གདོན་དང་བགེགས་རྣམས་མེད་པར་མཛོད། །རྨི་ལམ་ངན་དང་མཚན་མ་ངན། །བྱང་སྟེམས་ངན་པ་མེད་པར་མཛོད། །འཇིག་རྟེན་བདེ་ཞིང་ལོ་ལེགས་དང༌། །འབྲུ་རྣམས་འཕེལ་ཞིང་ཆོས་འཕེལ་དང༌། །བདེ་ལེགས་ཕུན་སུམ་ཚོགས་པ་དང༌། །ཡིད་ལ་འདོད་པ་ཀུན་འགྲུབ་ཤོག ། ཅེས་རྒྱས་པར་བསྔོ། ༈ །མ་ཧྲཱིཿསྔོན་གྱི་སྨོན་ལམ་གྱི་དབང་གིས། ཆོས་ཀྱི་དབྱིངས་ནས་སྐུ་བཞེངས། ཡེ་ཤེས་ཀྱི་དབྱིངས་ནས་སྤྱན་གྱིས་གཟིགས། རྫུ་འཕྲུལ་གྱི་དབང་གིས་ཕྲིན་ལས་ཀྱི་དཀྱིལ་འཁོར་འདི་ཉིད་དུ་བྱོན་པར་མཛད་དུ་གསོལ། ཨ་ཕྱི་སྣ་ནམ་བཟའ། དབང་ཕྱུག་ཆོས་ཀྱི་སྒྲོན་མ། ཡེ་ཤེས་ཀྱི་སྒྲོན་མ། དམ་ཚིག་གི་སྒྲོན་མ། དབང་གི་ལྷ་མོ། ཤ་ཟ་ལས་ཀྱི་མཁའ་འགྲོ་མ། ཇོ་མོ་བཀྲ་ཤིས་ཚེ་རིང་མ། མི་གཡོ་བློ་བཟང་མ། གཏད་དཀར་འགྲོ་བཟང་མ། ཅོད་པན་མགྲིན་བཟང་མ། མཐིང་གི་ཞལ་བཟང་མ། ཕ་མེས་རྒྱུད་ཀྱི་སྲུང་མ། དབེན་གནས་གཉན་པོའི་སྲུང་མ།

བཀའ་བརྒྱུད་འདི་ཉིད་ཀྱི་བསྟན་པ་བསྲུང་བར་ཞལ་གྱིས་བཞེས་པའི་སྲུང་མ་དམ་ཚིག་ཅན་ཁྱེད་རྣམས་ལ། ཡེ་ཤེས་ལ་གསལ་འགྲིབ་མི་མངའ་བ། ཐུགས་རྗེ་ལ་ཕྱི་བཤོལ་མི་མངའ་བ། བྱིན་རླབས་ལ་རྒྱང་རིང་ཐུང་མི་མངའ་བ། མཐུ་སྟོབས་ལ་ཐོགས་པ་མི་མངའ་བ་ཁྱེད་རྣམས་ཀྱིས། རྣལ་འབྱོར་པ་བདག་ཅག་འཁོར་དང་བཅས་པ་རྣམས་ལ། སྐུ་ཡི་སྒོ་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། གསུང་གི་སྒོ་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། ཐུགས་ཀྱི་སྒོ་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། ཡོན་ཏན་གྱི་སྒོ་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། ཕྲིན་ལས་ཀྱི་སྒོ་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། རྣལ་འབྱོར་པ་བདག་གིས་གསོལ་ཞིང་མཆོད་པའི་རྟེན་དམ་པ་འདི་ལ་བཞུགས་ནས། བླ་ན་མེད་པའི་བྱང་ཆུབ་སྒྲུབ་པའི་ལམ་གྱི་བར་ཆད་སེལ་ཞིང༌། བདུད་བཞིའི་འཇིགས་པ་ལས་སྐྱེལ་མ་དམ་པ་མཛད་དུ་གསོལ། ལུས་ངག་ཡིད་གསུམ་ལོག་པའི་ལམ་དུ་མི་ལྟུང་ཞིང༌། ཐར་པ་བྱང་ཆུབ་ཀྱི་ཕྱོགས་སུ་བློ་འགྲོ་བར་མཛད་དུ་གསོལ། འཇིག་རྟེན་འདིར་ཡང་མི་མཐུན་པའི་ཕྱོགས་ཐམས་ཅད་ལས་རྒྱལ་ནས། བཀྲ་ཤིས་དང་བདེ་ལེགས་ཕུན་སུམ་ཚོགས་པས་ཁྱབ་ཅིང་རྒྱལ་བར་གྱུར་ཅིག ལྷ་འདྲེ་མི་གསུམ་དབང་དུ་འདུས་ཤིང༌། དུས་དང་གནས་སྐབས་ཐམས་ཅད་དུ་ཕྲིན་ལས་ཀྱི་བྱ་བ་ལ་གསོལ་བ་ཇི་ལྟར་བཏབ་པ་བཞིན་དུ་འགྲུབ་པར་གྱུར་ཅིག འགྲོ་བ་ཡོངས་ཀྱི་དཔལ་དང་མགོན་དུ་གྱུར་ནས། མཆོག་དང་ཐུན་མོང་གི་དངོས་གྲུབ་ཐོགས་པ་མེད་པ་བདེ་བླག་ཏུ་སྩོལ་བའི་ཕྲིན་ལས་མཛོད་ཅིག ལོག་པར་ལྟ་བའི་དགྲ་བགེགས་ལ་སོགས་པ་མི་མཐུན་པའི་ཕྱོགས་ཐམས་ཅད་ལས་རྒྱལ་ནས་བཀྲ་ཤིས་བདེ་ལེགས་ཕུན་སུམ་ཚོགས་པས་ཁྱབ་ཅིང་རྒྱལ་བར་གྱུར་ཅིག 

ལྷག་མ་ནི། ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱཿཎཱཾ་ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ་ཨོཾ་ཨཱཿཧཱུྂ། ལན་གསུམ་བྱིན་གྱིས་བརླབས་ནས། བྷྱོཿ ཤར་ནས་གཤེགས་པའི་རྣལ་འབྱོར་བདུན། །རྣལ་འབྱོར་བདུན་གྱི་གཙོ་མོ་ནི། །དཀར་མོ་དུང་གི་ཐོར་གཙུག་ཅན། །གཅིག་ནི་ཐབས་ཀྱི་སྐུ་རུ་གསལ། །གྲུབ་ནི་ཆོས་ཀྱི་དབྱིངས་སུ་གྲུབ། །དེ་ལྟར་གཡོག་འཁོར་བྱེ་བ་ཡང༌། །འདིར་བྱོན་ལྷག་མའི་གཏོར་མ་བཞེས། །ཁྱེད་ཀྱིས་ཞི་བའི་ལས་མཛོད་ཅིག བྷྱོཿ ལྷོ་ནས་གཤེགས་པའི་རྣལ་འབྱོར་བདུན། །རྣལ་འབྱོར་བདུན་གྱི་གཙོ་མོ་ནི། །སེར་མོ་གསེར་གྱི་ཐོར་གཙུག་ཅན། །གཅིག་ནི་ཐབས་ཀྱི་སྐུ་རུ་གསལ། །གྲུབ་ནི་ཆོས་ཀྱི་དབྱིངས་སུ་གྲུབ། །དེ་ལྟར་གཡོག་འཁོར་བྱེ་བ་ཡང༌། །འདིར་བྱོན་ལྷག་མའི་གཏོར་མ་བཞེས། །ཁྱེད་ཀྱིས་རྒྱས་པའི་ལས་མཛོད་ཅིག ། བྷྱོཿ ནུབ་ནས་གཤེགས་པའི་རྣལ་འབྱོར་བདུན། །རྣལ་འབྱོར་བདུན་གྱི་གཙོ་མོ་ནི། །དམར་མོ་བྱུ་རུའི་ཐོར་གཙུག་ཅན། །གཅིག་ནི་ཐབས་ཀྱི་སྐུ་རུ་གསལ། །གྲུབ་ནི་ཆོས་ཀྱི་དབྱིངས་སུ་གྲུབ། །དེ་ལྟར་གཡོག་འཁོར་བྱེ་བ་ཡང༌། །འདིར་བྱོན་ལྷག་མའི་གཏོར་མ་བཞེས། །ཁྱོད་ཀྱིས་དབང་གི་ལས་མཛོད་ཅིག བྷྱོཿ བྱང་ནས་གཤེགས་པའི་རྣལ་འབྱོར་བདུན། །རྣལ་འབྱོར་བདུན་གྱི་གཙོ་མོ་ནི། །ལྗང་མོ་གཡུ་ཡི་ཐོར་གཙུག་ཅན། །གཅིག་ནི་ཐབས་ཀྱི་སྐུ་རུ་གསལ། །གྲུབ་ནི་ཆོས་ཀྱི་དབྱིངས་སུ་གྲུབ། །དེ་ལྟར་གཡོག་འཁོར་བྱེ་བ་ཡང༌། །འདིར་བྱོན་ལྷག་མའི་གཏོར་མ་བཞེས། ཁྱོད་ཀྱིས་དྲག་པོའི་ལས་མཛོད་ཅིག ། ཅེས་ལྷག་མ་བཏང༌། སྐབས་འདིར་བསྐང་ཡིག་བཀླག དེ་ནས་བསྟོད་པ་བྱ་བ་ནི། ཧྲཱིཿདབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་ཆོས་ཀྱི་སྒྲོན། །ཉི་ཤར་གངས་ལྟར་དཀར་དམར་མདངས་དང་ལྡན། །གཟུགས་བཟང་གཟི་བརྗིད་འོད་ཕྲེང་དྲྭ་བ་འགྱེད། །ལྟ་སྡུག་ཡིད་འོང་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །ཞལ་གཅིག་ཕྱག་གཉིས་ཞབས་གཉིས་མཉམ་པར་བཞེངས། །ཞབས་འོག་བདུད་བགེགས་འབྱུང་པོའི་ཚོགས་རྣམས་མནན། །ཕྱག་གཡས་དངུལ་དཀར་མེ་ལོང་མཛེས་པ་བསྣམས། །སྲིད་གསུམ་གསལ་མཛད་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །ཕྱག་གཡོན་ཡིད་བཞིན་ནོར་བུ་རིན་པོ་ཆེ། །རྒྱལ་བ་མ་ལུས་མཉེས་པར་མཛད་པ་ཡི། །སེམས་ཅན་ཀུན་གྱི་དགོས་འདོད་རེ་སྐོང་མ། །ཚོགས་གཉིས་རྫོགས་མཛད་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །ལྷ་མིན་གནོད་སྦྱིན་མཁའ་འགྲོའི་ཚོགས་དང་བཅས། །བདུད་འཇོམས་ལྷ་མིན་གཡུལ་ངོ་ཟློག་པར་མཛད། །དཀར་པོ་ལྷ་ཡི་ཆབ་སྲིད་དབུགས་འབྱིན་མ། །ཕྲིན་ལས་བཞི་ལྡན་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །སྲིད་གསུམ་དབང་བསྡུད་འགྲོ་བ་མ་ལུས་སྐྱོང༌། །སངས་རྒྱས་སྟོང་གི་བསྟན་པ་སྲུང་བར་མཛད། །འགྲོ་བའི་བསམ་པ་ཆོས་བཞིན་སྒྲུབ་མཛད་མ། །ཡིད་བཞིན་བསམ་འགྲུབ་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །བྱང་ཆུབ་ཐུགས་ཀར་ཀུན་བཟང་ཆོས་ཀྱི་སྒྲོན། །ལྷ་མི་འཇིག་རྟེན་ཀུན་ནས་སྐྱོང་མཛད་མ། །གླིང་བཞི་ཀུན་ཏུ་རྒྱུ་བའི་དྲག་མོ་རྒྱལ། །དཔལ་ལྡན་ཆོས་ཀྱི་སྒྲོན་མར་ཕྱག་འཚལ་བསྟོད། །དེང་སང་བོད་ཡུལ་དབུས་ཀྱི་རྡོ་རྗེ་གདན། །དཔལ་གྱི་འབྲི་གུང་ཕོ་བྲང་བརྟན་བཞུགས་ནས། །ཆོས་མཛད་སྒྲུབ་པ་པོ་རྣམས་སྐྱོང་མཛད་མ། །བཀའ་སྲུང་ཡང་དག་ཨ་ཕྱིར་ཕྱག་འཚལ་བསྟོད། །བསླབ་གསུམ་རྒྱན་ལྡན་གྲུབ་པའི་རྣལ་འབྱོར་པས། །ཞིང་བསམ་དངོས་གསུམ་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང༌། །མ་ནོར་ཚུལ་བཞིན་བསྒྲུབས་ནས་མཆོད་གྱུར་ན། །ཅི་འདོད་དངོས་གྲུབ་སྟེར་ལ་ཕྱག་འཚལ་བསྟོད། །བདག་གིས་ཁྱེད་ལ་མཆོད་ཅིང་གསོལ་བ་ཡིས། །འགྲོ་དྲུག་ཆུ་བོར་བྱིང་བའི་མ་རྒན་རྣམས། །ཉོན་མོངས་དུག་གསུམ་དབང་དུ་མི་གཏོང་ཞིང༌། །ཚོགས་གཉིས་རྫོགས་པའི་ཕྲིན་ལས་མཛད་དུ་གསོལ། །ཁྱད་པར་ཆོས་རྗེའི་བཀའ་སྲུང་ཐུགས་དམ་ཅན། །ལས་སུ་སངས་རྒྱས་བསྟན་པ་གཉེན་པོ་སྲུངས། །ཟས་སུ་དམ་རྫས་བདུད་རྩིའི་མཆོད་གཏོར་བཞེས། །རྣལ་འབྱོར་བཅོལ་བའི་ཕྲིན་ལས་གྲུབ་པར་མཛོད། །ཧྲཱིཿ

བཞིན་བཟང་དབྱིངས་ཕྱུག་དཔལ་ལྡན་ཆོས་ཀྱི་སྒྲོན། །སྐུ་མདོག་དཀར་དམར་འགྱུར་མེད་འགྱིང་བག་ཅན། །འཁོར་བའི་ལོག་ལྟ་སྐེམ་བྱེད་འོད་མངའ་མ། །བདེ་ཆེན་དགའ་སྟེར་རབ་དགྱེས་དཔའ་མོ་སྟེ། །ཞལ་གཅིག་ཕྱག་གཉིས་ཞི་བྱེད་འཛུམ་པའི་སྐུ། །ཕྱག་གཡས་ཌ་རུ་སྙན་པ་ཀུན་ཁྱབ་སྒྲོག །གཡོན་པ་ཐོད་པས་དངོས་གྲུབ་ཀུན་སྩོལ་ཞིང༌། །ཟ་འོག་སྣ་ལྔའི་ན་བཟའ་སྐུ་ལ་གསོལ། །རིན་པོ་ཆེས་བརྒྱན་དར་གྱི་ཅོད་པན་ཅན། །ཆིབས་སུ་སྤྲིན་གསེབ་ཆུ་རྟ་གྲོ་སྔོན་བཅིབས། །དར་ཆེན་བློ་ནན་མུ་མེན་ཡོབ་ཆེན་འཁྲོལ། །གཡུ་མཐུར་སྔོན་པོ་གསེར་སྲབ་སེར་པོས་རྒྱན། །ཟ་འོག་དར་གདན་གསེར་སྒའི་སྒ་སྟེང་དུ། །དབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་བཞད་པའི་ཚུལ། །སྙིང་རྗེ་ཆེན་པོས་ཕོ་ཉ་རྫུ་འཕྲུལ་སྟོན། །ལྷ་མིན་གནོད་སྦྱིན་མཁའ་འགྲོའི་ཚོགས་ཀྱིས་བསྐོར། །ཕྲིན་ལས་བཞི་ལྡན་སྤྲོས་མེད་ངང་ཚུལ་ཅན། །ཕུན་སུམ་ཚོགས་གནས་དཔལ་གྱི་འབྲི་གུང་གི། །ལོངས་སྤྱོད་སྤེལ་ཕྱིར་བཀའ་སྲུང་ཀུན་མཛད་མ། །བསྟན་པ་སྲུང་ཕྱིར་གཏོར་མ་འདི་བཞེས་ལས། །ཚོགས་གཉིས་རྫོགས་པའི་ཕྲིན་ལས་མཛད་དུ་གསོལ། བྷྱོཿ རིན་ཆེན་ལས་གྲུབ་རབ་གསལ་གཞལ་ཡས་སུ། །བདུད་བགེགས་རུ་ཏྲ་བསྣོལ་བའི་གདན་སྟེང་དུ། །ཧྲཱིཿལས་བྱུང་བའི་དཔལ་ལྡན་ཆོས་ཀྱི་སྒྲོན། །འགྲོ་བའི་གཉེན་མཆོག་མ་ལ་ཕྱག་འཚལ་བསྟོད། །སྤྲོས་བྲལ་ཆོས་ཀྱི་དབྱིངས་ལས་མ་གཡོས་ཀྱང༌། །བརྩེ་བ་ཆེན་པོས་འགྲོ་ལ་རབ་དགོངས་ཏེ། །སྣ་ཚོགས་སྤྲུལ་པས་སྒྱུ་མའི་གཟུགས་བཟུང་ནས། །སྣང་སྲིད་དབང་བསྡུད་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །ཁྱོད་ཀྱི་རབ་དཀར་དང་བའི་མདངས་མཆོག་ནི། །ཟླ་བའི་གཟི་བརྗིད་རྣམ་པར་འཕྲོ་བ་བཞིན། །འཛུམ་ཞིང་མཛེས་པའི་འགྱིང་བག་དང་ལྡན་པའི། །ཞི་བའི་ལས་མཛད་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །དུས་གསུམ་ཐོགས་མེད་སྤྱན་དང་ལྡན་པའི་ཕྱིར། །གསལ་དྭངས་དངུལ་དཀར་མེ་ལོང་གཡས་ན་འཛིན། །ཡིད་བཞིན་རེ་སྐོང་ནོར་བུ་གཡོན་ན་འཛིན། །དགོས་འདོད་ཀུན་གྲུབ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །གློག་འགྱུ་ཙམ་གྱིས་སྟོང་གསུམ་ཁྱབ་པར་ཉུལ། །བསྟན་པའི་གཉེར་མཛད་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །གཡས་པས་སྟོང་ཉིད་ཌཱ་རུ་དྲག་ཏུ་འཁྲོལ། །གཡོན་པ་སྙིང་རྗེས་བདུད་ཁྲག་ཞལ་དུ་གསོལ། །མི་རྐང་གླིང་བུས་མ་ཚོགས་དབང་དུ་བསྡུད། །མ་རུང་སྒྲོལ་མཛད་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །སངས་རྒྱས་བསྟན་པ་གཉན་པོ་བཤིག་པ་དང༌། །བླ་མ་ལ་སྨོད་ཆོས་འཁོར་འཇོམས་པའི་དགྲ། །རང་བཞིན་མ་རུངས་གདུག་པའི་སེམས་ལྡན་པ། །མ་ལུས་མྱུར་དུ་སྒྲོལ་བའི་ཕྲིན་ལས་མཛོད། །ཧཱུྂ་༌། དབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་ཡེ་ཤེས་སྒྲོན། །སྐུ་མདོག་དཀར་སེར་རང་བྱུང་རྣམ་དག་སྐུ། །ཞལ་གཅིག་ཕྱག་གཉིས་རྒྱས་པའི་ལས་མཛད་མ། །ཕྱག་གཡས་རཏྣ་གཡོན་ན་བནྡྷ་བསྣམས། །ཨིནྡྲ་ནཱི་ལ་དར་གྱི་ན་བཟའ་གསོལ། །ཆིབས་སུ་འབྲོང་བུ་ཁམ་པ་རབ་བཅིབས་ནས། །རྒྱས་པའི་ལས་བྱེད་རང་མཐུན་འཁོར་གྱིས་བསྐོར། །བསྟན་པ་བསྲུང་ཕྱིར་གཏོར་མ་འདི་བཞེས་ལ། །ཐུགས་དམ་རྒྱུད་བསྐུལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། ཧྲཱིཿ དབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་དམ་ཚིག་སྒྲོན། །སྐུ་མདོག་དམར་སྐྱ་རིན་ཆེན་འོད་འཕྲོ་ཞིང༌། །དགོས་འདོད་སྣ་ཚོགས་ཆར་བཞིན་འབེབས་མཛད་མ། །ཞལ་གཅིག་ཕྱག་གཉིས་སྤྱན་གསུམ་དགོས་འདོད་སྐོང༌། །ཕྱག་གཡས་ནེའུ་རྐྱལ་གཡོན་ན་ནོར་བུ་འཛིན། །རིན་ཆེན་དབུ་རྒྱན་སྤྱི་གཙུག་ནོར་བུ་དང༌། །ཨིནྡྲ་ནཱི་ལ་ནོར་བུ་སྣ་ཚོགས་མཚོན། །རིན་ཆེན་ཆར་རྒྱུན་སྒྲུབ་པ་པོ་ལ་འབེབས། །ཆིབས་སུ་ཅང་ཤེས་ལ་བཅིབས་སྲིད་གསུམ་ཉུལ། །འཁོར་དུ་དབང་ཕྱུག་ལྷ་མོ་འབུམ་གྱིས་བསྐོར། །དངོས་གྲུབ་གནང་ཕྱིར་འདིར་གཤེགས་གཏོར་མ་བཞེས། །ཐུགས་དམ་རྒྱུད་བསྐུལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། །ཧཱུྂ་༌། དབྱིངས་ཕྱུག་དཔལ་ལྡན་ལྷ་མོ་དབང་གི་སྒྲོན། །ཞལ་གཅིག་ཕྱག་གཉིས་གཡས་པས་གྲི་གུག་འཕྱར། །གཡོན་པས་ཟས་ནོར་སྡུད་པའི་ལྕགས་ཀྱུ་འཛིན། །ཐོད་སྐམ་དབུ་རྒྱན་ཐོད་རློན་དོ་ཤལ་འཕྱང༌། །སྲིད་གསུམ་དབང་སྡུད་ཆིབས་སུ་མཁའ་ལྡིང་བཅིབས། །དབང་གི་ལས་མཛད་རང་མཐུན་འཁོར་གྱིས་བསྐོར། །མི་ཟད་ནོར་དང་རྐང་བཞི་རྨ་གཡང་ཟས། །ཆེ་བཙན་མཐུ་དང་སྣང་སྲིད་གསལ་བའི་པྲ། །སྲིད་སྲུང་རིགས་རྒྱུད་ལ་སོགས་དངོས་གྲུབ་སྩོལ། །འདོད་པའི་དོན་གྲུབ་དབང་བསྡུད་མཁའ་འགྲོ་མ། །དངོས་གྲུབ་གནང་ཕྱིར་གཏོར་མ་འདི་བཞེས་ལ། །ཐུགས་དམ་རྒྱུད་བསྐུལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། །ཧཱུྂ་༌། ཤ་ཟ་ལས་ཀྱི་མཁའ་འགྲོ་ཟ་བྱེད་མ། །སྟོབས་ཆེན་དྲག་མོ་ཤ་ཟ་དུས་ལ་བབ། །སྐུ་མདོག་སྔོ་བསངས་ཁྲག་འཐུང་སྙིང་འབྱིན་མ། །ཞལ་གཅིག་ཕྱག་གཉིས་ཤིན་ཏུ་ཁྲོས་པའི་སྐུ། །ཕྱག་གཡས་རལ་གྲིས་ཉམས་པའི་མགོ་བོ་གཅོད། །གཡོན་པ་ཐོད་ཁྲག་གཙོ་མོའི་ཞལ་དུ་སྟོབ། །གླང་ཆེན་ཞིང་དང་སྟག་ལྤགས་སྐུ་ལ་གསོལ། །དུར་ཁྲོད་ཆས་བརྒྱན་རྒྱ་བོང་གྭ་པ་བཅིབས། །ལས་ཀྱི་མཁའ་འགྲོ་ཤ་ཟ་ཁྲག་འཐུང་མ། །རང་འདྲ་ཤ་ཁྲག་ཟ་བྱེད་འཁོར་གྱིས་བསྐོར། །རྔམ་ཞིང་རྔམ་ཞིང་མྱུར་དུ་ལས་ལ་རྒྱུག

།འདིར་གཤེགས་ཤ་ཁྲག་གཏོར་མ་འདི་བཞེས་ལ། །བསྟན་པ་དྲག་ཏུ་སྲུང་བའི་ཕྲིན་ལས་མཛོད། །བྷྱོཿཉི་མ་ཕྱོགས་ཕྱོགས་ཕ་གི་ན། །ཉི་མ་ནུབ་ཕྱོགས་ཕ་གི་ན། །གངས་དཀར་དགུང་དུ་རྩེ་མཐོ་བ། །མཐོ་ལ་གཉན་ནོ་ཚེ་རིང་མ། །དེ་ལ་བརྟེན་པའི་ལྷ་མོ་ནི། །གཙོ་མོ་བཀྲ་ཤིས་ཚེ་རིང་མ། །འཁོར་ནི་འཇིག་རྟེན་སྨན་མོ་བཞི། །མཐིང་གི་ཞལ་བཟང་མ་དང་ནི། །ཅོད་པན་མགྲིན་བཟང་མ་དང་བཅས། །མི་གཡོ་བློ་བཟང་མ་དང་ནི། །གཏད་ཀར་འགྲོ་བཟང་མ་དང་བཅས། །གཙོ་འཁོར་ཐམས་ཅད་མཚན་ཉིད་མཐུན། །སྐུ་ལ་དར་དཀར་རས་དཀར་གསོལ། །སོ་སོའི་ཕྱག་མཚན་ཕྱག་ན་བསྣམས། །བསྟན་མ་བཅུ་གཉིས་ཁྱོད་ཀྱི་འཁོར། །རྫུ་འཕྲུལ་ཆེན་པོ་མངའ་བས་ན། །བདག་གིས་གནས་འདིར་སྤྱན་འདྲེན་གྱི། །མཆོད་གཏོར་འདི་བཞེས་དངོས་གྲུབ་སྩོལ། །འཆི་མེད་ཚེ་ཡི་དངོས་གྲུབ་སྩོལ། །བང་མཛོད་ནོར་གྱི་དངོས་གྲུབ་སྩོལ། །སྣང་གསལ་པྲ་ཡི་དངོས་གྲུབ་སྩོལ། །རྐང་བཞི་ཕྱུགས་ཀྱི་དངོས་གྲུབ་སྩོལ། །རྨ་གཡང་ཟས་ཀྱི་དངོས་གྲུབ་སྩོལ། །ཆེ་བཙན་མཐུ་ཡི་དངོས་གྲུབ་སྩོལ། །རྗེ་བཙུན་ཆེན་པོའི་སྤྱན་སྔ་རུ། །ཁས་བླངས་བཞིན་དུ་དམ་ལ་འདུས། །བརྒྱན་པའི་གཏོར་མ་འདི་བཞེས་ལ། །བཅོལ་བའི་ཕྲིན་ལས་གྲུབ་པར་མཛོད། །ཧཱུྂ་༌། རྡོ་རྗེ་གདན་གྱི་བྱང་ཕྱོགས་ན། །བལ་བོད་གཉིས་ཀྱི་སོ་མཚམས་སུ། །གངས་དཀར་ཤེལ་གྱི་ཅོད་པན་ཅན། །ཕུ་ཆུ་སྙན་པའི་རྒྱུན་སྒྲ་ཅན། །ནགས་འདབས་མེ་ཏོག་ཤམ་བུ་ཅན། །གཡའ་གངས་གསེར་གཡུའི་སྐྲ་འཆིང་ཅན། །གནས་མཆོག་དེ་འདྲའི་གཡོན་མགུལ་ན། །དར་དཀར་གུར་གྱི་ནང་ཤེད་ན། །སྨན་བཀྲ་ཤིས་ཚེ་ཡི་དབང་ཕྱུག་མ། །མིང་པོའི་གདུང་བས་ཁྱོད་འབོད་ན། །དེ་རིང་སྒྲུབ་པའི་གནས་མཆོག་འདིར། །ལྷབ་ལྷུབ་དར་གྱི་རྒྱན་དང་ལྡན། །སིལ་སྙན་རོལ་མོའི་སྒྲ་དང་བཅས། །ནམ་མཁའི་མཐོངས་ནས་སྤྱན་འདྲེན་ན། །འཁོར་དང་བཅས་ཏེ་གཤེགས་སུ་གསོལ། །ཁྱོད་ཀྱི་ཆོ་དང་འབྲང་སྨོས་པ། །ཡབ་ནི་ལྷོ་ལྷ་མཐིང་པོ་ཡིན། །ཡུམ་ནི་འབྲང་ཡུམ་སྨན་ཅིག་མ། །མིང་པོ་བདུན་གྱི་སྲིང་མོ་ལྔ། །ལྔ་ཚིགས་དབུས་ཀྱི་གཙོ་མོ་སྟེ། །ཞལ་བཟང་ཟླ་བ་ཉ་བ་འདྲ། །ལྷ་མོ་རྣམས་ཀྱང་ཟིལ་གྱིས་གནོན། །མགྲིན་པ་གཅིག་ལ་གླུ་སྣ་སྟོང༌། །ཚངས་པའི་དབྱངས་ཀྱང་ཟིལ་གྱིས་གནོན། །དབུ་སྐྲ་ཉག་མ་རེ་རེ་ལ། །གཡུ་ཕྲན་སྟོང་སྟོང་ལན་ཚར་ཅན། །ན་བཟའ་དར་དཀར་གོས་ཟུང་གསོལ། །ཉིན་གསུམ་གངས་ཀྱི་རྩེ་ལ་བཞུགས། །སྨན་མོ་རྣམས་དང་ལྷན་ཅིག་རོལ། །མཚན་གསུམ་རྒྱ་གར་ཡུལ་ན་བཞུགས། །རབ་ཏུ་འཇིགས་པའི་དུར་ཁྲོད་དུ། །མཁའ་འགྲོ་རྣམས་དང་ཚོགས་འཁོར་རོལ། །བཀའ་ཉན་སྤྲུལ་པའི་ཕོ་ཉ་ནི། །མ་ལུས་འཁོར་དང་བཅས་པ་རྣམས། །སྔོན་གྱི་ཐུགས་དམ་དགོངས་མཛོད་ལ། །དེ་རིང་གནས་འདིར་གཤེགས་སུ་གསོལ། །ཁྱོད་ལ་པུ་ཙ་འབུལ་བ་ནི། །བདུད་རྩི་སྨན་གྱི་རྩི་བཅུད་དང༌། །མེ་ཏོག་ཡིད་འོང་མཛེས་པའི་ཚལ། །སྤོས་དྲི་ཕུན་སུམ་ཚོགས་པ་དང༌། །འོད་དང་དྲི་མཆོག་བསང་བཤོས་དང༌། །སྒྲ་སྙན་གསང་བའི་རྔ་ཆུང་རྣམས། །ཐུགས་རྗེས་བརྩེར་དགོངས་བཞེས་ནས་ཀྱང༌། །བྱིན་རླབས་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །སྔོན་ཚེ་དབེན་གནས་ཡང་རྫོང་དུ། །དཔལ་ལྡན་བཞད་པ་རྡོ་རྗེ་ལ། །རང་རྟགས་སྲོག་གི་སྙིང་པོ་ཕུལ། །བཀའ་ཉན་བགྱིད་པར་རྩིས་ལ་ཕབ། །བུ་རྒྱུད་སྤེལ་བར་གཡར་དམ་བཅས། །དེ་རིང་སྨན་བཙུན་རྒྱལ་མོ་ལ། །མིང་པོས་མཆོད་བསྟོད་འདི་ཕུལ་བས། །སྲིད་སྲུང་རིགས་རྒྱུད་དངོས་གྲུབ་དང༌། །རྨ་གཡང་ཟས་ཀྱི་དངོས་གྲུབ་དང༌། །གཞན་ཡང་ཕྱི་ནང་གསང་བ་ཡི། །དངོས་གྲུབ་ཐམས་ཅད་བདག་ལ་སྩོལ། །གཞན་ཡང་གསོལ་བ་འདེབས་པ་ནི། །རྣལ་འབྱོར་ཚེ་སྲིངས་ལུས་མདངས་བསྐྱེད།

ནང་གི་ལུས་སེམས་དབང་དུ་བསྒྱུར། །ཕྱི་ཡི་སྣང་བ་དབང་དུ་བགྱིས། །ཕྲིན་ལས་སྟོབས་ཀྱིས་བསྟན་པ་སྤེལ། །ཚེ་སྲིངས་དཔལ་བསྐྱེད་འཁོར་རྣམས་སྐྱོངས། །བསྟན་ལ་སྡང་བའི་དགྲ་བོ་རྣམས། །གཏོར་མ་འདི་དང་མཉམ་དུ་བཞེས། །བཅོལ་བའི་ལས་ཀྱི་འབྲས་བུ་དེ། །མཆིས་པར་མཛོད་ཅིག་མཐུ་རྩལ་ཅན། ། ཞེས་པས་བསྟོད། དེ་ནས་གཏོར་འབུལ་གྱི་བསྐུལ་འདི་སྐད་བརྗོད། བྷྱོཿ ནམ་མཁའ་མཐིང་བཟང་གཡུ་འོད་འབར་བའི་ཀློང༌། །མུ་མེན་མཐིང་གི་གཡུ་བྲག་ཏི་སྒྲོ་ནས། །ཆེ་ཞེས་ཆེན་མོ་ཨ་ཕྱི་སྣ་ནམ་བཟའ། །གདུག་པ་འདུལ་ཕྱིར་གནས་འདིར་གཤེགས་སུ་གསོལ། །དབྱིངས་ཕྱུག་ཡེ་ཤེས་མཁའ་འགྲོའི་ཚོགས་དང་བཅས། །རྡོ་རྗེ་རིགས་ཀྱི་ལས་བྱེད་མཁའ་འགྲོ་དང༌། །རིན་ཆེན་རིགས་ཀྱི་ལས་བྱེད་མཁའ་འགྲོ་དང༌། །ཕྲིན་ལས་རིགས་ཀྱི་ལས་བྱེད་མཁའ་འགྲོ་དང༌། །ལས་ཀྱི་མཁའ་འགྲོ་ཟ་བྱེད་འབུམ་གྱིས་བསྐོར། །གདུག་པ་སྒྲོལ་བ་བཤན་པ་དམར་མོའི་ཚོགས། །ལྕེ་སྤྱང་སྔོན་མོར་རྫུ་ཞིང་ཕྱི་ནས་འབྲང༌། །དམ་ཉམས་དགྲ་སྲོག་གཅོད་ཕྱིར་གནས་འདིར་བྱོན། །ཤ་ཁྲག་བརྒྱན་པའི་གཏོར་མ་འདི་བཞེས་ལ། །མ་བུའི་དམ་ཚིག་བཞིན་དུ་དགྲ་བོ་སྒྲོལ། །ཡིད་ལ་ངན་སེམས་འཆང་བའི་དགྲ་བོ་འདི། །རིངས་པར་ཁུག་ལ་མྱུར་དུ་སྒྲོལ་བར་གྱིས། །སོད་རོལ་ཆིངས་གཏུབས་སྒྲོལ་ཅིག་ས་མ་ཡ། །མ་གཡེལ་མ་འདའ་ཤ་ཟ་མཁའ་འགྲོའི་ཚོགས། །བདུད་མོ་ལས་བྱེད་མཁའ་འགྲོའི་ཚོགས་རྣམས་ཀྱིས། །དགྲ་ལ་ཐོག་དང་ནད་ཡམས་དམར་ནད་ཕོབ། །ནམ་མཁའི་ལྷ་དང་སྡོངས་ལ་གཟུགས་ཕུང་ཕྲོལ། །སེམས་ཅན་ལོག་པར་བལྟ་བ་སྒྲོལ་བར་གྱིས། །ས་གཞི་ཀླུ་དང་སྡོངས་ལས་གཟུགས་ཕུང་ཕྲོལ། །སྐྱེ་མཆེད་ཁམས་རྣམས་ཐལ་བའི་རྡུལ་དུ་རློག །བར་སྣང་གཉན་དང་སྡོངས་ལ་གཟུགས་ཕུང་ཕྲོལ། །སེམས་ཅན་ངན་སེམས་ཅན་རྣམས་ཕུང་བར་གྱིས། །སོད་ཐུལ་ཆིངས་ཤིག་དགྲ་དང་བགེགས་ཀྱི་ཚོགས། །དམར་གྱི་གཏོར་མ་འདི་དང་མཉམ་དུ་བཞེས། །བཅོལ་བའི་ཕྲིན་ལས་གྲུབ་པར་མཛོད། །

ཅེས་བསྐུལ་ཞིང་སྨོད་པ་བོར། དམ་བསྒྲགས་ལ་གཏོར་མ་གཙང་སར་བསྐྱལ། དེ་ནས་ལྷག་ཆད་བསྐང་བའི་མཆོད་པ་ནི། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་ཨརྒྷཾ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་པུཥྤེ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་དྷཱུ་པེ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་ཨཱ་ལོ་ཀེ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་གནྡྷེ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་ནཻ་ཝི་དྱ་ཨཱཿཧཱུྂ། ཨོཾ་གྷ་མུ་ཨ་ཛི་ཏ་ཤྭ་རཱི་དྷརྨ་ཨོཾ་བཛྲ་ཤཔད་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྥ་ར་ཎ་ས་མ་ཡེ་ཧཱུྂ། ཕྱི་ནང་གསང་གསུམ་མཆོད་པའི་བྱེ་བྲག གཟུགས་མཛེས་པ། སྒྲ་སྙན་པ། དྲི་ཞིམ་པ། རོ་མངར་བ། རེག་བྱ་འཇམ་པ་ཐམས་ཅད་བླ་མ་ཡི་དམ་དཀྱིལ་འཁོར་གྱི་ལྷ་ཚོགས། དཔའ་བོ་མཁའ་འགྲོའི་ཚོགས་དང་བཅས་པ་རྣམས་ལ་འབུལ་ལོ། །མཆོད་དོ། །ཐུགས་དམ་བསྐང་ཕྱིར་བཞེས་སུ་གསོལ་ལོ། །ལྷག་ཆད་བསྐང་ཕྱིར་བཞེས་སུ་གསོལ། །གང་ཡང་བདག་བློ་གཏི་མུག་པས། །ལེགས་བྱས་འདི་ལ་ཉེས་གྱུར་གང༌། །ལུས་ཅན་ཀུན་གྱི་སྐྱོན་གྱུར་པ། །དེ་ཀུན་མཁའ་འགྲོས་བཟོད་པར་གསོལ། །གཙོ་བོ་ནུས་པ་མ་མཆིས་ཤིང༌། །མ་མཉེས་པ་དང་མ་ཚང་བ། །འདིར་ནི་བྱ་བ་གང་བགྱིས་པ། །དེ་ཀུན་མཁའ་འགྲོས་བཟོད་པར་གསོལ། །དེ་ནས་དམ་ཚིག་པ་རང་ལ་བསྟིམ། །ཡེ་ཤེས་པ་རང་བཞིན་གྱི་གནས་སུ་གཤེགས། །

དེ་ནས་བཀྲ་ཤིས་བརྗོད་པ་ནི། ཞིང་ཡོངས་སུ་དག་པའི་ཕོ་བྲང་ནས། །དྲུག་པ་རྡོ་རྗེ་འཆང་ཆེན་དང༌། །རྗེ་བཙུན་ཏིལླི་ནཱ་རོ་དང༌། །བདག་གི་བླ་མ་ཡན་ཆོད་དུ། །གྲུབ་ཐོབ་ཀྱི་ཆོས་བརྒྱུད་མ་ཆད་པའི། །བརྒྱུད་པའི་བླ་མ་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །འོག་མིན་གནས་མཆོག་འབྲི་གུང་ནས། །རྗེ་རྩ་བའི་བླ་མ་རིན་པོ་ཆེ། །སྐུ་མཚན་དང་དཔེ་བྱད་དཔལ་འབར་ཞིང༌། །གསུང་ཐེག་ཆེན་ཆོས་ཀྱི་ང་རོ་སྒྲོག །ཐུགས་རྟོགས་པ་མི་བསྐྱོད་རྡོ་རྗེ་འཆང༌། །ཕ་སངས་རྒྱས་དེ་ཡི་བཀྲ་ཤིས་ཤོག །ཐོག་མ་བར་དང་མཐར་དགེ་ཞིང༌། །ཟབ་ཞི་སྤྲོས་བྲལ་འདུས་མ་བྱས། །བདུད་རྩི་ལྟ་བུའི་རིན་པོ་ཆེ། །དམ་ཆོས་དེ་ཡི་བཀྲ་ཤིས་ཤོག །བསླབ་པ་གསུམ་གྱིས་རྒྱུད་སྦྱངས་ཤིང༌། །མོས་གུས་མཐར་ཕྱིན་རྟོགས་པར་ལྡན། །སེམས་ཅན་དོན་མཛད་ཐུགས་རྗེ་ཅན། །དགེ་འདུན་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །ཡི་དམ་ལྷ་ཚོགས་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །དཔའ་བོ་མཁའ་འགྲོ་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །ཆོས་སྐྱོང་སྲུང་མ་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །ནོར་ལྷ་གཏེར་བདག་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །མི་འགྱུར་ལྷུན་པོ་སྐུ་ཡི་བཀྲ་ཤིས་ཤོག །ཡན་ལག་དྲུག་ཅུ་གསུང་གི་བཀྲ་ཤིས་ཤོག །མཐའ་བྲལ་དྲི་མེད་ཐུགས་ཀྱི་བཀྲ་ཤིས་ཤོག །རྒྱལ་བའི་སྐུ་གསུང་ཐུགས་ཀྱི་བཀྲ་ཤིས་ཤོག །སྔོན་བཅོམ་ལྡན་འདས་ཀྱི་སྤྱན་སྔ་རུ། །ཆོས་སྒྲུབ་པར་བྱེད་པའི་གང་ཟག་ལ། །བུ་བཞིན་སྐྱོང་བར་ཞལ་བཞེས་པའི། །ཆོས་སྐྱོང་སྲུང་མ་རྣམས་ཀྱི་བཀྲ་ཤིས་ཤོག །ཉིན་མོ་བདེ་ལེགས་མཚན་བདེ་ལེགས། །ཉི་མའི་གུང་ཡང་བདེ་ལེགས་ཤིང༌། །ཉིན་མཚན་རྟག་ཏུ་བདེ་ལེགས་པའི། །དཀོན་མཆོག་གསུམ་གྱི་དེང་འདིར་བདེ་ལེགས་ཤོག ། ཅེས་བརྗོད། དགེ་བའི་རྩ་བ་ཐམས་ཅད་ནམ་མཁའ་དང་མཉམ་པའི་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་བསྔོ། ཇི་ལྟར་བསམ་པའི་དོན་གྲུབ་ཅིང༌། འགྲོ་ཀུན་ཐམས་ཅད་མཁྱེན་པ་ཐོབ་པར་ཤོག ། ཨ་ཕྱིའི་སྒྲུབ་ཐབས་རིན་པོ་ཆེའི་ཅོད་པན་གྱི་ཕྱག་ལེན།

བླ་མ་གོང་མ་རྣམས་ཀྱི་ཕྱག་བཞེས་ཇི་ལྟར་མཛད་པའི་ལུགས་བཞིན། མ་ནོར་བར་བསྒྲིགས་པ་ནོར་བུ་བསམ་འཕེལ་ཞེས་བྱ་བ་འདི་ནི། གདན་ས་ཆེན་པོ་འབྲི་གུང་དུ་ཤཱཀྱའི་དགེ་སློང་ཨཱ་ནནྡ་རཏྣ་དྷརྨདྷྭ་ཛཤྲཱི་བྷ་དྲས་སྦྱར་བ་དགེ་བར་གྱུར་ཅིག། །།མངྒལཾ།།
